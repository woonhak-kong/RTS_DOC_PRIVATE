135 private void Place(bool completeConstruction, bool playerCommand)
136 {
137 // Hide the selection marker since it was used to display whether the building can be placed or not.
138 SelectionMarker?.Disable();
139
140 CompleteInit();
141 PlacerComponent.InitPlaced(placedData);
142
143 globalEvent.RaiseBuildingPlacedGlobal(this);
144
145 if (completeConstruction)
146 CompleteConstruction();
147
148 if (IsFree) //free builidng? job is done here
149 return;
150
151 if (!RTSHelper.IsLocalPlayerFaction(this))
152 return;
153
154 // Since the *SetTarget* sends an input command and would not return whether the target would be set or not.
155 // We determine the amount of missing workers and see what builder units from the ones selected can fulfill these positions
156 // And if the missing workers amount can be filled from the selected units, we stop looking to make it more efficient
157 int missingWorkers = WorkerMgr.MaxAmount - WorkerMgr.Amount;
158 foreach (IUnit unit in selectionMgr.GetEntitiesList(EntityType.unit, exclusiveType: false, localPlayerFaction: true))
159 {
160 if (missingWorkers <= 0)
161 break;
162 if (unit.BuilderComponent.IsValid()
163 && unit.BuilderComponent.IsTargetValid(this.ToSetTargetInputData(playerCommand :false)) == ErrorMessage.none)
164 {
165 if(unit.TasksQueue.IsValid() && taskMgr.IsTaskQueueEnabled)
166 {
167 unit.TasksQueue.Add(new SetTargetInputData
168 {
169 componentCode = unit.BuilderComponent.Code,
170
171 target = this,
172 playerCommand = playerCommand,
173 });
174 }
175 else
176 unit.BuilderComponent.SetTarget(this, playerCommand);
177
178 missingWorkers--;
179 }
180 }
181 }

183 private void CompleteConstruction()
184 {
185 Health.EntityHealthUpdated -= HandleBuildingHealthUpdated;
186
187 if (IsBuilt)
188 return;
189
190 // If this is not a placement instance but it has not been placed (initialized fully)
191 // Place it before it gets completely constructed.
192 if (!IsPlacementInstance && !IsInitialized)
193 {
194 Place(completeConstruction: true, playerCommand: false);
195 // return here because Place() will call CompleteConsutrction() after setting IsInitialized to true
196 return;
197 }
198
199 bool wasSelectedOnly = selectionMgr.IsSelectedOnly(this);
200 if (wasSelectedOnly)
201 selectionMgr.Remove(this);
202
203 IsBuilt = true;
204
205 // To reload the single selection UI handlers post construction that listen to selection events
206 if (wasSelectedOnly)
207 selectionMgr.Add(this, SelectionType.single, isLocalPlayerClickSelection: false);
208
209 if (IsFree)
210 {
211 TriggerConstructionCompleteEvents();
212 return;
213 }
214
215 if (BorderComponent.IsValid())
216 {
217 BorderComponent.Init(gameMgr, this);
218 CurrentCenter = BorderComponent;
219 }
220
221 if(canGiveInitResources)
222 resourceMgr.UpdateResource(FactionID, InitResources, add:true);
223
224 TriggerConstructionCompleteEvents();
225 }
