106 private void ActivateCollectorRegulator(ResourceTypeInfo resourceType, NPCActiveRegulatorMonitor collectorMonitor)
107 {
108 foreach (GameObject unitObj in collectors)
109 {
110 IUnit collectorUnit = unitObj?.GetComponent<IUnit>();
111 if (!logger.RequireValid(collectorUnit,
112 $"[{GetType().Name} - {factionMgr.FactionID}] 'Collectors' field has some unassigned elements."))
113 return;
114
115 IResourceCollector collectorComponent = collectorUnit.gameObject.GetComponentInChildren<IResourceCollector>();
116 if (!logger.RequireValid(collectorComponent,
117 $"[{GetType().Name} - {factionMgr.FactionID}] 'Collectors' field has some assigned units with no component that implements '{typeof(IResourceCollector).Name}' interface attached to them."))
118 return;
119
120 NPCUnitRegulator nextRegulator;
121 if (collectorComponent.IsResourceTypeCollectable(resourceType)
122 && (nextRegulator = npcUnitCreator.ActivateUnitRegulator(collectorUnit)).IsValid())
123 collectorMonitor.AddCode(nextRegulator.Prefab.Code);
124 }
125
126 if (collectorMonitor.Count == 0)
127 {
128 logger.LogWarning($"[{GetType().Name} - {factionMgr.FactionID}] No resource collector regulators have been assigned for resource type '{resourceType.Key}'!");
129 IsActive = false;
130 }

177 #region Handling Resource Collection
178 protected override void OnActiveUpdate()
179 {
180 if (collectionTimer.ModifiedDecrease())
181 {
182 collectionTimer.Reload(collectionTimerRange);
183
184 collectionTimerTicks++;
185
186 bool canEnforceMinCollectors = collectionTimerTicks == enforceMinCollectorsTimerTicks;
187 if (canEnforceMinCollectors)
188 collectionTimerTicks = 0;
189
190 // Assume that all resource instances have the required amount of collectors and deactivate this component
191 IsActive = false;
192
193 //go through the ResourceTypeCollection instances that regulate each resource type
194 foreach (NPCResourceTypeCollectionTracker nextCollectionTracker in collectionTrackers.Values)
195 {
196 int collectorCount = 0;
197 for (int i = 0; i < nextCollectionTracker.CollectorMonitor.AllCodes.Count; i++)
198 {
199 NPCUnitRegulator collectorRegulator = npcUnitCreator.GetActiveUnitRegulator(nextCollectionTracker.CollectorMonitor.AllCodes[i]);
200 if (!collectorRegulator.IsValid())
201 collectorCount += collectorRegulator.Count;
202 }
203
204 if (!nextCollectionTracker.CanAddCollector(collectorCount))
205 continue;
206
207 // Set state back to active so we can keep monitoring whether resource collectors have been correctly assigned or not.
208 IsActive = true;
209
210 foreach (IResource resource in nextCollectionTracker.ResourceInstances)
211 {
212 int targetCollectorsAmount = nextCollectionTracker.GetTargetCollectorsAmount(resource);
213
214 if (targetCollectorsAmount > resource.WorkerMgr.Amount)
215 OnResourceCollectionRequestInternal(
216 resource,
217 targetCollectorsAmount,
218 out _,
219 forceSwitch: canEnforceMinCollectors && !nextCollectionTracker.IsMinAmountReached);
220 }
221 }
222 }
223 }

238 private void OnResourceCollectionRequestInternal(IResource resource, int targetCollectorsAmount, out int assignedCollectors, bool forceSwitch = false)
239 {
240 assignedCollectors = 0;
241 if (!resource.IsValid())
242 return;
243 if (resource.Health.IsDead
244 || !collectionTrackers.ContainsKey(resource.ResourceType))
245 {
246 LogEvent($"{resource.Code}: Unable to collect resource! Resource empty? {resource.Health.IsDead}");
247 return;
248 }
249
250 int requiredCollectors = targetCollectorsAmount - resource.WorkerMgr.Amount;
251
252 IReadOnlyList<IUnit> currentCollectors = npcUnitCreator
253 .GetActiveUnitRegulator(collectionTrackers[resource.ResourceType].CollectorMonitor.RandomCode)
254 .InstancesIdleFirst;
255
256 foreach (IUnit nextCollector in currentCollectors)
257 {
258 if (requiredCollectors <= 0)
259 break;
260
261 bool canStillForceSwitch = forceSwitch
262 && nextCollector.CollectorComponent.HasTarget
263 && (collectionTrackers[nextCollector.CollectorComponent.Target.instance.ResourceType].CollectorsAmount - 1 >= collectionTrackers[nextCollector.CollectorComponent.Target.instance.ResourceType].CollectionData.minCollectorsAmount);
264
265 if (nextCollector.IsValid()
266 && (nextCollector.IsIdle || canStillForceSwitch)
267 && nextCollector.CollectorComponent.Target.instance != resource)
268 {
269 nextCollector.CollectorComponent.SetTarget(RTSHelper.ToTargetData(resource), playerCommand: false);
270
271 requiredCollectors--;
272 assignedCollectors++;
273 }
274 }
275
276 LogEvent($"{resource.Code}: Sent {assignedCollectors}/{assignedCollectors+requiredCollectors} collectors to gather resource!");
277 }
