155 public void SetTargetFaction()
156 {
157 // Order the potential target factions via the attack resources assigned for this component
158 IFactionSlot[] activeFactions = gameMgr.ActiveFactionSlots
159 .OrderByDescending(faction => attackResources.Sum(resourceType => resourceMgr.FactionResources[faction.ID].ResourceHandlers[resourceType].Amount))
160 .ToArray();
161
162 if (activeFactions.Length == 0)
163 return;
164
165 // Cancel current attack (if there is one) and start new one.
166 CancelAttack();
167
168 // Depending on the strategy of this component in relation to the attack resources, pick one
169 switch (targetFactionType)
170 {
171 case NPCAttackTargetPickerType.random:
172 targetFactionSlot = activeFactions[UnityEngine.Random.Range(0, activeFactions.Length)];
173 break;
174
175 case NPCAttackTargetPickerType.mostAttackResources:
176 targetFactionSlot = activeFactions[0];
177 break;
178
179 case NPCAttackTargetPickerType.leastAttackResources:
180 targetFactionSlot = activeFactions[activeFactions.Length - 1];
181 break;
182 }
183
184 LogEvent($"New target faction ID: {targetFactionSlot.ID}!");
185 targetFactionSlot.FactionSlotStateUpdated += HandleTargetFactionSlotStateUpdated;
186 }

191 public bool SetTargetFaction(IFactionSlot newTargetFactionSlot, bool launchAttack)

220 private void OnInactiveAttackUpdate()
221 {
222 // Can not launch a new attack if the NPC faction is already in a defensive mode
223 if (npcDefenseMgr.IsDefending)
224 return;
225
226 if (launchAttackTimer.ModifiedDecrease())
227 {
228 launchAttackTimer.Reload(launchAttackReloadRange);

241 public bool LaunchAttack()
242 {
243 if (!targetFactionSlot.IsValid()
244 || gameMgr.InPeaceTime)
245 return false;
246
247 LogEvent($"Launching attack on target faction ID {targetFactionSlot.ID}!");
248 IsAttacking = true;
249
250 // Start with the faction's spawn position as the last position to search from
251 lastAttackPos = gameMgr.GetFactionSlot(factionMgr.FactionID).FactionSpawnPosition;
252
253 // Set initial target faction entity
254 currentTargetEntity = null;
255 SetTargetEntity(targetFactionSlot.FactionMgr.Buildings, true);
256
257 // Start the attack order timer to handle attack commands in the active attack engagement
258 attackOrderTimer = new TimeModifiedTimer(attackOrderReloadRange);
259
260 return true;
261 }

268 public bool SetTargetEntity(IEnumerable<IFactionEntity> factionEntities, bool resetCurrentTarget)
269 {
270 if (resetCurrentTarget)
271 ResetCurrentTarget();
272
273 float distance = Mathf.Infinity;
274 IFactionEntity nextTarget = null;
275 float priorityDistance = Mathf.Infinity;
276 IFactionEntity nextPriorityTarget = null;
277 foreach(IFactionEntity nextEntity in factionEntities)
278 {
279 if(targetPicker.IsValidTarget(nextEntity))
280 {
281 if(priorityDistance > Vector3.Distance(nextEntity.transform.position, lastAttackPos))
282 {
283 priorityDistance = Vector3.Distance(nextEntity.transform.position, lastAttackPos);
284 nextPriorityTarget = nextEntity;
285 }
286 }
287 else
288 {
289 if(distance > Vector3.Distance(nextEntity.transform.position, lastAttackPos))
290 {
291 distance = Vector3.Distance(nextEntity.transform.position, lastAttackPos);
292 nextTarget = nextEntity;
293 }
294 }
295 }
296
297 currentTargetEntity = nextPriorityTarget.IsValid() ? nextPriorityTarget : nextTarget;
298 if (currentTargetEntity.IsValid())
299 {
300 lastAttackPos = currentTargetEntity.transform.position;
301 LogEvent($"Active attack target entity updated to {currentTargetEntity.Code}");
302 return true;
303 }
304
305 return false;
306 }

346 #region Handling Active Attack
347 private void OnActiveAttackUpdate()
348 {
349 if (!attackOrderTimer.ModifiedDecrease())
350 return;
351
352 attackOrderTimer.Reload(attackOrderReloadRange);
353
354 // Did the current attack power hit the surrender resource type amount?
355 if (!resourceMgr.HasResources(cancelAttackResources, factionMgr.FactionID))
356 {
357 CancelAttack();
358 return;
359 }
360
361 IEnumerable<IFactionEntity> primaryTargetSet;
362 IEnumerable<IFactionEntity> secondaryTargetSet;
363
364 bool randPickBuildingsPrimary = targetPreference == NPCAttackFactionEntityPreference.random
365 && UnityEngine.Random.value > 0.5f;
366
367 if (targetPreference == NPCAttackFactionEntityPreference.buildings
368 || randPickBuildingsPrimary)
369 {
370 primaryTargetSet = targetFactionSlot.FactionMgr.Buildings;
371 secondaryTargetSet = targetFactionSlot.FactionMgr.Units;
372 }
373 else
374 {
375 secondaryTargetSet = targetFactionSlot.FactionMgr.Buildings;
376 primaryTargetSet = targetFactionSlot.FactionMgr.Units;
377 }
378
379 // Update current target faction entity
380 if (!currentTargetEntity.IsValid()
381 && SetTargetEntity(primaryTargetSet, resetCurrentTarget: false) == false)
382 SetTargetEntity(secondaryTargetSet, resetCurrentTarget: false);
383
384 EngageCurrentTargetFactionEntity();
385 }

390 public void CancelAttack()
391 {
392 if(IsAttacking)
393 npcEvent.RaiseAttackCancelled(this);
394
395 currentTargetEntity = null;
396
397 if (targetFactionSlot.IsValid())
398 {
399 LogEvent($"Cancelled attack on target faction ID: {targetFactionSlot.ID}!");
400 targetFactionSlot.FactionSlotStateUpdated -= HandleTargetFactionSlotStateUpdated;
401 }
402
403 targetFactionSlot = null;
404
405 IsAttacking = false;
406 }

408 private bool EngageCurrentTargetFactionEntity()
409 {
410 if (!IsAttacking
411 || !targetFactionSlot.IsValid()
412 || gameMgr.InPeaceTime)
413 return false;
414
415 if (!currentTargetEntity.IsValid()
416 || currentTargetEntity.Health.IsDead)
417 {
418 ResetCurrentTarget();
419 return false;
420 }
421
422 IBuilding targetBuilding = currentTargetEntity.IsBuilding() ? currentTargetEntity as IBuilding : null;
423 // If the current target is a building and it is being constructed, then assign an active builder (if it exists) as the current target.
424 if (targetBuilding.IsValid() && targetBuilding.WorkerMgr.Amount > 0)
425 {
426 int i = 0;
427 while(i < targetBuilding.WorkerMgr.Workers.Count)
428 {
429 if(targetBuilding.WorkerMgr.Workers[i].BuilderComponent.InProgress)
430 {
431 currentTargetEntity = targetBuilding.WorkerMgr.Workers[i];
432 LogEvent($"Active attack target entity updated to builder {currentTargetEntity.Code}");
433 break;
434 }
435 i++;
436 }
437 }
438
439 npcEvent.RaiseAttackEngageOrder(this, new NPCAttackEngageEventArgs(currentTargetEntity, lastAttackPos));
440
441 return true;
442 }
