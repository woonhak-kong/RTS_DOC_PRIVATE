133 private void HandleCameraPositionUpdated(IMainCameraController source, EventArgs args)
134 {
135 Vector3 screenCenterWorldPos = mainCameraController.ScreenToWorldPoint(new Vector3(Screen.width / 2, Screen.height / 2, 0.0f), applyOffset: false);

137 // Convert the main camera's position to the minimap camera's screen space

156 public bool IsMouseOverMinimap(Vector2 viewportPoint, out Vector3 terrainHitPosition)
157 {
158 terrainHitPosition = Vector3.zero;
159
160 //if (minimapCamera.rect.Contains(mainCameraController.ScreenToViewportPoint(Input.mousePosition))
161
162 // Draw a ray using the current mouse position towards the minimap camera and see if it hits the terrain
163 if (minimapCamera.rect.Contains(viewportPoint)
164 && Physics.Raycast(minimapCamera.ViewportPointToRay(viewportPoint), out RaycastHit hit, Mathf.Infinity, terrainLayerMask))
165 {
166 terrainHitPosition = hit.point;
167 return true;
168 }
169
170 return false;
171 }

173 private IEnumerator TakeSnapshot()
174 {
175 yield return new WaitForEndOfFrame();
176
177 minimapCamera.enabled = true;
178
179 RenderTexture texture = MinimapCamera.targetTexture;
180
181 if (!logger.RequireValid(texture,
182 $"[{GetType().Name}] Test subject is not valid!"))
183 yield break;
184
185 minimapCamera.Render();
186
187 minimapCamera.enabled = false;
188 }
189 private void Update()
190 {
191 // Only allow to interact with the minimap camera UI if movement is allowed and no other service is reserving the UI priority.
192 if (!movementEnabled)
193 return;
194
195 bool leftClickEvent = Input.GetMouseButtonDown(0) || (dragMovementEnabled && Input.GetMouseButton(0));
196 bool rightClickEvent = selectedUnitsMovementEnabled && Input.GetMouseButtonDown(1);
197
198 if (!leftClickEvent && !rightClickEvent)
199 return;
200
201 if (!minimapCameraHandler.TryGetMinimapViewportPoint(out Vector2 viewportPoint))
202 return;
203
204 if (IsMouseOverMinimap(viewportPoint, out Vector3 terrainHitPosition))
205 {
206 if (leftClickEvent)
207 OnLeftMouseClick(terrainHitPosition);
208 else
209 OnRightMouseClick(terrainHitPosition);
210 }
211 }

227
228 if (selectedUnits.Count > 0)
229 mvtMgr.SetPathDestination(

232 source = selectedUnits,
233 destination = hitPoint,
234 offsetRadius = 0.0f,
235 target = null,

237 });
