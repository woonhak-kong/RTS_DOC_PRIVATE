224 public ErrorMessage CanMove(IUnit unit, AddableUnitData addableData = default)
225 {
226 if (!Entity.CanLaunchTask)
227 return ErrorMessage.taskSourceCanNotLaunch;
228
229 else if (!unit.IsValid())
230 return ErrorMessage.invalid;
231 else if (!unit.IsInteractable)
232 return ErrorMessage.uninteractable;
233 else if (unit.Health.IsDead)
234 return ErrorMessage.healthDead;
235 else if (!unit.CarriableUnit.IsValid())
236 return ErrorMessage.carriableComponentMissing;
237 else if (!addableData.ignoreMvt && !unit.CanMove())
238 return ErrorMessage.mvtDisabled;

242 return ErrorMessage.factionMismatch;
243
244 else if (!targetPicker.IsValidTarget(unit))
245 return ErrorMessage.targetPickerUndefined;
246
247 else if (CurrAmount + customUnitSlots.GetAmount(unit) > MaxAmount)
248 return ErrorMessage.carrierCapacityReached;
249
250 else if (addableData.forceSlot && !freeCarrierPositionIndexes.Contains(addableData.forcedSlotID))
251 return ErrorMessage.carrierForceSlotOccupied;

253 return ErrorMessage.none;
