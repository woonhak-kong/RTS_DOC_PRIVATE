166 public void OnEntityPreInit(IGameManager gameMgr, IEntity entity)
167 {
168 this.gameMgr = gameMgr;
169 this.logger = gameMgr.GetService<IGameLoggingService>();
170 this.Entity = entity;
171
172 // In case of entity conversion, there is an attempt to re-initialize this component but we do not allow it.
173 if (!logger.RequireTrue(!IsInitialized,
174 $"[{GetType().Name} - {Entity.Code}] Component already initialized! It is not supposed to be initialized again! Please retrace and report!"))
175 return;
176
177 this.inputMgr = gameMgr.GetService<IInputManager>();
178 this.globalEvent = gameMgr.GetService<IGlobalEventPublisher>();
179 this.selectionMgr = gameMgr.GetService<ISelectionManager>();
180 this.effectObjPool = gameMgr.GetService<IEffectObjectPool>();
181 this.audioMgr = gameMgr.GetService<IGameAudioManager>();
182 this.resourceMgr = gameMgr.GetService<IResourceManager>();
183 this.playerMsgHandler = gameMgr.GetService<IPlayerMessageHandler>();
184
185 if (!logger.RequireTrue(maxHealth > 0,
186 $"[{GetType().Name} - {entity.Code}] 'Max Health' field value must be > 0!"))
187 return;
188
189 // Each health component must serve one entity type
190 // In case of a building resource for example, two health components for each entity type must be present
191 // Check whether the health component is currently serving one entity type only
192 if (!((int)EntityType == 1 << 0 || (int)EntityType == 1 << 1 || (int)EntityType == 1 << 2))
193 {
194 logger.LogError("[EntityHealth] The health component must serve only entity type and not multiple ones!", source: this);
195 return;
196 }
197
198 hitEffectSpawnInput = new EffectObjectSpawnInput(
199 parent: Entity.transform,
200
201 useLocalTransform: true,
202 spawnPosition: Vector3.zero,
203 spawnRotation: Quaternion.identity);
204
205 IsDead = false;
206
207 stateHandler = new EntityHealthStateHandler();
208 stateHandler.Init(gameMgr, this, states.Count);
209
210 CurrHealth = 0;
211
212 OnEntityHealthInit();
213
214 IsInitialized = true;
215
216 Entity.EntityInitiated += HandleEntityInitialized;
217 }

286 public ErrorMessage AddLocal(HealthUpdateArgs args, bool force = false)
287 {
288 if (!force)
289 {
290 ErrorMessage errorMessage;
291 if ((errorMessage = CanAdd(args)) != ErrorMessage.none)
292 return errorMessage;
293 }
294
295 if (force || !LockHealth)
296 {
297 if (CurrHealth + args.Value < 0)
298 args = new HealthUpdateArgs(-CurrHealth, args.Source);
299
300 CurrHealth += args.Value;
301 CurrHealth = Mathf.Clamp(CurrHealth, 0, MaxHealth);
302 }
303
304 stateHandler.Update(args.Value > 0, CurrHealth);
305
306 OnHealthUpdated(args);
307
308 RaiseEntityHealthUpdated(args);
309
310 if (CurrHealth >= MaxHealth)
311 OnMaxHealthReached(args);
312 else if (CurrHealth <= 0)
313 {
314 OnZeroHealthReached(args);
315
316 Destroy(false, args.Source);
317 }
318
319 if(args.Value < 0)
320 {
321 // Hit effect and audio
322 effectObjPool.Spawn(hitEffect.Output, hitEffectSpawnInput);
323 audioMgr.PlaySFX(Entity.AudioSourceComponent, hitAudio.Fetch(), loop:false);
324 }
325
326 return ErrorMessage.none;
327 }

358 public ErrorMessage DestroyLocal(bool upgrade, IEntity source)
359 {
360 ErrorMessage errorMessage;
361 if ((errorMessage = CanDestroy(upgrade, source)) != ErrorMessage.none)
362 return errorMessage;
363
364 IsDead = true;
365
366 TerminatedBy = source;
367
368 CurrHealth = 0;
369
370 float nextDestroyDelay = upgrade ? 0.0f : DestroyObjectDelay;
371
372 if (destroyObject || upgrade)
373 {
374 // If the entity object is inactive then we can not start coroutines on it and we should destroy directly
375 if(!Entity.gameObject.activeInHierarchy)
376 Destroy(gameObject);
377 else
378 //Destroy the faction entity's object:
379 StartCoroutine(DestroyCoroutine(nextDestroyDelay));
380 }
381
382 if (Entity.IsInitialized)
383 {
384 //If this is no upgrade
385 if (!upgrade)
386 {
387 IEffectObject nextEffect = effectObjPool.Spawn(destructionEffect.Output, transform.position);
388
389 audioMgr.PlaySFX(nextEffect.IsValid() && nextEffect.AudioSourceComponent.IsValid()
390 ? nextEffect.AudioSourceComponent
391 : Entity.AudioSourceComponent,
392 destructionAudio.Fetch(), false);
393
394 if (source?.IsFree == false)
395 resourceMgr.UpdateResource(source.FactionID, destroyAward, add: true);
396
397 stateHandler.Activate(destroyState);
398 }
399 }
400
401 if(Entity.MovementComponent.IsValid())
402 Entity.MovementComponent.TargetPositionMarker.Toggle(false);
403
404 OnDestroyed(upgrade, source);
405
406 RaiseEntityDead(new DeadEventArgs(upgrade, source, nextDestroyDelay));
407
408 selectionMgr.Remove(Entity);
409
410 return ErrorMessage.none;
411 }
