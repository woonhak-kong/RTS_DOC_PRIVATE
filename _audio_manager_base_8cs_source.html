134 public void PlaySFX(AudioSource inputSource, AudioClip inputClip, bool loop = false)
135 {
136 if (inputClip == null) //in case no clip is assigned, do not continue.
137 return;
138
139 //make sure that there's a valid audio source before playing the clip
140 if (inputSource == null)
141 {
142 logger.LogWarning($"[AudioManager] AudioSource is missing, can not play the audio clip.");
143 return;
144 }
145
146 StopSFX(inputSource);
147
148 bool playAudio = true;
149
150 if (muteSameSFXWithinRange)
151 {
152 if (sfxClipSourceTransforms.TryGetValue(inputClip, out List<AudioSource> currentSources))
153 {
154 currentSources.Remove(inputSource);
155
156 int i = 0;
157 while(i < currentSources.Count)
158 {
159 AudioSource nextSource = currentSources[i];
160 if(!nextSource.IsValid())
161 {
162 currentSources.RemoveAt(i);
163 continue;
164 }
165
166 // Only consider an audio source to mute another one if it is currently playing the same clip and it is within the defined range.
167 if (nextSource.isPlaying
168 && Vector3.Distance(nextSource.transform.position, inputSource.transform.transform.position) < sameSFXMuteRange)
169 {
170 playAudio = false;
171 break;
172 }
173
174 i++;
175 }
176 }
177
178 if (currentSources.IsValid())
179 currentSources.Add(inputSource);
180 else
181 sfxClipSourceTransforms.Add(inputClip, new List<AudioSource>() { inputSource });
182
183 if (sfxSourceActiveClips.ContainsKey(inputSource))
184 sfxSourceActiveClips.Remove(inputSource);
185
186 sfxSourceActiveClips.Add(inputSource, inputClip);
187 }
188
189 if (playAudio)
190 {
191 inputSource.clip = inputClip;
192 inputSource.loop = loop;
193
194 inputSource.Play(); //play the new clip
195 }
196 }
