64
65 if(!logger.RequireTrue(minimapCameraController.WorldPointToLocalPointInMinimapCanvas(
66 input.SourceEntity.transform.position, out Vector3 spawnPosition, height: input.Height),

70 base.OnSpawn(new PoolableObjectSpawnInput(

73 useLocalTransform: true,
74 spawnPosition: spawnPosition,
75 spawnRotation: Quaternion.identity
76 ));
77
78 image.sprite = input.Data.icon.IsValid() ? input.Data.icon : minimapIconMgr.DefaultIcon;
79
80 Color color = input.SourceEntity.SelectionColor;
81 if (input.Data.changeDarkness)
82 {
83 Color.RGBToHSV(color, out float hue, out float saturation, out float v);
84 color = Color.HSVToRGB(hue, saturation, v);
85 }
86 if(input.Data.changeTransparency)
87 color.a = 1.0f - input.Data.transparency;
88 image.color = color;

90 ResetFollowEntity();
91 SetFollowEntity(input.SourceEntity);
92 }
