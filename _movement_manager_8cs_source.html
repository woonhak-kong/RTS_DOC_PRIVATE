432 int emptyAttemptsCount = 0;

451

453 },

458 {
459 //Debug.Log("GeneratePathDestination()2 = ");
460 attemptedFormationTypes.Add(formationSelector.type);
461
462 // Reset empty attemps count and offset for next fallback formation type
463 emptyAttemptsCount = 0;
464 offset = originalOffset;
465
466 // Current formation type could not compute all path destinations then generate path destinations with the fall back formation if there's one
467 if (formationHandlers[formationSelector.type].FallbackFormationType.IsValid()
468 // And make sure we have not attempted to generate path destinations with the fallback formation in this call

470 {
471 formationSelector = new MovementFormationSelector
472 {
473 type = formationHandlers[formationSelector.type].FallbackFormationType,
474 properties = formationSelector.properties
475 };
476
477 continue;
478 }
479
480 // No fallback formation? exit!
481 return errorMessage;
482 }
483
484 // Only if the last attempt resulted in no generated path destinations.
485 if (generatedAmount == 0)
486 emptyAttemptsCount++;
487 }
488
489 // We have computed at least one path destination, the count of the list is either smaller or equal to the initial value of the "amount" argument.
490 return ErrorMessage.none;

493 // Generate single path destination around an origin position within a certain range
494 public ErrorMessage GeneratePathDestination (Vector3 originPosition, float range, IMovementComponent refMvtComp, out Vector3 targetPosition)
495 {
496 targetPosition = originPosition + (Random.insideUnitSphere * range);
497
498 int expectedPositionCount = Mathf.FloorToInt(2.0f * Mathf.PI * range / (refMvtComp.Entity.Radius * 2.0f));
499
500 // If no expected positions are to be found and the radius offset is zero then set the expected position count to 1 to test the actual target position if it is valid
501 if (expectedPositionCount == 0)
502 expectedPositionCount = 1;
503
504 // Represents increment value of the angle inside the current circle with the above perimeter
505 float angleIncValue = 360f / expectedPositionCount;
506 float currentAngle = 0.0f;
507
508 Vector3 nextDestination = originPosition + Vector3.right * range;
509
510 int counter = 0;
511
512 // As long as we haven't inspected all the expected free positions inside this cirlce
513 while (counter < expectedPositionCount)
514 {
515 // Always make sure that the next path destination has a correct height in regards to the height of the map.
516 if (terrainMgr.SampleHeight(nextDestination, refMvtComp, out nextDestination.y))
517 {
518 // Check if there is no obstacle and no other reserved target position on the currently computed potential path destination
519 if (IsPositionClear(ref nextDestination, refMvtComp, playerCommand: false) == ErrorMessage.none)
520 {
521 targetPosition = nextDestination;
522 return ErrorMessage.none;
523 }
524
525 // Increment the angle value to find the next position on the circle
526 currentAngle += angleIncValue;
527
528 // Rotate the nextDestination vector around the y axis by the current angle value
529 nextDestination = originPosition + range * new Vector3(Mathf.Cos(Mathf.Deg2Rad * currentAngle), 0.0f, Mathf.Sin(Mathf.Deg2Rad * currentAngle));
530 }
531
532 counter++;
533 }
534
535 return ErrorMessage.invalid;
536 }
