137 private void Toggle(bool enable, IBuilding building)
138 {
139 if (!building.IsValid())
140 {
141 logger.LogError($"[{GetType().Name}] ");
142 IsEnabled = false;
143
144 return;
145 }
146
147 int index = 0;
148
149 // Enabling
150 if (enable)
151 {
152 // Try to get an already created slot but a disabled one (to avoid adding new elements to the data list and better handle garbage collection)
153 BuildingPlacementGridMeshData nextData = new BuildingPlacementGridMeshData();
154 while (index < data.Count)
155 {
156 if(data[index].enabled)
157 {
158 index++;
159 continue;
160 }
161
162 nextData = data[index];
163 nextData.targetBuilding = building;
164 nextData.enabled = true;
165 break;
166 }
167 if(index == data.Count)
168 {
169 nextData = new BuildingPlacementGridMeshData
170 {
171 enabled = true,
172 targetBuilding = building,
173 cellRenderers = new List<List<ActiveCellRenderer>>()
174 };
175 data.Add(nextData);
176 enabledDataCount++;
177 }
178
179 if (disableSelectionMarker)
180 {
181 nextData.targetBuilding.SelectionMarker.Disable();
182 nextData.targetBuilding.SelectionMarker.IsLocked = true;
183 }
184
185 placerMgr.GridHandler.TryGetCellPosition(
186 nextData.targetBuilding.PlacerComponent.GridOptions.ApplyPivotPointReverse(
187 nextData.targetBuilding.transform.position,
188 GridAreaPivotPoint.bottomLeft),
189 out Int2D buildingLowerLeftCornerPosition);
190
191 Int2D area = nextData.targetBuilding.PlacerComponent.GridOptions.area;
192
193 for (int i = -range; i < area.x + range; i++)
194 {
195 nextData.cellRenderers.Add(new List<ActiveCellRenderer>());
196
197 for (int j = -range; j < area.y + range; j++)
198 {
199 IEffectObject nextCellEffect = effectObjPool.Spawn(
200 singleCellRendererPrefab.Output,
201 new EffectObjectSpawnInput(
202 parent: null,
203 useLocalTransform: false,
204 spawnPosition: Vector3.zero,
205 spawnRotation: Quaternion.identity,
206 enableLifeTime: false));
207
208 nextData.cellRenderers[i + range].Add(new ActiveCellRenderer
209 {
210 effectObject = nextCellEffect,
211 meshRenderer = nextCellEffect.gameObject.GetComponent<MeshRenderer>(),
212 targetBuilding = building,
213 isInnerArea = IsCellIndexInnerArea(i, j, area)
214 });
215
216 // inner area vs outer area data
217 CellRendererData nextRendererData = (i >= 0 && i < area.x && j >= 0 && j < area.y)
218 ? innerAreaCellData
219 : outerAreaCellData;
220
221 nextData.cellRenderers[i + range][j + range].meshRenderer.materials[nextRendererData.materialID] = nextRendererData.material.IsValid()
222 ? nextRendererData.material : new Material(Shader.Find("Sprites/Default"));
223 }
224 }
225
226 nextData.targetBuilding.PlacerComponent.BuildingPlacementTransformUpdated += HandleBuildingPlacementTransformUpdated;
227 placementBuildingToDataIndex.Add(nextData.targetBuilding, index);
228
229 IsEnabled = true;
230
231 HandleBuildingPlacementTransformUpdated(building, EventArgs.Empty);
232 }
233 // Disabling
234 else
235 {
236 while (index < data.Count)
237 {
238 BuildingPlacementGridMeshData nextData = data[index];
239
240 if (!nextData.enabled || nextData.targetBuilding != building)
241 {
242 index++;
243 continue;
244 }
245
246 Int2D area = nextData.targetBuilding.PlacerComponent.GridOptions.area;
247 for (int i = -range; i < area.x + range; i++)
248 {
249 for (int j = -range; j < area.y + range; j++)
250 {
251 if (placerMgr.GridHandler.TryGetCellPosition(
252 nextData.cellRenderers[i + range][j + range].effectObject.transform.position,
253 out Int2D cellPosition))
254 cellRenderersPositions.Remove(cellPosition);
255
256 nextData.cellRenderers[i + range][j + range].effectObject.Deactivate(useDisableTime: false);
257 }
258 }
259
260 nextData.cellRenderers.Clear();
261 nextData.enabled = false;
262
263 nextData.targetBuilding.PlacerComponent.BuildingPlacementTransformUpdated -= HandleBuildingPlacementTransformUpdated;
264 placementBuildingToDataIndex.Remove(nextData.targetBuilding);
265 nextData.targetBuilding = null;
266
267 enabledDataCount--;
268 break;
269 }
270
271 if (enabledDataCount == 0)
272 IsEnabled = false;
273 }

276 private void HandleBuildingPlacementTransformUpdated(IBuilding targetBuilding, EventArgs args)
277 {
278 if (!targetBuilding.IsLocalPlayerFaction())
279 return;
280
281 BuildingPlacementGridMeshData nextData = data[placementBuildingToDataIndex[targetBuilding]];
282
283 placerMgr.GridHandler.TryGetCellPosition(
284 nextData.targetBuilding.PlacerComponent.GridOptions.ApplyPivotPointReverse(
285 nextData.targetBuilding.transform.position,
286 GridAreaPivotPoint.bottomLeft),
287 out Int2D buildingLowerLeftCornerPosition);
288
289 Int2D area = nextData.targetBuilding.PlacerComponent.GridOptions.area;
290
291 // Remove the positions of the current cell renderers from the last building placement position
292 for (int i = -range; i < area.x + range; i++)
293 {
294 for (int j = -range; j < area.y + range; j++)
295 {
296 if (placerMgr.GridHandler.TryGetCellPosition(
297 nextData.cellRenderers[i + range][j + range].effectObject.transform.position,
298 out Int2D cellPosition) && cellRenderersPositions.TryGetValue(cellPosition, out ActiveCellRenderer nextCellRenderer))
299 {
300 // Check for the target building of cell renderer so we know the cell with that position belongs to the target building we are currently handling
301 if(nextCellRenderer.targetBuilding == targetBuilding)
302 cellRenderersPositions.Remove(cellPosition);
303 }
304 }
305 }
306
307 for (int i = -range; i < area.x + range; i++)
308 {
309 for (int j = -range; j < area.y + range; j++)
310 {
311 IEffectObject nextCellEffect = nextData.cellRenderers[i + range][j + range].effectObject;
312
313 Vector3 cellWorldPosition = new Vector3(
314 buildingLowerLeftCornerPosition.x + (targetBuilding.PlacerComponent.GridOptions.IsHorizontal ? i : j) + placerMgr.GridHandler.CellSize / 2.0f,
315 0.0f,
316 buildingLowerLeftCornerPosition.y + (targetBuilding.PlacerComponent.GridOptions.IsHorizontal ? j : i) + placerMgr.GridHandler.CellSize / 2.0f);
317
318 bool isInnerArea = nextData.cellRenderers[i + range][j + range].isInnerArea;
319 bool isCellExist = placerMgr.GridHandler.TryGetCellPosition(cellWorldPosition, out Vector3 cellBottomLeftPosition);
320 Int2D nextCellPosition = new Int2D { x = (int)cellBottomLeftPosition.x, y = (int)cellBottomLeftPosition.z };
321
322 if (!isCellExist
323 || cellRenderersPositions.ContainsKey(nextCellPosition))
324 {
325 // if the current cell is an inner area one and there is one occupying its current position that is not an inner area one
326 // then disable the non-inner area cell that usually belongs to another placement instance
327 if (isCellExist && isInnerArea
328 && !cellRenderersPositions[nextCellPosition].isInnerArea)
329 {
330 cellRenderersPositions[nextCellPosition].effectObject.gameObject.SetActive(false);
331 cellRenderersPositions.Remove(nextCellPosition);
332 }
333 else
334 {
335 nextCellEffect.gameObject.SetActive(false);
336 continue;
337 }
338 }
339
340 // Add the new positions for the cell renderers so that no two cells will spawn on top of each other.
341 cellRenderersPositions.Add(nextCellPosition, nextData.cellRenderers[i + range][j + range]);
342
343 nextCellEffect.gameObject.SetActive(true);
344 cellWorldPosition.y = cellBottomLeftPosition.y;
345 nextCellEffect.transform.position = cellWorldPosition;
346
347 bool isOccupied;
348 CellRendererData nextRendererData;
349
350 if(isInnerArea)
351 {
352 nextRendererData = innerAreaCellData;
353 isOccupied = !nextData.targetBuilding.PlacerComponent.CanPlace;
354 }
355 else
356 {
357 nextRendererData = outerAreaCellData;
358 isOccupied = placerMgr.GridHandler.IsOccupied(cellWorldPosition);
359 }
360
361 nextData.cellRenderers[i + range][j + range].meshRenderer.materials[nextRendererData.materialID].color = isOccupied
362 ? nextRendererData.occupiedColor : nextRendererData.freeColor;
363 }
364 }
365 }
