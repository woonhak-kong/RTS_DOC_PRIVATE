78 #region Handling Unit Regulators
79 public NPCUnitRegulator ActivateUnitRegulator(IUnit unitPrefab)
80 {
81 if (!logger.RequireValid(unitPrefab,
82 $"[{GetType().Name} - Faction ID: {factionMgr.FactionID}] Can not activate a regulator for an invalid unit prefab, check the 'Independent Units' list for unassigned elements or any other unit input field in other NPC components."))
83 return null;
84
85 // If no valid regulator data for the unit is returned then do not continue
86 NPCUnitRegulatorData regulatorData = unitPrefab.GetComponent<NPCUnitRegulatorDataInput>()?.GetFiltered(factionType: factionSlot.Data.type, npcType: npcMgr.Type);
87 if (!regulatorData.IsValid())
88 {
89 if (defaultRegulatorData.IsValid())
90 regulatorData = defaultRegulatorData;
91 else
92 {
93 LogEvent($"{unitPrefab.Code}: Unable to find a valid regulator for unit.");
94 return null;
95 }
96 }
97
98 // If there is an active regulator for the current unit type
99 NPCUnitRegulator activeInstance = GetActiveUnitRegulator(unitPrefab.Code);
100 if (activeInstance.IsValid())
101 {
102 LogEvent($"{unitPrefab.Code}: A valid active regulator already exists!");
103 return activeInstance;
104 }
105
106 // At this stage, there is valid data for the regulator and it has not been created yet
107 NPCActiveUnitRegulatorData newUnitRegulator = new NPCActiveUnitRegulatorData()
108 {
109 instance = new NPCUnitRegulator(regulatorData, unitPrefab, gameMgr, npcMgr),
110
111 // Initial spawning timer: regular spawn reload + start creating after delay
112 spawnTimer = new TimeModifiedTimer(regulatorData.CreationDelayTime + regulatorData.SpawnReload)
113 };
114
115 newUnitRegulator.instance.AmountUpdated += HandleUnitRegulatorAmountUpdated;
116
117 activeUnitRegulators.Add(unitPrefab.Code, newUnitRegulator);
118
119 // Whenever a new regulator is added to the active regulators list, then move the unit creator into the active state
120 IsActive = true;
121
122 LogEvent($"{unitPrefab.Code}: Created and activated a regulator for unit. Using default regulator data? {regulatorData == defaultRegulatorData}");
123 return newUnitRegulator.instance;
124 }

175 #region Handling Unit Creation
176 protected override void OnActiveUpdate()
177 {
178 // Assume that the unit creator has finished its job with the current active unit regulators.
179 IsActive = false;
180
181 foreach (NPCActiveUnitRegulatorData nextUnitRegulator in activeUnitRegulators.Values)
182 {
183 //if we can auto create this:
184 if (nextUnitRegulator.instance.Data.CanAutoCreate
185 && !nextUnitRegulator.instance.HasTargetCount)
186 {
187 // To keep this component monitoring the creation of the next units
188 IsActive = true;
189
190 if (nextUnitRegulator.spawnTimer.ModifiedDecrease())
191 {
192 nextUnitRegulator.spawnTimer.Reload(nextUnitRegulator.instance.Data.SpawnReload);
193
194 OnCreateUnitRequestInternal(
195 nextUnitRegulator.instance,
196 nextUnitRegulator.instance.TargetCount - nextUnitRegulator.instance.Count,
197 out _);
198 }
199 }
200 }

203 public bool OnCreateUnitRequest(string unitCode, int requestedAmount, out int createdAmount)
204 {
205 createdAmount = 0;
206
207 NPCUnitRegulator nextUnitRegulator = GetActiveUnitRegulator(unitCode);
208 if (!nextUnitRegulator.IsValid())
209 {
210 LogEvent($"{unitCode}: Unable to create {requestedAmount} instances. No valid regulator found!");
211 return false;
212 }
213 else if (nextUnitRegulator.Data.CanCreateOnDemand == false)
214 {
215 LogEvent($"{unitCode}: Unable to create {requestedAmount} instances. Creation on demand is disabled!");
216 return false;
217 }
218
219 return OnCreateUnitRequestInternal(nextUnitRegulator, requestedAmount, out createdAmount);
220 }
221
222 private bool OnCreateUnitRequestInternal(NPCUnitRegulator instance, int requestedAmount, out int createdAmount)
223 {
224 createdAmount = 0;
225
226 if (!instance.IsValid())
227 return false;
228 if(instance.HasReachedMaxAmount
229 || requestedAmount <= 0)
230 {
231 LogEvent($"{instance.Prefab.Code}: Unable to create {requestedAmount} instances. Has reached max amount? {instance.HasReachedMaxAmount}");
232 return false;
233 }
234
235 // If there are no task launchers assigned to this unit regulator that can create instances of the units...
236 if (instance.CreatorsCount == 0)
237 {
238 LogEvent($"{instance.Prefab.Code}: Unable to create {requestedAmount} instances. No valid UnitCreator found!");
239 // FUTURE FEATURE: Allow NPC faction to scan its available units/buildings to create one that can produce this unit type
240 return false;
241 }
242
243 createdAmount = requestedAmount;
244 instance.Create(ref requestedAmount);
245
246 createdAmount -= requestedAmount;
247
248 LogEvent($"{instance.Prefab.Code}: Attempting to create {requestedAmount+createdAmount} instances, successfully launched {createdAmount} creation tasks!");
249
250 return true;
251 }
