218 public override ErrorMessage IsTargetValid(SetTargetInputData data)
219 {
220 TargetData<IFactionEntity> potentialTarget = data.target;
221
222 if (!potentialTarget.instance.IsValid() || !potentialTarget.instance.CanLaunchTask)
223 return ErrorMessage.invalid;
224 else if (!potentialTarget.instance.IsSameFaction(unit))
225 return ErrorMessage.factionMismatch;
226 else if (!potentialTarget.instance.IsInteractable)
227 return ErrorMessage.uninteractable;
228 else if (potentialTarget.instance.Health.IsDead)
229 return ErrorMessage.targetDead;
230 else if (!potentialTarget.instance.DropOffTarget.IsValid())
231 return ErrorMessage.dropOffTargetMissing;
232 else if (!targetPicker.IsValidTarget(potentialTarget.instance))
233 return ErrorMessage.targetPickerUndefined;
234 else if (!IsTargetInRange(transform.position, potentialTarget))
235 return ErrorMessage.targetOutOfRange;
236 else if (!potentialTarget.instance.DropOffTarget.CanDropResourceType(LastCollectedResourceType))
237 return ErrorMessage.resourceTypeMismatch;
238
239 return potentialTarget.instance.DropOffTarget.CanMove(unit);
240 }

356 // Forcing drop off means that if the collector has at least one resource unit of any type, it will be dropped off.
357 public bool AttemptStartDropOff (bool force = false, ResourceTypeInfo resourceType = null)
358 {
359 if (CollectedResourcesSum == 0
360 || (!force
361 && !HasReachedMaxCapacity(resourceType)))
362 return false;
363
364 // When attempting to drop off a resource type that has already reached its maximum capacity limit
365 if(resourceMgr.HasResourceTypeReachedLimitCapacity(resourceType, unit.FactionID))
366 {
367 if (unit.IsLocalPlayerFaction())
368 {
369 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
370 {
371 message = ErrorMessage.resourceTypeLimitCapacityReached,
372 text = resourceType.DisplayName
373 });
374 }
375
376 unit.CollectorComponent.Stop();
377
378 return false;
379 }
380
381 RaiseDropOffStateUpdated(DropOffState.ready);
382
383 SendToTarget(false);
384
385 return true;
386 }

388 public void Unload()
389 {
390 RaiseDropOffStateUpdated(unit.CollectorComponent.HasTarget ? DropOffState.goingBack : DropOffState.inactive);
391
392 if (dropOffObject.IsValid())
393 dropOffObject.SetActive(false);
394
395 // Only units that belong to a faction can update their faction's resources.
396 if(!unit.IsFree)
397 foreach (ResourceTypeInfo resourceType in collectedResources.Keys.ToArray())
398 {
399 if (collectedResources[resourceType] == 0)
400 continue;
401
402 ResourceInput nextInput = new ResourceInput
403 {
404 type = resourceType,
405 value = new ResourceTypeValue
406 {
407 amount = collectedResources[resourceType],
408 capacity = 0
409 }
410 };
411
412 resourceMgr.UpdateResource(
413 unit.FactionID,
414 nextInput,
415 add: true,
416 out int restAmount);
417
418 int unloadedAmount = (nextInput.value.amount - restAmount);
419 collectedResources[nextInput.type] -= unloadedAmount;
420 CollectedResourcesSum -= unloadedAmount;
421
422 // In case not all of the amount could be unloaded due to a limited capacity type resource that reached its maximum capacity...
423 if(restAmount > 0 && unit.IsLocalPlayerFaction())
424 {
425 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
426 {
427 message = ErrorMessage.resourceTypeLimitCapacityReached,
428 amount = restAmount,
429 text = nextInput.type.DisplayName
430 });
431 }
432
433 globalEvent.RaiseUnitResourceDropOffCompleteGlobal(unit, new ResourceAmountEventArgs(nextInput));
434 }
435
436 unit.AnimatorController?.ResetOverrideController();
437
438 RaiseDropOffUnloaded();
439 RaiseCollectedResourcesUpdated();
440
441 // Back to collect the last resource
442 if (State == DropOffState.goingBack)
443 {
444 unit.CollectorComponent.SetTarget(unit.CollectorComponent.Target, false);
445 }
446 // If the unit is idle (ResourceCollector is inactive) and we are allowed to drop off when the target available...
447 // ...then this means we are resuming the resource collection of the same resource (LastTarget) that was being collected before...
448 // ...the unit had to stop collecting due to not having a drop off point available
449 else if (dropOffOnTargetAvailable && unit.IsIdle)
450 {
451 unit.CollectorComponent.SetTarget(unit.CollectorComponent.LastTarget, false);
452 }
453 }
