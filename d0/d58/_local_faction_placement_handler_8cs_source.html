84 // Keep moving the building by following the player's mouse
85 private void MoveBuilding()
86 {
87 // Using a raycheck, we will make the current building follow the mouse position and stay on top of the terrain.
88 if (Physics.Raycast(
89 mainCameraController.MainCamera.ScreenPointToRay(Input.mousePosition),
90 out RaycastHit hit,
91 Mathf.Infinity,
92 placerMgr.PlacableLayerMask))
93 {
94 // Depending on the height of the terrain, we will place the building on it
95 Vector3 nextBuildingPos = hit.point;
96
97 // Make sure that the building position on the y axis stays inside the min and max height interval
98 nextBuildingPos.y += placerMgr.BuildingPositionYOffset;
99
100 if (current.instance.transform.position != nextBuildingPos)
101 {
102 // Check if the building can be placed in this new position
103 current.instance.PlacerComponent.OnPlacementUpdate(nextBuildingPos);
104 }
105
106 }

110 #region Handling Rotation
111 private void RotateBuilding()
112 {
113 if (!canRotate
114 || current.options.disableRotation)
115 return;
116
117 Vector3 nextEulerAngles = current.instance.transform.rotation.eulerAngles;
118 // Only rotate if one of the keys is pressed down (check for direction) and rotate on the y axis only.
119 nextEulerAngles.y += rotationSpeed * (controls.Get(positiveRotationKey) ? 1.0f : (controls.Get(negativeRotationKey) ? -1.0f : 0.0f));
120
121 current.instance.transform.rotation = Quaternion.Euler(nextEulerAngles);
122 }
