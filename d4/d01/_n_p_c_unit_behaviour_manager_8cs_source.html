201 private void HandleAttackEngageOrderResponse()
202 {
203 if (!nextAttackEngageOrderTargetData.delayTimer.ModifiedDecrease())
204 return;
205
206 State = NPCUnitBehaviourState.attacking;
207 LogEvent($"[ORDER UPDATED] Attack Order Received.");
208
209 awaitingAttackEngageOrderResponse = false;
210
211 sendableUnits.Clear();
212 for (int i = 0; i < regulatorMonitor.AllCodes.Count; i++)
213 {
214 string unitCode = regulatorMonitor.AllCodes[i];
215 if (attackEngageOrderBehaviour.sendIdleOnly)
216 {
217 sendableUnits.AddRange(npcUnitCreator.GetActiveUnitRegulator(unitCode).IdleInstances);
218 }
219 else if (attackEngageOrderBehaviour.sendNoTargetThreatOnly)
220 {
221 sendableUnits.AddRange(npcUnitCreator.GetActiveUnitRegulator(unitCode).IdleInstances);
222 for (int j = 0; j < npcUnitCreator.GetActiveUnitRegulator(unitCode).NonIdleInstances.Count; j++)
223 {
224 IUnit unit = npcUnitCreator.GetActiveUnitRegulator(unitCode).NonIdleInstances[j];
225 if (!unit.CanAttack || !unit.FirstActiveAttackComponent.HasTarget || !unit.FirstActiveAttackComponent.Target.instance.CanAttack)
226 sendableUnits.Add(unit);
227 }
228 }
229 else
230 sendableUnits.AddRange(npcUnitCreator.GetActiveUnitRegulator(unitCode).Instances);
231 }
232
233 if (sendableUnits.Count == 0)
234 return;
235
236 if (attackEngageOrderBehaviour.attack)
237 {
238 attackMgr.LaunchAttack(
239 new LaunchAttackData<IReadOnlyList<IEntity>>
240 {
241 source = sendableUnits,
242 targetEntity = nextAttackEngageOrderTargetData.target,
243 targetPosition = nextAttackEngageOrderTargetData.target.IsValid() ? nextAttackEngageOrderTargetData.target.transform.position : nextAttackEngageOrderTargetData.targetPosition,
244 playerCommand = true
245 });
246
247 LogEvent($"[ATTACK ORDER] Sending in {sendableUnits.Count} units to attack target {nextAttackEngageOrderTargetData.target}");
248 }
249 else
250 {
251 LogEvent($"[ATTACK ORDER] Sending in {sendableUnits.Count} units to move to target {nextAttackEngageOrderTargetData.target}");
252
253 mvtMgr.SetPathDestination(
254 new SetPathDestinationData<IReadOnlyList<IEntity>>
255 {
256 source = sendableUnits,
257 destination = nextAttackEngageOrderTargetData.target.IsValid() ? nextAttackEngageOrderTargetData.target.transform.position : nextAttackEngageOrderTargetData.targetPosition,
258 offsetRadius = nextAttackEngageOrderTargetData.target.IsValid() ? nextAttackEngageOrderTargetData.target.Radius : 0.0f,
259 target = nextAttackEngageOrderTargetData.target,
260 mvtSource = new MovementSource { playerCommand = true }
261 });
262 }
263 }

266 #region Territory Defense Behaviour
267 private void HandleTerritoryDefenseOrder(INPCDefenseManager sender, NPCTerritoryDefenseEngageEventArgs args)
268 {
269 if (!territoryDefenseOrderBehaviour.defend)
270 return;
271
272 State = NPCUnitBehaviourState.defending;
273 LogEvent($"[ORDER UPDATED] Defense Order Received.");
274
275 int count = 0;
276 for (int i = 0; i < regulatorMonitor.AllCodes.Count; i++)
277 {
278 for (int j = 0; j < npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances.Count; j++)
279 {
280 IUnit unit = npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances[j];
281
282 if (!unit.CanAttack
283 || (unit.FirstActiveAttackComponent.HasTarget && !territoryDefenseOrderBehaviour.forceChangeDefenseCenter))
284 continue;
285
286 // the playerCommand is set to ON in order to bypass the LOS constraints when having the attack units automatically find targets all over the border territory
287 unit.FirstActiveAttackComponent.SetSearchRangeCenterAction(args.NextDefenseCenter.BorderComponent, playerCommand: true);
288 count++;
289 }
290 }
291
292 LogEvent($"[DEFENSE ORDER] Sending in {count} units to defend building center {args.NextDefenseCenter.BorderComponent?.Building.gameObject.name}");
293 }
294
295 private void HandleTerritoryDefenseCancelled(INPCDefenseManager sender, EventArgs args)
296 {
297 int count = 0;
298 for (int i = 0; i < regulatorMonitor.AllCodes.Count; i++)
299 {
300 for (int j = 0; j < npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances.Count; j++)
301 {
302 IUnit unit = npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances[j];
303
304 if (!unit.CanAttack)
305 continue;
306
307 unit.FirstActiveAttackComponent.SetSearchRangeCenterAction(null, playerCommand: false);
308 count++;
309 }
310 }
311
312 LogEvent($"[DEFENSE ORDER] Cancelling defense order for {count} units.");
313
314 if (!territoryDefenseOrderBehaviour.sendBackOnDefenseCancel)
315 return;
316
317 SendToSpawnPoints();
318 }

322 private void SendToSpawnPoints()
323 {
324 State = NPCUnitBehaviourState.idle;
325 LogEvent($"[ORDER UPDATED] Idle Order Receieved, Sending To Spawn.");

327
328 for (int i = 0; i < regulatorMonitor.AllCodes.Count; i++)
329 {
330 LogEvent($"[SEND TO SPAWN ORDER] Sending units of code {regulatorMonitor.AllCodes[i]} back to spawn positions.");

332 if (npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances.Count == 0)
333 continue;
334
335 IUnit refUnit = npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances[0];
336 mvtMgr.SetPathDestination(

338 {
339 source = npcUnitCreator.GetActiveUnitRegulator(regulatorMonitor.AllCodes[i]).Instances,
340 destination = refUnit.SpawnRallypoint.IsValid() ? refUnit.SpawnRallypoint.Entity.transform.position : factionSlot.FactionSpawnPosition,
341 offsetRadius = refUnit.SpawnRallypoint.IsValid() ? refUnit.SpawnRallypoint.Entity.Radius : 0.0f,
342 target = null,
343 mvtSource = new MovementSource { playerCommand = true }
344 });
345 }
