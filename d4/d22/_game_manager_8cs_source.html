87 private T TryGetService<T>(bool isPreService = false, bool requireValid = true)
88 {
89 Type type = typeof(T);
90 T service = default;
91 if (isPreService)
92 {
93 if (preRunServices.TryGetValue(type, out IPreRunGameService value))
94 service = (T)value;
95 }
96 else
97 {
98 if (postRunServices.TryGetValue(type, out IPostRunGameService value))
99 service = (T)value;
100 }
101
102 if(requireValid && !service.IsValid())
103 {
104 if (typeof(IOptionalGameService).IsAssignableFrom(type))
105 logger.LogWarning($"[GameManager] No service of type '{type}' has been registered! This is just a warning because the service is marked as optional!");
106 else
107 logger.LogError($"[GameManager] No service of type '{type}' has been registered!");
108 }
109
110 return service;
111 }

249 private bool Build ()
250 {
251 activeFactionSlots = new List<IFactionSlot>();
252
253 // No pre defined builder? Use the default game settings!
254 if(!CurrBuilder.IsValid())
255 {
256 RandomizeFactionSlots();
257
258 int i = 0;
259 while(i < factionSlots.Count)
260 {
261 if(!factionSlots[i].Enabled)
262 {
263 factionSlots[i].InitDestroy();
264 factionSlots.RemoveAt(i);
265 continue;
266 }
267
268 factionSlots[i].Init(factionSlots[i].Data, ID: i, this);
269 activeFactionSlots.Add(factionSlots[i]);
270 i++;
271 }
272
273 return true;
274 }
275
276 RandomizeFactionSlots(CurrBuilder.Data.factionSlotIndexSeed?.ToList());
277
278 defeatCondition = CurrBuilder.Data.defeatCondition;
279
280 if(CurrBuilder.FactionSlotCount > FactionCount)
281 {
282 Debug.LogError($"[GameManager] Game Builder is attempting to initialize {CurrBuilder.FactionSlotCount} slots while there are only {FactionCount} slots available!");
283 return false;
284 }
285
286 for (int i = 0; i < CurrBuilder.FactionSlotCount; i++)
287 {
288 factionSlots[i].Init(CurrBuilder.FactionSlotDataSet.ElementAt(i), ID:i, this);
289 activeFactionSlots.Add(factionSlots[i]);
290 }
291
292 // Remove the extra unneeded slots
293 while (CurrBuilder.FactionSlotCount < factionSlots.Count)
294 {
295 factionSlots[factionSlots.Count - 1].InitDestroy();
296 factionSlots.RemoveAt(factionSlots.Count - 1);
297 }
298
299 return true;
300 }

357 private void RandomizeFactionSlots(IReadOnlyList<int> indexSeedList)
358 {
359 if (!randomFactionSlots
360 || !logger.RequireTrue(indexSeedList.IsValid() && indexSeedList.Count == factionSlots.Count,
361 $"[{GetType().Name}] Unable to randomize faction slots due to an index seed list that does not match with the faction slots count. Faction slots will not be randomized!",
362 type: LoggingType.warning))
363 return;
364
365 int i = 0;
366 while(i < indexSeedList.Count)
367 {
368 if (i == indexSeedList[i] || i > indexSeedList[i])
369 {
370 i++;
371 continue;
372 }
373
374 var tempSlot = factionSlots[i];
375 factionSlots[i] = factionSlots[indexSeedList[i]];
376 factionSlots[indexSeedList[i]] = tempSlot;
377
378 i++;
379 }
380 }

401 public ErrorMessage OnFactionDefeatedLocal(int factionID)
402 {
403 if (!factionSlots[factionID].IsActiveFaction())
404 return ErrorMessage.inactive;
405
406 factionSlots[factionID].UpdateState(FactionSlotState.eliminated);
407 factionSlots[factionID].UpdateRole(FactionSlotRole.client);
408
409 activeFactionSlots.Remove(factionSlots[factionID]);
410
411 gameUIText.FactionSlotDefeatToText(factionSlots[factionID], out string defeatMsg);
412
413 globalEvent.RaiseShowPlayerMessageGlobal(
414 this,
415 new MessageEventArgs
416 (
417 type: MessageType.info,
418 message: defeatMsg
419 ));
420
421 globalEvent.RaiseFactionSlotDefeatedGlobal(factionSlots[factionID], new DefeatConditionEventArgs(type: DefeatCondition));
422
423 if (LocalFactionSlot.IsValid() && State == GameStateType.running)
424 {
425 // If this is the local player rfaction?
426 if (factionID.IsLocalPlayerFaction())
427 LooseGame();
428 else if (factionSlots.Count(slot => slot.State == FactionSlotState.active) == 1)
429 WinGame();
430 }
431
432 return ErrorMessage.none;
433 }
