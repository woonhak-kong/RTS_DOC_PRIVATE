91 private void HandleGameStartRunning(IGameManager gameMgr, EventArgs args)
92 {
93 GameObject prefabsParent = new GameObject("PrefabsParent");
94 prefabsParent.transform.SetParent(gameMgr.transform);
95 prefabsParent.transform.localPosition = Vector3.zero;
96
97 entityPrefabs = (fetchSpawnablePrefabsType == FetchSpawnablePrefabsType.manual
98 ? manualSpawnablePrefabs
99 : Resources
100 .LoadAll("Prefabs", typeof(GameObject))
101 .Cast<GameObject>())
102 .Where(prefabObj =>
103 {
104 IEntity nextPrefab = prefabObj.IsValid() ? prefabObj.GetComponent<IEntity>() : null;
105 return nextPrefab.IsValid()
106 && (fetchSpawnablePrefabsType != FetchSpawnablePrefabsType.codeCategoryPicker || spawnablePrefabsTargetPicker.IsValidTarget(prefabObj.GetComponent<IEntity>()));
107 })
108 .Select(prefabObj =>
109 {
110 IEntity prefabInstance = GameObject.Instantiate(prefabObj, Vector3.zero, prefabObj.transform.rotation).GetComponent<IEntity>();
111 return prefabInstance;
112 })
113 .ToList();
114
115 int index = -1;
116 entityCodeToPrefabIndex = entityPrefabs
117 .ToDictionary(entity => entity.Code, entity => { index++; return index; });
118
119 // Initialize prefabs as some entity components (i.e. IEntityModel) must perform some actions on the prefab level.
120 foreach (IEntity entity in entityPrefabs)
121 {
122 entity.InitPrefab(gameMgr);
123 entity.transform.SetParent(prefabsParent.transform);
124 entity.transform.localPosition = Vector3.zero;
125 entity.gameObject.SetActive(false);
126 }
127
128 // Entity prefabs are actually just inactive instances of the original prefabs that have been initialized with InitPrefab
129
130 foreach (IFactionSlot faction in gameMgr.FactionSlots)
131 faction.InitDefaultFactionEntities();
132 }

149 public int RegisterEntity(IEntity newEntity, InitEntityParameters initParams)
150 {
151 if (!logger.RequireValid(newEntity,
152 $"[InputManager] Register an invalid entity is not allowed!"))
153 return INVALID_ENTITY_KEY;
154
155 int newEntityKey = INVALID_ENTITY_KEY;
156 if (initParams.enforceKey)
157 {
158
159 if (spawnedEntities.ContainsKey(initParams.key))
160 {
161 logger.LogError($"[InputManager] Enforced input key {initParams.key} has been already registered for another entity {spawnedEntities[initParams.key].gameObject.name}!");
162 return INVALID_ENTITY_KEY;
163 }
164 newEntityKey = initParams.key;
165 }
166 else
167 {
168 while(nextKey < 0 || spawnedEntities.ContainsKey(nextKey))
169 nextKey++;
170
171 newEntityKey = nextKey;
172 }
173
174 spawnedEntities.Add(newEntityKey, newEntity);
175 newEntity.Health.EntityDead += HandleRegisteredEntityDead;
176 return newEntityKey;
177 }

189 public ErrorMessage SendInput(CommandInput newInput, IEntity source, IEntity target)
190 {
191 if (!logger.RequireValid(source,
192 "[InputManager] Can not process input without a valid source!"))
193 return ErrorMessage.invalid;
194
195 if (newInput.playerCommand)
196 {
197 if (!RTSHelper.IsLocalPlayerFaction(source) && !(RTSHelper.IsNPCFaction(source) && RTSHelper.IsMasterInstance()))
198 return ErrorMessage.noAuthority;
199 }
200 else if (!RTSHelper.IsMasterInstance())
201 return ErrorMessage.noAuthority;
202
203 // The logic below does not allow playerCommand flagged inputs to come from NPC factions.
204 /*if (!RTSHelper.IsMasterInstance() && !newInput.playerCommand
205 || (newInput.playerCommand && !RTSHelper.IsLocalPlayerFaction(source)))
206 return ErrorMessage.noAuthority;*/
207
208 // New input can now be processed after checking for the game's permissions over the input's source.
209
210 // If we're creating an object, then look in the spawnable prefabs list
211 // Otherwise, use the unique key assigned to each IEntity as the sourceID
212 if (newInput.isSourcePrefab)
213 {
214 if (!entityCodeToPrefabIndex.TryGetValue(source.Code, out newInput.sourceID))
215 {
216 logger.LogError("[InputManager] Attempting to launch command with entity prefab of code '{source.Code}' as source while the entity prefab is not within a path that ends with '**/Resources/Prefabs' or not added manually to the inspector of this component!");
217 return ErrorMessage.invalid;
218 }
219 }
220 else
221 newInput.sourceID = source.Key;
222
223 newInput.targetID = target.IsValid() ? target.Key : INVALID_ENTITY_KEY;
224
225 return SendInputFinal(newInput);
226 }

259 private ErrorMessage SendInputFinal(CommandInput newInput)
260 {
261 if (gameMgr.State == GameStateType.frozen)
262 return ErrorMessage.gameFrozen;
263 else if (gameMgr.CurrBuilder.IsValid())
264 {
265 // If the input adder instance is not ready yet then cache the received inputs to be relayed when the input adder is ready
266 if (!gameMgr.CurrBuilder.IsInputAdderReady)
267 awaitingInputAdderInputs.Add(newInput);
268 else
269 gameMgr.CurrBuilder.InputAdder.AddInput(newInput);
270 }
271 else
272 LaunchInput(newInput);
273
274 return ErrorMessage.none;
275 }

691 private List<T> KeyStringToEntities<T>(string inputString) where T : IEntity
692 {
693 List<T> nextEntities = new List<T>();
694
695 if (String.IsNullOrEmpty(inputString))
696 return nextEntities;
697
698 string[] entityKeys = inputString.Split('.');
699 for (int i = 0; i < entityKeys.Length; i++)
700 {
701 int key = Int32.Parse(entityKeys[i]);
702 spawnedEntities.TryGetValue(key, out IEntity entity);
