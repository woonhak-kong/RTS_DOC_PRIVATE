164 //a method that converts time in seconds to a string MM:SS
165 public static string TimeToString(float time)
166 {
167 if (time <= 0.0f)
168 return "00:00";
169
170 timeStringBuilder.Clear();
171
172 int seconds = Mathf.RoundToInt(time);
173 int minutes = Mathf.FloorToInt(seconds / 60.0f);
174
175 seconds -= minutes * 60;
176
177 if (minutes < 10)
178 timeStringBuilder.Append("0");
179 timeStringBuilder.Append(minutes);
180 timeStringBuilder.Append(":");
181 if (seconds < 10)
182 timeStringBuilder.Append("0");
183 timeStringBuilder.Append(seconds);
184
185 return timeStringBuilder.ToString();
186 }

274 public static Type GetSuperInterfaceType<T>(this Type implementation)
275 {
276 // Making sure that the implementation type is not an interface.
277 if (!LoggingService.RequireTrue(!implementation.IsInterface,
278 $"[RTSHelper] You are not allowed to use the 'GetSuperInterfaceType' helper method with an interface type."))
279 return null;
280
281 // Get all interfaces of the implemetation type, except the IGameService one
282 // Because if a higher level interface implements the IGameService one then both will appear from using the GetInterfaces() method.
283 Type interfaceType = implementation
284 .GetInterfaces()
285 .Where(nextInterface => !nextInterface.Equals(typeof(T)))
286 .FirstOrDefault(nextInterface => nextInterface.GetInterfaces().ToList().Contains(typeof(T)));
287
288 // Goal here is to find the first interface from the above collection that implements IGameService at any levels of its interface implementation levels.
289 return interfaceType.IsValid() ? interfaceType : implementation;
290 }

706 private enum SetTargetFirstManyType { toQueue, attack, mvt, targetFirst }
707
708 public static void SetTargetFirstMany(this IEnumerable<IEntity> entities, SetTargetInputData input)
709 {
710 if (!input.isMoveAttackRequest && entities.IsLocalPlayerFaction())
711 input.isMoveAttackRequest = (AttackMgr.CanAttackMoveWithKey && input.playerCommand);
712
713 var entityGroups = entities
714 .GroupBy(entity =>
715 {
716 if (entity.TasksQueue.IsValid() && entity.TasksQueue.CanAdd(input))
717 return SetTargetFirstManyType.toQueue;
718 else if (entity.CanAttack
719 && entity.FirstActiveAttackComponent.IsTargetValid(input) == ErrorMessage.none)
720 return SetTargetFirstManyType.attack;
721 else if (input.includeMovement && entity.CanMove(input.playerCommand))
722 return SetTargetFirstManyType.mvt;
723 else
724 return SetTargetFirstManyType.targetFirst;
725 });
726
727 foreach(var group in entityGroups)
728 {
729 switch(group.Key)
730 {
731 case SetTargetFirstManyType.toQueue:
732 foreach (IEntity entity in group)
733 entity.SetTargetFirst(input);
734
735 break;
736 case SetTargetFirstManyType.attack:
737
738 AttackMgr.LaunchAttack(
739 new LaunchAttackData<IReadOnlyList<IEntity>>
740 {
741 source = group.ToList(),
742
743 targetEntity = input.target.instance as IFactionEntity,
744 targetPosition = input.target.instance.IsValid() ? input.target.instance.transform.position : input.target.position,
745
746 playerCommand = input.playerCommand,
747 isMoveAttackRequest = input.isMoveAttackRequest
748 });
749
750 break;
751
752 case SetTargetFirstManyType.mvt:
753
754 MvtMgr.SetPathDestination(
755 new SetPathDestinationData<IReadOnlyList<IEntity>>
756 {
757 source = group.ToList(),
758 destination = input.target.position,
759 offsetRadius = 0.0f,
760 target = input.target.instance,
761 mvtSource = new MovementSource {
762 playerCommand = input.playerCommand,
763 isMoveAttackRequest = input.isMoveAttackRequest
764 }
765 });
766
767 break;
768
769 default:
770
771 foreach (IEntity entity in group)
772 entity.SetTargetFirst(input);
773
774 break;
775 }
776 }
777 }
