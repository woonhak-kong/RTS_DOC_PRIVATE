188 #region Handling Progress
189 private void EnableProgress()
190 {
191 audioMgr.PlaySFX(factionEntity.AudioSourceComponent, progressEnabledAudio.Fetch(), loop: true);
192
193 if (nextProgressDataEnabled
194 && (!nextProgressDataTargetEnabled || Target.instance.Equals(nextProgressDataTarget)))
195 {
196 // next data custom progress timer
197 progressTimer.Reload(nextProgressData.progressTime);
198 }
199 else
200 {
201 progressTimer.Reload(); //start timer with default value
202 }
203
204 // Disable next progress data for next EnableProgress call.
205 nextProgressDataEnabled = false;
206
207 InProgress = true; //the unit's job is now in progress
208
209 OnInProgressEnabledEffects();
210
211 if (factionEntity.AnimatorController.IsValid())
212 factionEntity.AnimatorController.LockState = true;
213
214 if (factionEntity.CanMove() && StopMovementOnProgressEnabled)
215 factionEntity.MovementComponent.Stop();
216
217 if (factionEntity.AnimatorController.IsValid())
218 {
219 factionEntity.AnimatorController.LockState = false;
220
221 factionEntity.AnimatorController.SetOverrideController(progressOverrideController.Fetch());
222
223 factionEntity.AnimatorController.SetState(AnimatorState.inProgress);
224 }
225
226 RaiseProgressStart();
227 OnInProgressEnabled();
228 }
