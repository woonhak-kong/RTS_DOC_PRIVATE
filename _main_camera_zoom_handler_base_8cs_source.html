161 protected virtual void ApplyTransformZoom(Vector3 middleScreenTerrainPoint, Vector3 zoomDirection, Vector3 targetDirection)
162 {
163 // Hold the last camera position so that we restore it later if the height leaves the allowed boundaries
164 Vector3 lastCamPos = cameraController.MainCamera.transform.position;
165
166 // Handling terrain height offset elevation
167 if (middleScreenTerrainPoint.y >= offsetMinTerrainHeight)
168 {
169 // Only enable adding a new offset if there is not one active at the moment...
170 // ... or if there is one already active and it reached the target offset elevation (lastOffset <= 0.0f) and the new one is higher than the last accumulated ones
171 if (!isAddingOffset || (lastOffset <= 0.0f && (middleScreenTerrainPoint.y - offsetMinTerrainHeight - accumOffset) > 0.5f))
172 {
173 // The new target height elevation (lastOffset) is the new terrain height difference minus the last
174 lastOffset = (middleScreenTerrainPoint.y - offsetMinTerrainHeight - accumOffset);
175 accumOffset += lastOffset;
176 isAddingOffset = true;
177 }
178 }
179 else if (isAddingOffset)
180 {
181 lastOffset = accumOffset - lastOffset;
182 // When activating the height offset elevation then reset the accumulated offsets values so that we can substract later when deactivating height offset elevation
183 accumOffset = 0;
184 isAddingOffset = false;
185 }
186 if (lastOffset > 0.0f)
187 {
188 float change = Time.deltaTime * zoomSpeed.value;
189 lastOffset -= change;
190 targetDirection += (isAddingOffset ? -1.0f : 1.0f) * change * cameraController.MainCamera.transform.forward;
191 }
192
193 // Handling actual zooming in/out
194 lastZoomValue = Mathf.Lerp(lastZoomValue, zoomValue, zoomSpeed.smoothFactor);
195 if ((lastCamPos.y > minHeight && lastZoomValue > 0.0f) || (lastCamPos.y < maxHeight && lastZoomValue < 0.0f))
196 targetDirection += zoomSpeed.value * lastZoomValue * zoomDirection;
197
198 // Updating the camera height by adding the target movement direction
199 cameraController.MainCamera.transform.position += targetDirection;
200
201 // Apply zooming limit
202 if (cameraController.MainCamera.transform.position.y < minHeight || cameraController.MainCamera.transform.position.y > maxHeight)
203 {
204 lastCamPos.y = Mathf.Clamp(lastCamPos.y, minHeight, maxHeight);
205 cameraController.MainCamera.transform.position = lastCamPos;
206 }
207
208 if (lastZoomValue != zoomValue)
209 cameraController.RaiseCameraTransformUpdated();
210
211 zoomValue = 0.0f;
212 }

214 private void ApplyFOVZoom(Vector3 middleScreenTerrainPoint)
215 {
216 lastZoomValue = Mathf.Clamp01(lastZoomValue + zoomValue);

223 targetHeight += middleScreenTerrainPoint.y - offsetMinTerrainHeight;
224

226

242 private void HandleNearMinHeightPivot()
243 {
244 if (!pivotNearMinHeight)
245 return;
246
247 float currHeight = cameraController.MainCamera.transform.position.y;
248 float heightDiff = currHeight - minHeight;
249
250 // Can only enabling pivoting near min height when the camera is in initial rotation values
251 // So if the player manipulates the rotation beforehand, no pivoting towards min height can occur
252 if (!isPivoting && heightDiff < minHeight && cameraController.RotationHandler.HasInitialRotation)
253 {
254 isPivoting = true;
255 // Calculate the target camera position to be set when min height is reached in case no camera movement occurs
256 pivotCameraPositionTargetRef = new Vector3(
257 cameraController.MainCamera.transform.position.x - pivotMoveBackwardsDistance,
258 0.0f,
259 cameraController.MainCamera.transform.position.z - pivotMoveBackwardsDistance);
260
261 // Stop following any target entity
262 cameraController.PanningHandler.SetFollowTarget(null);
263 }
264
265 if (isPivoting && heightDiff >= minHeight)
266 {
267 // Disabling pivoting to min height resets the rotation back to the initial rotation values since it can only start from initial rotation values
268 cameraController.RotationHandler.ResetRotation(smooth: true);
269 isPivoting = false;
270 }
271
272 if (isPivoting)
273 {
274 float heightPercentage = (currHeight - minHeight) / (pivotMinHeightRange);
275
276 // In case the player moves the camera while pivoting towards min height, recalculate the reference position to be set when min height is reached
277 pivotCameraPositionTargetRef += Quaternion.Euler(new Vector3(0f, cameraController.MainCamera.transform.eulerAngles.y, 0f)) * cameraController.PanningHandler.LastPanDirection * cameraController.PanningHandler.PanningSpeed.value * Time.deltaTime;
278 Vector3 nextCameraPosition = new Vector3(
279 pivotCameraPositionTargetRef.x + (pivotMoveBackwardsDistance * heightPercentage),
280 cameraController.MainCamera.transform.position.y,
281 pivotCameraPositionTargetRef.z + (pivotMoveBackwardsDistance * heightPercentage)
282 );
283 nextCameraPosition.x = Mathf.Clamp(nextCameraPosition.x, pivotCameraPositionTargetRef.x, pivotCameraPositionTargetRef.x + pivotMoveBackwardsDistance);
284 nextCameraPosition.z = Mathf.Clamp(nextCameraPosition.z, pivotCameraPositionTargetRef.z, pivotCameraPositionTargetRef.z + pivotMoveBackwardsDistance);
285 cameraController.MainCamera.transform.position = Vector3.Lerp(cameraController.MainCamera.transform.position, nextCameraPosition, cameraController.PanningHandler.PanningSpeed.value * Time.deltaTime);
286
287 Vector3 nextEulerAngles = cameraController.MainCamera.transform.rotation.eulerAngles;
288 float nextAngle = pivotTargetAngle + (cameraController.RotationHandler.InitialEulerAngles.x - pivotTargetAngle) * heightPercentage;
289 nextEulerAngles.x = Mathf.Lerp(nextEulerAngles.x, nextAngle, cameraController.RotationHandler.RotationSpeed.value * Time.deltaTime);
290 nextEulerAngles.x = Mathf.Clamp(nextEulerAngles.x, pivotTargetAngle, cameraController.RotationHandler.InitialEulerAngles.x);
291 cameraController.MainCamera.transform.rotation = Quaternion.Euler(nextEulerAngles);
292
293 if (cameraController.RotationHandler.IsRotating)
294 isPivoting = false;
295 }
