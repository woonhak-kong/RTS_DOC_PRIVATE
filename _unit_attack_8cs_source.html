87 private void HandleMovementStart(IMovementComponent sender, MovementEventArgs args)
88 {
89 // This movement source is neither the start of a new attack-move command nor it is part of an exisitng one
90 if (!IsAttackMoveEnabled
91 || (!args.Source.inMoveAttackChain && !args.Source.isMoveAttackRequest))
92 {
93 // disable attack move in case it was enabled before
94 DisableAttackMove();
95 return;
96 }
97
98 // If the movement source is part of an attack-move chain but it is not the initiator of that chain
99 // See if this movement (target) is the original attack-move chain initiator or not, if yes then update the target destination but not the source (since the movement manager recalculates it slightly differently)
100 else if (args.Source.inMoveAttackChain && !args.Source.isMoveAttackRequest)
101 {
102 if (args.Source.isMoveAttackSource)
103 lastAttackMoveTarget = unit.MovementComponent.Target;
104
105 return;
106 }
107
108 // At this point this movement start command is the initiator of a new attack-move chain
109 // Therefore, we set the "isOriginalAttackMove" boolean to true and cache the target and source of this attack-move initiator command
110 lastAttackMoveSource = new MovementSource
111 {
112 sourceTargetComponent = args.Source.sourceTargetComponent,
113
114 targetAddableUnit = args.Source.targetAddableUnit,
115 targetAddableUnitPosition = args.Source.targetAddableUnitPosition,
116
117 playerCommand = false,
118
119 isMoveAttackRequest = false,
120 inMoveAttackChain = true,
121 isMoveAttackSource = true,
122 };
123 lastAttackMoveTarget = unit.MovementComponent.Target;
124
125 // Disable idle only target search so that the attack unit can move and search for potential attack targets.
126 TargetFinder.IdleOnly = false;
127
128 IsAttackMoveActive = true;
129 }

132 {
133 if (!IsAttackMoveActive

139 DisableAttackMove();
140 }

149 #region Engaging Target
150 protected override bool MustStopProgress()
151 {
152 if (base.MustStopProgress())
153 return true;
154
155 //attacker can not move
156 else if(!unit.CanMove())
157 {
158 // Target has already entered the attacking range but it is no longer there or it is blocked by an obstacle.
159 if (IsInTargetRange
160 && (!IsTargetInRange(transform.position, Target)))
161 return true;
162 }
163 //attacker unit is movable
164 else
165 {
166 //attacker has a unit as a target (movable target) and it is currently moving.
167 if (Target.instance.IsValid()
168 && Target.instance.CanMove() && Target.instance.MovementComponent.HasTarget)
169 {
170 //the target leaves the allowed follow distance of the attacker after the target being once in the attack range.
171 if(IsInTargetRange
172 && Vector3.Distance(transform.position, RTSHelper.GetAttackTargetPosition(Target)) > Mathf.Max(followDistance, initialEngagementDistance))
173 {
174 //stop attack as the attacker can not follow its target anymore.
175 return true;
176 }
177
178 // Either attacker is not moving and it is not inside the attack range.
179 // Or target is now blocked by an obstacle
180 // Or Target might have moved but it is still inside the attacking range but it might have moved enough to trigger a re-calculation for the attack position
181 if( (!unit.MovementComponent.HasTarget && !IsTargetInRange(transform.position, Target))
182 || LineOfSight.IsInSight(Target) != ErrorMessage.none
183 || formation.MustUpdateAttackPosition(Target.opPosition, RTSHelper.GetAttackTargetPosition(Target), unit.MovementComponent.Destination, Target.instance))
184 {
185 TargetData<IFactionEntity> lastTarget = new TargetData<IFactionEntity> { instance = Target.instance, position = Target.instance.transform.position };
186
187 SetTarget(lastTarget, playerCommand: false);
188
189 return false;
190 }
191 else if(LineOfSight.IsAngleBlocked(transform.position, transform.rotation, Target.instance.transform.position))
192 {
193 unit.MovementComponent.UpdateRotationTarget(Target.instance, Target.instance.transform.position);
194
195 return false;
196 }
197 }
198 }
199
200 return false;
201 }

269 #region Searching/Updating Target
270 public override ErrorMessage SetTargetLocal (SetTargetInputData input)
271 {
272 // input.target.position: Generated attack path destination
273 // input.target.opPosition: Target position when the attack order is made
274 ErrorMessage errorMessage = ErrorMessage.none;
275 // Error message on attack position out of range
276
277 // Unit can already attack from its position, inform AttackManager about it (which might have called this method).
278 if (IsTargetInRange(transform.position, input.target))
279 errorMessage = ErrorMessage.attackAlreadyInPosition;
280 else if (!IsTargetInRange(input.target.position, input.target)) //check if the attack position is outside the unit's attacking range.
281 {
282 //if we're allowed to move even if the attack position is out of range then do it.
283 if (unit.MovementComponent.IsActive && input.playerCommand)
284 {
285 //move towards attack position without attacking the target.
286 //bypass MovementManager and directly move unit (even in multiplayer games) since this is called in a local synced method.
287 unit.MovementComponent.OnPathDestination(input.target, new MovementSource { playerCommand = false });
288
289 errorMessage = ErrorMessage.attackMoveToTargetOnly; //if an attack unit is supposed to move even if it is out of range then no error is produced.
290 }
291 else
292 errorMessage = ErrorMessage.attackPositionOutOfRange;
293 }
294
295 if (errorMessage != ErrorMessage.attackAlreadyInPosition && errorMessage != ErrorMessage.none)
296 {
297 if (input.playerCommand && RTSHelper.IsLocalPlayerFaction(factionEntity))
298 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
299 {
300 message = errorMessage,
301
302 source = Entity,
303 target = input.target.instance
304 });
305
306 return errorMessage;
307 }
308
309 base.SetTargetLocal(input);
310
311 bool updateRotation = true; //update rotation directly instead of allowing the UnitMovement component to update it (in case unit should not be moved)
312
313 if (unit.MovementComponent.IsActive && errorMessage != ErrorMessage.attackAlreadyInPosition) //only if the current unit's position is not valid for the attack
314 {
315 updateRotation = false;
316
317 if (!moveOnAttack || !unit.MovementComponent.HasTarget)
318 {
319 //move towards attack position and supply attack-move mode
320 unit.MovementComponent.OnPathDestination(
321 input.target,
322 new MovementSource
323 {
324 sourceTargetComponent = this,
325 playerCommand = input.playerCommand,
326 isMoveAttackRequest = input.isMoveAttackRequest,
327 inMoveAttackChain = IsAttackMoveActive
328 }
329 );
330 }
331 }
332 else if(!moveOnAttack)
333 {
334 // current unit position is valid for attack, do not move but stop unit from moving in case it was.
335 // stop unit from moving in case they were already moving.
336 // in order to not disable the attack, we enable the below boolean field
337 disableStopOnMovementSetIdleSource = true;
338 unit.MovementComponent.Stop();
339 disableStopOnMovementSetIdleSource = false;
340 }
341
342 if (updateRotation)
343 unit.MovementComponent.UpdateRotationTarget(Target.instance, Target.instance.IsValid() ? Target.instance.transform.position : Target.opPosition);
344
345 return errorMessage;
346 }
