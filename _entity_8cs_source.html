323 //the assumption here is that the targetComponent is attached to an empty prefab game object that includes no additional components!
324 //initTime is set to true when this method is called from the initializer method of the Entity, in that case, no need to init the new component/re-fetch components
325 //if this method is called from outside the Init() method of this class, then the initTime must be set to false so that components can be refetched and the new component is initialized
326 public void UpgradeComponent(UpgradeElement<IEntityComponent> upgradeElement)
327 {
328 //get the component to be upgraded, destroy it and replace it with the target upgrade component
329 //both components must be valid
330 if (!upgradeElement.target.IsValid())
331 return;
332
333 RTSHelper.TryGetEntityComponentWithCode(this, upgradeElement.sourceCode, out IEntityComponent sourceComponent);
334
335 //since components with their field values can not be added directly, we create a child object of the entity with the upgraded component
336 Transform newComponentTransform = Instantiate(upgradeElement.target.gameObject).transform;
337 newComponentTransform.SetParent(transform, true);
338 newComponentTransform.transform.localPosition = Vector3.zero;
339 IEntityComponent newEntityComponent = newComponentTransform.GetComponent<IEntityComponent>();
340
341 if (AnimatorController.IsValid())
342 AnimatorController.LockState = true;
343
344 RaiseEntityComponentUpgraded(new EntityComponentUpgradeEventArgs(sourceComponent, newEntityComponent));
345
346 UnsubFromEvents();
347
348 if (IsInitialized)
349 {
350 // Disable the old entity component flagging it to make it unfetchable // this can happen in the upgrade method itself
351 // Fetch mvt, attack and special entity components through the fetched entity components that do not include disabled ones
352 FetchEntityComponents(sourceComponent);
353 FetchComponents();
354
355 SubToEvents();
356
357 newEntityComponent.OnEntityPostInit(gameMgr, this);
358 if (sourceComponent.IsValid())
359 newEntityComponent.HandleComponentUpgrade(sourceComponent);
360 }
361
362 //disable old component
363 if (sourceComponent.IsValid())
364 {
365 if(IsInitialized) // Only disable the component if the entity has been initialized
366 sourceComponent.Disable();
367
368 DestroyImmediate(sourceComponent as UnityEngine.Object);
369 }
370
371 if (AnimatorController.IsValid())
372 AnimatorController.LockState = false;
373 }

544 public ErrorMessage SetTargetFirst (SetTargetInputData input)
545 {
546 if (!input.isMoveAttackRequest && this.IsLocalPlayerFaction())
547 input.isMoveAttackRequest = (attackMgr.CanAttackMoveWithKey && input.playerCommand);
548
549 if (CanAttack && FirstActiveAttackComponent.IsTargetValid(input) == ErrorMessage.none)
550 {
551 if(TasksQueue.IsValid() && TasksQueue.CanAdd(input))
552 {
553 return TasksQueue.Add(new SetTargetInputData
554 {
555 componentCode = FirstActiveAttackComponent.Code,
556
557 target = input.target,
558 playerCommand = input.playerCommand,
559
560 includeMovement = input.includeMovement
561 });
562 }
563
564 return attackMgr.LaunchAttack(
565 new LaunchAttackData<IEntity>
566 {
567 source = this,
568 targetEntity = input.target.instance as IFactionEntity,
569 targetPosition = input.target.instance.IsValid() ? input.target.instance.transform.position : input.target.position,
570 playerCommand = input.playerCommand,
571 isMoveAttackRequest = input.isMoveAttackRequest
572 });
573 }
574 else if (input.includeMovement && CanMove(input.playerCommand))
575 {
576 if(TasksQueue.IsValid() && TasksQueue.CanAdd(input))
577 {
578 return TasksQueue.Add(new SetTargetInputData
579 {
580 componentCode = MovementComponent.Code,
581
582 target = input.target,
583 playerCommand = input.playerCommand,
584
585 includeMovement = input.includeMovement
586 });
587 }
588
589 return mvtMgr.SetPathDestination(
590 new SetPathDestinationData<IEntity>
591 {
592 source = this,
593 destination = input.target.position,
594 offsetRadius = input.target.instance.IsValid() ? input.target.instance.Radius : 0.0f,
595 target = input.target.instance,
596 mvtSource = new MovementSource
597 {
598 playerCommand = input.playerCommand,
599 isMoveAttackRequest = input.isMoveAttackRequest
600 }
601 });
602 }
603 else
604 {
605 ErrorMessage notInvalidErrorMsg = ErrorMessage.invalid;
606 foreach (IEntityTargetComponent comp in entityTargetComponents)
607 {
608 if (comp.IsActive
609 && comp != MovementComponent
610 && comp != FirstActiveAttackComponent)
611 {
612 if(comp.IsTargetValid(input, out ErrorMessage lastErrorMsg))
613 return comp.SetTarget(input);
614
615 if (lastErrorMsg != ErrorMessage.invalid && lastErrorMsg != ErrorMessage.uninteractable)
616 notInvalidErrorMsg = lastErrorMsg;
617 }
618 }
619
620 if (input.playerCommand && RTSHelper.IsLocalPlayerFaction(this) && notInvalidErrorMsg != ErrorMessage.none)
621 {
622 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
623 {
624 message = notInvalidErrorMsg,
625
626 source = this,
627 target = input.target.instance
628 });
629 return notInvalidErrorMsg;
630 }
631
632 }
633
634 return ErrorMessage.failed;
635 }
