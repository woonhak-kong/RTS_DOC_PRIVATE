205 private void CalculateContour(IReadOnlyList<Vector2> points)
206 {
207
208 // locate lower-leftmost point
209 int hull = 0;
210 int i;
211 for (i = 1; i < points.Count; i++)
212 {
213 if (ComparePoint(points[i], points[hull]))
214 {
215 hull = i;
216 }
217 }
218
219 // wrap contour
220 var outIndices = new int[points.Count];
221 int endPt;
222 i = 0;
223 do
224 {
225 outIndices[i++] = hull;
226 endPt = 0;
227 for (int j = 1; j < points.Count; j++)
228 {
229 if (hull == endPt || IsLeft(points[hull], points[endPt], points[j]))
230 {
231 endPt = j;
232 }
233 }
234
235 hull = endPt;
236 } while (endPt != outIndices[0]);
237
238 // build countour points
239 int results = i;
240 for (i = 0; i < results; i++)
241 {
242 _polyPoints[i] = points[outIndices[i]];
243 }
244 }

246 private bool ComparePoint(Vector2 a, Vector2 b)
247 {
248 if (a.x < b.x) return true;
249 if (a.x > b.x) return false;
250 if (a.y < b.y) return true;
251 if (a.y > b.y) return false;

253 return false;

256 private bool IsLeft(Vector2 a, Vector2 b, Vector2 c)
257 {
258 var u1 = b.x - a.x;
259 var v1 = b.y - a.y;
260 var u2 = c.x - a.x;
261 var v2 = c.y - a.y;
