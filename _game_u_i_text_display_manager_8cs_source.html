38 public virtual bool FactionEntityRequirementsToString(IEnumerable<FactionEntityRequirement> reqs, out string text)
39 {
40 text = String.Empty;
41
42 if (!reqs.IsValid() || !reqs.Any())
43 return false;
44
45 List<FactionEntityRequirement> requirementList = reqs.ToList();
46
47 StringBuilder builder = new StringBuilder();
48
49 for(int i = 0; i < requirementList.Count; i++)
50 {
51 FactionEntityRequirement req = requirementList[i];
52
53 string reqAmountColored = req.TestFactionEntityRequirement(gameMgr.LocalFactionSlot.FactionMgr)
54 ? $"<color=green>{req.amount}</color>"
55 : $"<color=red>{req.amount}</color>";
56
57 builder.Append($"<b>{req.name}</b>: {reqAmountColored}");
58
59 if (i < requirementList.Count - 1)
60 builder.Append(" -");
61 }
62
63 text = builder.ToString();
64
65 return true;
66 }

79 public virtual bool ResourceInputToString(IEnumerable<ResourceInput> resourceInputs, out string text)
80 {
81 text = String.Empty;
82
83 if (!resourceInputs.IsValid() || !resourceInputs.Any())
84 return false;
85
86 List<ResourceInput> resourceInputList = resourceInputs.ToList();
87
88 StringBuilder builder = new StringBuilder();
89
90 for(int i = 0; i < resourceInputList.Count; i++)
91 {
92 ResourceInput input = resourceInputList[i];
93
94 if (!input.type.IsValid())
95 continue;
96
97 ResourceTypeValueToString(input.value, out string inputAmountText);
98
99 string inputAmountTextColored = resourceMgr.HasResources(input, gameMgr.LocalFactionSlot.ID)
100 ? $"<color=green>{inputAmountText}</color>"
101 : $"<color=red>{inputAmountText}</color>";
102
103 builder.Append($"<b>{input.type.DisplayName}</b>: {inputAmountTextColored}");
104
105 if (i < resourceInputList.Count - 1)
106 builder.Append(" -");
107 }
108
109 text = builder.ToString();
110
111 return true;
112 }
