159 public NPCBuildingRegulator ActivateBuildingRegulator(IBuilding buildingPrefab, IBuilding buildingCenter)
160 {
161 if (!logger.RequireValid(buildingPrefab,
162 $"[{GetType().Name} - {factionMgr.FactionID}] Can not activate a regulator for an invalid building prefab! Check the 'Independent Buildings' list for unassigned elements or any other building input field in other NPC components or provide a valid building if you are calling the activation method from an external component.")
163 || !logger.RequireTrue(buildingCenterRegulators.ContainsKey(buildingCenter),
164 $"[{GetType().Name} - {factionMgr.FactionID}] The provided building center has not been registered as a valid building center for this NPC faction!"))
165 return null;
166
167 // If no valid regulator data for the building is returned then do not continue
168 NPCBuildingRegulatorData regulatorData = buildingPrefab.GetComponent<NPCBuildingRegulatorDataInput>()?.GetFiltered(factionType: factionSlot.Data.type, npcType: npcMgr.Type);
169 if (!regulatorData.IsValid())
170 {
171 if (defaultRegulatorData.IsValid())
172 regulatorData = defaultRegulatorData;
173 else
174 {
175 LogEvent($"{buildingPrefab.Code}: Unable to find a valid regulator for building.");
176 return null;
177 }
178 }
179
180 // If there is a regulator for the provided building type that is already active on the provided building center then return it directly.
181 NPCBuildingRegulator activeInstance = GetBuildingRegulator(buildingPrefab.Code, buildingCenter);
182 if (activeInstance.IsValid())
183 {
184 LogEvent($"{buildingPrefab.Code}: A valid active regulator already exists!");
185 return activeInstance;
186 }
187
188 // At this stage, we create a new regulator for the building to be added to the provided building center.
189 NPCActiveBuildingRegulatorData newBuildingRegulator = new NPCActiveBuildingRegulatorData()
190 {
191 instance = new NPCBuildingRegulator(regulatorData, buildingPrefab, gameMgr, npcMgr, buildingCenter),
192
193 // Initial spawning timer: regular spawn reload + start creating after delay
194 spawnTimer = new TimeModifiedTimer(regulatorData.SpawnReload + regulatorData.CreationDelayTime)
195 };
196
197 newBuildingRegulator.instance.AmountUpdated += HandleBuildingRegulatorAmountUpdated;
198
199 buildingCenterRegulators[buildingCenter].activeBuildingRegulators.Add(newBuildingRegulator);
200 buildingCenterRegulators[buildingCenter].activeBuildingRegulatorsDic.Add(buildingPrefab.Code, newBuildingRegulator);
201
202 if (!regulatedBuildingPrefabs.Contains(buildingPrefab))
203 regulatedBuildingPrefabs.Add(buildingPrefab);
204
205 // Whenever a new regulator is added to the active regulators list, then move the building creator into the active state
206 IsActive = true;
207
208 LogEvent($"{buildingPrefab.Code}: Created and activated a regulator for building. Active Border: {buildingCenter}. Using default regulator data? {regulatorData == defaultRegulatorData}");
209 return newBuildingRegulator.instance;
210 }

221 public NPCBuildingRegulator GetCreatableBuildingRegulatorFirst(string buildingCode, bool creatableOnDemandOnly)
222 {
223 // The first creatable building regulator is the one that can be created on demand (if it is a requirement) and whose max amount has not been reached yet.
224 // Therefore, we go through all building center regulators and get the first building regulator that satisfies the above two conditions
225
226 for (int i = 0; i < buildingCenters.Count; i++)
227 {
228 NPCBuildingCenterRegulatorData nextBuildingCenterRegulator = buildingCenterRegulators[buildingCenters[i]];
229 if (nextBuildingCenterRegulator.activeBuildingRegulatorsDic.TryGetValue(buildingCode, out NPCActiveBuildingRegulatorData nextBuildingRegulator))
230 if ((!creatableOnDemandOnly || nextBuildingRegulator.instance.Data.CanCreateOnDemand) && !nextBuildingRegulator.instance.HasReachedMaxAmount)
231 return nextBuildingRegulator.instance;
232 }
233
234
235 return null;

321 IsActive = false;

323 for (int i = 0; i < buildingCenters.Count; i++)

329 // Buildings are only automatically created if they haven't reached their min amount and still haven't reached their max amount

363 private bool OnCreateBuildingRequestInternal(NPCBuildingRegulator instance, IBuilding buildingCenter = null)
364 {
365 if (!instance.IsValid())
366 return false;
367 if (instance.HasReachedMaxAmount)
368 {
369 LogEvent($"{instance.Prefab.Code}: Unable to create instance. Has reached max amount? {instance.HasReachedMaxAmount}");
370 return false;
371 }
372
373 IReadOnlyList<string> builderCodes = npcBuildingConstructor.GetBuildingBuilderCodes(instance.Prefab.Code);
374 if (!builderCodes.IsValid())
375 {
376 LogEvent($"{instance.Prefab.Code}: Can not find valid builder types for the building prefab to start placing it!");
377 return false;
378 }
379 else if (!RTSHelper.TestFactionEntityRequirements(instance.Data.FactionEntityRequirements, factionMgr))
380 {
381 LogEvent($"{instance.Prefab.Code}: Unable to fullfill faction entity requirements!");
382 return false;
383 }
384
385 // Try to get a valid building creation task from the builder units to create the provided building type
386 BuildingCreationTask nextBuildingCreationTask = null;
387 for (int i = 0; i < builderCodes.Count; i++)
388 {
389 NPCUnitRegulator nextBuilderRegulator;
390 if ((nextBuilderRegulator = npcUnitCreator.GetActiveUnitRegulator(builderCodes[i])).IsValid())
391 {
392 IBuilder nextBuilder = null;
393 int j = 0;
394 while (j < nextBuilderRegulator.Instances.Count)
395 {
396 if (nextBuilderRegulator.Instances[j].BuilderComponent.IsValid())
397 {
398 nextBuilder = nextBuilderRegulator.Instances[j].BuilderComponent;
399 break;
400 }
401 j++;
402 }
403
404 if (nextBuilder.IsValid())
405 for (int t = 0; t < nextBuilder.CreationTasks.Count; t++)
406 {
407 if (nextBuilder.CreationTasks[t].Prefab.Code == instance.Prefab.Code)
408 {
409 nextBuildingCreationTask = nextBuilder.CreationTasks[t];
410 break;
411 }
412 }
413 }
414 }
415
416 if (!nextBuildingCreationTask.IsValid())
417 {
418 LogEvent($"{instance.Prefab.Code}: No valid building creation task has been found!");
419 return false;
420 }
421
422 // In case the building center where the building will be placed has not been specified then attempt to pick one
423 if (!buildingCenter.IsValid()
424 && !factionMgr.BuildingCenters.GetEntityFirst(out buildingCenter, building => building.BorderComponent.IsBuildingAllowedInBorder(instance.Prefab as IBuilding)))
425 {
426 //FUTURE FEATURE -> no building center is found -> request to place a building center.
427 LogEvent($"{instance.Prefab.Code}: Attempting to start placement but no valid building center is found!");
428 return false;
429 }
430
431 if (instance.Data.ForcePlaceAround && instance.Data.AllPlaceAroundData.Count == 0)
432 {
433 LogEvent($"{instance.Prefab.Code}: Attempting to start placement with place around parameters forced but no place around data has been defined in the regulator!");
434 return false;
435 }
436
437 // Either take the pre defined place around data or if none exists (and we have made sure that force place around is not set), we define the place around data around the building center
438 nextPlaceAroundDataSet.Clear();
439 nextPlaceAroundDataSet.AddRange(instance.Data.AllPlaceAroundData);
440 if(instance.Data.ForcePlaceAround)
441 {
442 nextPlaceAroundDataSet.Add(new BuildingPlaceAroundData
443 {
444 entityType = new CodeCategoryField { codes = new string[] { buildingCenter.Code } },
445 range = new FloatRange(0.0f, buildingCenter.BorderComponent.Size)
446 });
447 }
448
449 LogEvent($"{instance.Prefab.Code}: Sending building placement request");
450 npcBuildingPlacer.OnBuildingPlacementRequest(
451 nextBuildingCreationTask,
452 buildingCenter,
453 nextPlaceAroundDataSet.AsReadOnly(),
454 instance.Data.CanRotate);
455
456 return true;
457 }
