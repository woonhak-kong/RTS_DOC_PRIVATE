78 #region Faction Resources Manipulation
79 private void AddBorderResource (IBuilding buildingCenter, IResource newResource)
80 {
81 // when border is activated then we'd send some of them to the resources to collect list of the NPC Resource Collector ...
82 // ...and leave the rest idle and that is all depending on the resource exploit ratio defined in this component:
83
84 if (!borderResourceTrackers.TryGetValue(buildingCenter, out NPCBorderResourceTracker nextBorderResourceTracker))
85 {
86 nextBorderResourceTracker = new NPCBorderResourceTracker();
87
88 borderResourceTrackers.Add(
89 buildingCenter,
90 nextBorderResourceTracker);
91 }
92
93 if (nextBorderResourceTracker.Add(newResource, resourceDefaultExploitRatioRange.RandomValue))
94 npcResourceCollector.AddResourceToCollect(newResource);
95
96 LogEvent($"Added resource of type {newResource.Code} to Border {buildingCenter.Code} territory!");
97 }
98
99 private bool RemoveBorderResource (IBuilding buildingCenter, IResource removedResource)
100 {
101 if (!borderResourceTrackers.TryGetValue(buildingCenter, out NPCBorderResourceTracker nextBorderResourceTracker))
102 {
103 nextBorderResourceTracker.Remove(removedResource);
104
105 npcResourceCollector.RemoveResourceToCollect(removedResource);
106 LogEvent($"Removed resource of type {removedResource.Code} from Border {buildingCenter.Code} territory!");
107
108 return true;
109 }
110
111 return false;
112 }

114 private void RemoveAllResourcesInBorder (IBuilding buildingCenter)

161 #region Handling Missing Resources
162 public void OnIncreaseMissingResourceRequest(IEnumerable<ResourceInput> resourceInputs)
163 {
164 // Currently, this only treats capacity resources.
165 // FUTURE: Handle missing non-capacity resources
166 foreach(ResourceInput nextInput in resourceInputs)
167 {
168 if (!nextInput.type.HasCapacity
169 || resourceMgr.HasResources(nextInput, factionMgr.FactionID))
170 continue;
171
172 INPCCapacityResourceManager nextCapacityResourceMgr = npcCapacityResourceMgrs.Where(element => element.TargetCapacityResource == nextInput.type).FirstOrDefault();
173 if (!nextCapacityResourceMgr.IsValid())
174 {
175 LogEvent($"MISSING CAPACITY RESOURCE REQUEST: {nextInput.type.Key}! No valid capacity resource manager found!");
176 continue;
177 }
178
179 LogEvent($"MISSING (CAPACITY) RESOURCE REQUEST: {nextInput.type.Key}! Calling for increasing capacity resource!");
180 nextCapacityResourceMgr.OnIncreaseCapacityRequest();
181 }
182 }
