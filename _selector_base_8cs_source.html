164 private void HandleEntitySelectedGlobal(IEntity entity, EntitySelectionEventArgs args)
165 {
166 if (lockDoubleClick || !args.IsLocalPlayerClickSelection)
167 return;
168
169 if (doubleClickTimer > 0.0f)
170 {
171 doubleClickTimer = 0.0f;
172 lockDoubleClick = true;

174 if (!MultipleSelectionModeEnabled)
175 selectionMgr.RemoveAll();
176
177 //if this is the second click (double click), select all entities of the same type within a certain range
178 SelectEntitisInRange(entity, playerCommand: true);
179 lockDoubleClick = false;
180 return;
181 }

183 doubleClickTimer = 0.5f;

186 private IEntity nextRangeSelectionSource = null;
187 private ErrorMessage IsTargetValidForRangeSelection(SetTargetInputData data)
188 {
189 if (!nextRangeSelectionSource.IsValid() || !data.target.instance.IsValid() || !data.target.instance.Selection.IsValid())
190 return ErrorMessage.invalid;
191 else if (data.target.instance.Code != nextRangeSelectionSource.Code)
192 return ErrorMessage.entityCodeMismatch;
193 else if (!nextRangeSelectionSource.IsSameFaction(data.target.instance))
194 return ErrorMessage.factionMismatch;
195 else if (doubleClickSelectVisibleEntities)
196 {
197 Vector3 entityScreenPosition = mainCameraController.MainCamera.WorldToScreenPoint(data.target.instance.Selection.transform.position);
198
199 if (entityScreenPosition.x < 0.0f || entityScreenPosition.x > Screen.width
200 || entityScreenPosition.y < 0.0f || entityScreenPosition.y > Screen.height)
201 return ErrorMessage.positionOutOfSelectionBounds;
202 }
203
204 return ErrorMessage.none;
205 }

207 public void SelectEntitisInRange(IEntity source, bool playerCommand)
208 {
209 if (!enableDoubleClickSelect
210 || !source.IsValid()
211 || source.IsFree)
212 return;
213
214 nextRangeSelectionSource = source;
215
216 IReadOnlyList<IEntity> entitiesInRange;
217 if (doubleClickSelectVisibleEntities)
218 {
219 gridSearch.SearchVisible(
220 IsTargetValidForRangeSelection,
221 playerCommand,
222 out entitiesInRange);
223 }
224 else
225 {
226 gridSearch.Search(
227 source.transform.position,
228 doubleClickSelectRange,
229 -1,
230 IsTargetValidForRangeSelection,
231 playerCommand,
232 out entitiesInRange);
233 }
234 selectionMgr.Add(entitiesInRange);
235 }
