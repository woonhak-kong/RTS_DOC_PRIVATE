493 {

525 {
526 if (!IsActive
527 || Count == 0
528 || !current.IsValidSegmentIndex(segmentIndex))
529 return false;
530
531 // If this is a placed snapped instance then remove it directly and do nothing else
532 if (!current.snappedIndexes.Contains(segmentIndex))
533 DeleteCurrentSegment(segmentIndex);
534
535 current.RemoveAt(segmentIndex);
536
537 if (!isPlacementComplete)
538 {
539 for (int index = 0; index < current.SegmentCount; index++)
540 {
541 if (!current.instances[index].IsPlacementInstance)
542 continue;
543
544 current.instances[index].PlacerComponent.OnPlacementStop(new BuildingPlacementUpdateData
545 {
546 targetSegmentIndex = segmentIndex,
547 sourceSegmentIndex = index,
548 segmentCount = current.SegmentCount
549 });
550 }
551 }
552
553 // No more active placement instances? Disable placement

556

588 protected bool AttemptComplete()
589 {
590 if (current.segmentData.enabled)
591 {
592 if (current.segmentData.minAmount > current.SegmentCount)
593 {
594 while (current.SegmentCount > 1)
595 StopSegment(current.SegmentCount - 1, isPlacementComplete: false);
596
597 SegmentationActive = false;
598
599 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
600 {
601 message = ErrorMessage.placementSegmentAmountNotMet,
602
603 source = current.mainTask.Prefab,
604 amount = current.segmentData.minAmount
605 });
606 return false;
607 }
608 }
609
610 //if ((disablePlacementOnAnyInvalid && !SegmentationActive)
611 if(SegmentationActive && disablePlacementOnAnyInvalidSegment)
612 {
613 // Do not allow placement for all current instances if at least one them does not meet placement requirements
614 ErrorMessage errorMessage;
615 for (int index = 0; index < current.SegmentCount; index++)
616 {
617 if ((errorMessage = CanComplete(index)) != ErrorMessage.none)
618 {
619 OnComplete(errorMessage, new CompletedPlacementData { });
620 Stop();
621 return false;
622 }
623 }
624
625 // All placement instances meet placement conditions, place each one individually
626 while (IsActive && current.SegmentCount > 0)
627 CompleteOnConditionMet(segmentIndex: 0);
628
629 return true;
630 }
631
632 // Only place instances that meet conditions and stop placing instances that do not
633 while (IsActive && current.SegmentCount > 0)
634 {
635 if (!Complete(index: 0))
636 {
637 if (!SegmentationActive)
638 break;
639
640 StopSegment(segmentIndex: 0, isPlacementComplete: false);
641 }
642 }
643
644 return true;

797 protected void OnSegmentationUpdate(Vector3 finalPosition)
798 {
799 // Determine the amount of segments/parts of the building type to have depending on the distance between the starting and last drag and drop position
800 float distance = Mathf.Ceil(Vector3.Distance(finalPosition, SegmentationInitialPosition));
801
802 int segments = 1;
803 // Calculate the amount of the expected segments to be in this placement
804 // Going through every existing segment to take into account different segment lengths
805 foreach (var instance in current.instances)
806 {
807 if (distance - instance.PlacerComponent.SegmentData.segmentLength <= 0)
808 break;
809
810 distance -= instance.PlacerComponent.SegmentData.segmentLength;
811 segments++;
812 }
813
814 if (distance > 0)
815 segments += Mathf.FloorToInt((distance / current.segmentData.segmentLength));
816
817 // If the amount of segments does not match the amount of placement instances, then fix that
818 while (segments != current.SegmentCount)
819 {
820 if (segments < current.SegmentCount)
821 {
822 int stopIndex = current.SegmentCount - 1;
823 if (current.SegmentCount > 2)
824 stopIndex = current.SegmentCount - 2;
825 if (!StopSegment(stopIndex, isPlacementComplete: false))
826 break;
827 }
828 else
829 {
830 if (Add(current.mainTask, options: new BuildingPlacementOptions { isAddingSegment = true }) != ErrorMessage.none)
831 break;
832 }
833 }
834
835 // Go through all the segment instances and assign their position and rotation accordingly
836
837 // Determine the segments direction and consider snapped buildings and whether snapping direction is an option
838 segmentDirection = (finalPosition - SegmentationInitialPosition).normalized;
839
840 // Here we have snapped into placed buildings and are allowed to change the direction according to them.
841 if (current.snappedIndexes.Count > 0
842 && (segmentSnapAlignAlways || controls.Get(segmentSnapKey)))
843 {
844 Vector3 snapSegmentsDirection = current.instances[current.snappedIndexes[0]].PlacerComponent.PlacedData.segmentDirection;
845 // Use the Dot operation and the result's sign to determine the direction of the current segments
846 float dot = Vector3.Dot(snapSegmentsDirection, segmentDirection);
847 if (Mathf.Abs(dot) >= 0.5f)
848 {
849 segmentDirection = snapSegmentsDirection * Mathf.Sign(dot);
850 }
851 else
852 {
853 Vector3 nextSegmentationDirection = Vector3.Cross(
854 snapSegmentsDirection, Vector3.up).normalized;
855
856 segmentDirection = nextSegmentationDirection * Mathf.Sign(Vector3.Dot(segmentDirection, nextSegmentationDirection));
857 }
858 }
859 // In case there is no changing the direction with the snapped instances and we have grid placement enabled.
860 else if (IsGridPlacementActive)
861 {
862 if (placerMgr.GridHandler.TryGetCellPosition(finalPosition, out finalPosition))
863 {
864 // Determine whether the vertical or the horizontal axis is the closest to the current segment direction
865 // And update the direction accordingly
866 float xDistance = Mathf.Abs(finalPosition.x - SegmentationInitialPosition.x);
867 float yDistance = Mathf.Abs(finalPosition.z - SegmentationInitialPosition.z);
868
869 if (xDistance > yDistance)
870 finalPosition.z = SegmentationInitialPosition.z;
871 else
872 finalPosition.x = SegmentationInitialPosition.x;
873
874 segmentDirection = (finalPosition - SegmentationInitialPosition).normalized;
875 }
876 }
877 // Finally calculate the resulting rotation from the segments direction
878 Quaternion segmentRotation = Quaternion.LookRotation(segmentDirection);
879
880 // Holds the length of the previous segment lengths
881 float accumSegmentLength = 0;
882
883
884 Int2D currFirstArea = current.firstInstance.PlacerComponent.GridOptions.area;
885
886 for (int index = 0; index < current.SegmentCount; index++)
887 {
888 IBuilding instance = current.instances[index];
889
890 // Update the next segment position according to length and update the Y position to fit on the placable terrain
891 Vector3 nextInstancePosition = SegmentationInitialPosition;
892
893 // if the first instance is one that has been snapped
894 // we can check whether the first instance has changed from the original one and adjust the starting position of the segments accordingly
895 if (IsGridPlacementActive
896 && current.firstInstance.IsPlacementInstance
897 && (firstSegmentArea.x != currFirstArea.x
898 || firstSegmentArea.y != currFirstArea.y))
899 {
900 if (current.firstInstance.PlacerComponent.GridOptions.IsHorizontal)
901 nextInstancePosition.z += (segmentDirection.z > 0.0f ? 1 : -1) * (current.firstInstance.PlacerComponent.SegmentData.segmentLength - firstSegmentLength) / 2.0f;
902 else
903 nextInstancePosition.x += (segmentDirection.x > 0.0f ? 1 : -1) * (current.firstInstance.PlacerComponent.SegmentData.segmentLength - firstSegmentLength) / 2.0f;
904 }
905
906 // Add only half of the segment length for the initial segment
907 // And for each subsequent segment, add half of its length before updating its position
908 accumSegmentLength += instance.PlacerComponent.SegmentData.segmentLength / 2.0f;
909
910 if (index > 0)
911 {
912 // Then for segments past the initial one, we use the accumulated segment length + half of its segment length to position it
913 // Using half segment lengths here because conditions for building segment prefabs include having the transform pivot point on the botton center
914 // as well as having the parent prefab object's scale set to (1,1,1)
915 nextInstancePosition += (segmentDirection * accumSegmentLength);
916 // Then add the other half of the segment length so it is considered for positioning the next segment
917 accumSegmentLength += instance.PlacerComponent.SegmentData.segmentLength / 2.0f;
918 }
919
920 // In case of grid placement, snap the segment position into the cell position
921 if (IsGridPlacementActive)
922 {
923 // when snapping the segment position, we either
924 // get the cell position from the position of the segment (whose pivot is in the bottom middle of the segment) if the first instance is not snapped and is a non-square shape
925 // or apply reverse pivoting if not
926 if(current.firstInstance.IsPlacementInstance
927 && current.firstInstance.PlacerComponent.GridOptions.area.x == current.firstInstance.PlacerComponent.GridOptions.area.y)
928 placerMgr.GridHandler.TryGetCellPosition(nextInstancePosition, out nextInstancePosition);
929 else
930 nextInstancePosition = instance.PlacerComponent.GridOptions.ApplyPivotPointReverse(nextInstancePosition);
931 }
932 else
933 {
934 terrainMgr.SampleHeight(instance.transform.position, instance.PlacerComponent.PlacableTerrainAreas, out nextInstancePosition.y);
935 nextInstancePosition.y += placerMgr.BuildingPositionYOffset;
936 }
937
938 if (!instance.IsPlacementInstance)
939 {
940 // Check if position still matches the snapped one.
941 Vector3 expectedPosition = IsGridPlacementActive
942 ? instance.PlacerComponent.GridOptions.ApplyPivotPoint(nextInstancePosition)
943 : nextInstancePosition;
944
945 if(Vector3.Distance(expectedPosition, instance.transform.position) > segmentUnsnapRange)
946 {
947 // Unsnap
948 RevertSnappedInstance(index, nextInstancePosition, segmentRotation);
949 }
950 continue;
951 }
952
953 // Update segment rotation according to direction
954 instance.transform.rotation = segmentRotation;
955
956 // Check if the individual segment can be placed in this new position
957 instance.PlacerComponent.OnPlacementUpdate(nextInstancePosition);
958 }
959 }
960
961 #endregion
962
963 #region Helper Functions
964 // Deletes a segment from the current placement process without replacing it or clearing its slot.
965 // This is intended to be used as a helper function prior to replacing the segment slot with a snapped target building or a new segment type.
966 private bool DeleteCurrentSegment(int index)
967 {
968 if (!current.IsValidSegmentIndex(index))
969 return false;
970
971 if (reservePlacementResources)
972 gameMgr.GetService<IResourceManager>().ReleaseResources(
973 current.instanceTasks[index].RequiredResources,
974 current.instanceTasks[index].FactionID
975 );
976
977 IBuilding replacedInstance = current.instances[index];
978
979 current.instanceTasks[index].OnCancel();
980
981 if (replacedInstance.IsValid())
982 replacedInstance.Health.DestroyLocal(false, null);
983
984 globalEvent.RaiseBuildingPlacementStopGlobal(replacedInstance);
985
986 current.instanceTasks[index] = null;
987 current.instances[index] = null;
988
989 return true;
990 }
991
992 #endregion
993 }
