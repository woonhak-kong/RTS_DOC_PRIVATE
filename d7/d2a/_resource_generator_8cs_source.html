156 #region Resource Generation Action
160 private void Update()
161 {
162 // Only allow the master instance to run the timers for generating resources
163 if (!IsInitialized
164 || !FactionEntity.CanLaunchTask
165 || !IsActive
166 || (stopGeneratingOnThresholdMet && isThresholdMet))
167 return;
168
169 // Turn into Action
170 // Generating Resources:
171 if (timer.ModifiedDecrease())
172 // Directly generate resources locally on all client instances if it is a multiplayer game
173 // Because sending every resource generation increase over the network is expensive and in this case, useless
174 // since the collection will only be performed by the master instance
175 // This also ensures that the resource generation is not halted by delay due to sending data over the network
176 GeneratePeriodResourcesActionLocal(playerCommand: false);
177 }

179 private ErrorMessage GeneratePeriodResourcesAction(bool playerCommand)
180 {
181 return LaunchAction(
182 (byte)ActionType.generateResources,
183 new SetTargetInputData { playerCommand = playerCommand });
184 }
185
189 private ErrorMessage GeneratePeriodResourcesActionLocal(bool playerCommand)
190 {
191 if (!resourceMgr.HasResources(requiredResources, FactionEntity.FactionID))
192 return ErrorMessage.taskMissingResourceRequirements;
193
194 // Assume that the target threshold is met:
195 isThresholdMet = true;
196
197 for (int i = 0; i < generatedResources.Length; i++)
198 {
199 generatedResources[i].UpdateValue(resources[i].value);
200
201 // One of the resources haven't met the threshold yet? => threshold not met
202 if (isThresholdMet
203 && collectionThresholdDic.TryGetValue(resources[i].type.Key, out ResourceTypeValue thresholdValue)
204 && !generatedResources[i].Has(thresholdValue))
205 isThresholdMet = false;
206 }
207
208 // If the threshold is met and we can either autocollect the resources or this is a NPC faction:
209 if (isThresholdMet && RTSHelper.IsMasterInstance() && (autoCollect || FactionEntity.IsNPCFaction()))
210 CollectResourcesAction(playerCommand: false);
211
212 if (isThresholdMet)
213 {
214 onThresholdMet.Invoke();
215
216 globalEvent.RaiseEntityComponentTaskUIReloadRequestGlobal(this);
217 }
218
219 // Consume the required resources per period:
220 resourceMgr.UpdateResource(FactionEntity.FactionID, requiredResources, add:false);
221
222 timer.Reload();
223
224 return ErrorMessage.none;
225 }
226 #endregion

228 #region Resource Collection Action
233 private ErrorMessage CollectResourcesAction(bool playerCommand)
234 {
235 return LaunchAction(
236 (byte)ActionType.collectResources,
237 new SetTargetInputData { playerCommand = playerCommand });
238 }
239
240 private ErrorMessage CollectResourcesActionLocal(bool playerCommand)
241 {
242 // We no longer meet the threshold.
243 isThresholdMet = false;
244
245 for (int i = 0; i < generatedResources.Length; i++)
246 {
247 var nextInput = new ResourceInput
248 {
249 type = resources[i].type,
250 value = new ResourceTypeValue
251 {
252 amount = generatedResources[i].Amount,
253 capacity = generatedResources[i].Capacity
254 }
255 };
256
257
258 resourceMgr.UpdateResource(
259 FactionEntity.FactionID,
260 nextInput,
261 add:true,
262 out int restAmount);
263
264 // In case not all of the amount could be added due to a limited capacity type resource that reached its maximum capacity...
265 if(restAmount > 0)
266 {
267 if (FactionEntity.IsLocalPlayerFaction() && !hasShownLimitCapacityError)
268 {
269 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
270 {
271 message = ErrorMessage.resourceTypeLimitCapacityReached,
272 amount = restAmount,
273 text = nextInput.type.DisplayName
274 });
275
276 hasShownLimitCapacityError = true;
277 }
278 }
279 else
280 {
281 hasShownLimitCapacityError = false;
282 }
283
284 globalEvent.RaiseResourceGeneratorCollectedGlobal(this, new ResourceAmountEventArgs(nextInput));
285
286 //and reset them.
287 generatedResources[i].Reset(restAmount);
288 }
289
290 onCollected.Invoke();
291
292 if(playerCommand && Entity.IsLocalPlayerFaction())
293 audioMgr.PlaySFX(collectionAudio.Fetch(), false);
294
295 return ErrorMessage.none;
296 }
297 #endregion
