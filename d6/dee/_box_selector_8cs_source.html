116 private void Update()
117 {
118 if (gameMgr.State != GameStateType.running
119 || placementMgr.IsLocalPlayerPlacingBuilding)
120 return;
121
122 // If the player is holding the left mouse button -> potentially drawing a selection box
123 if (Input.GetMouseButton(0))
124 OnDrawingProgress();
125
126 // If the player releases the left mouse button while they were drawing the selection box
127 else if (isDrawing && Input.GetMouseButtonUp(0))
128 {
129 OnDrawingComplete();
130 Disable();
131 }
132 }
133
134 private void OnDrawingProgress()
135 {
136 // Just started drawing?
137 if (!isDrawing)
138 {
139 // Do not start drawing the box if this component does not have priority to use the UI.
140 if (!gameUIMgr.HasPriority(this)
141 || (EventSystem.current.IsPointerOverGameObject() && !canStartDrawingOnUI))
142 return;
143
144 // Initial selection box position edges
145 initialMousePosition = finalMousePosition = Input.mousePosition;
146 isDrawing = true;
147 }
148 // Continue drawing
149 else
150 finalMousePosition = Input.mousePosition;
151
152 // Do not display if we haven't reached the minimum size
153 if (Vector3.Distance(finalMousePosition, initialMousePosition) < minSize)
154 return;
155
156 // Passed the minimum required size
157
158 // Start displaying the selection box
159 if (!IsActive)
160 {
161 image.gameObject.SetActive(true);
162 IsActive = true;
163
164 // lock mouse selector so no selection can occur outside of the drawn box selector
165 selector.Lock(this);
166
167 // Reserve a slot to prioritize this service to handle its box drawing UI.
168 UIMgr.PrioritizeServiceUI(this);
169 }
170
171 image.sizeDelta = new Vector2(Mathf.Abs(finalMousePosition.x - initialMousePosition.x), Mathf.Abs(finalMousePosition.y - initialMousePosition.y));
172
173 // Center the selection box position between initial and final mouse positions and offset by the canvas position
174 image.localPosition = (initialMousePosition + finalMousePosition) / 2.0f - canvas.localPosition;
175 }
176
177 private void OnDrawingComplete()
178 {
179 if (!IsActive || Vector3.Distance(finalMousePosition, initialMousePosition) < minSize)
180 return;
181
182 // If the player is not holding down the multiple selection key down then deselect all currently selected entities
183 if (!selector.MultipleSelectionModeEnabled)
184 selectionMgr.RemoveAll();
185
186 if (!gameMgr.LocalFactionSlot.IsValid())
187 return;
188
189 lowerLeftCorner = new Vector2(Mathf.Min(finalMousePosition.x, initialMousePosition.x), Mathf.Min(finalMousePosition.y, initialMousePosition.y));
190 upperRightCorner = new Vector2(Mathf.Max(finalMousePosition.x, initialMousePosition.x), Mathf.Max(finalMousePosition.y, initialMousePosition.y));
191
192 gridSearch.SearchVisible(
193 IsTargetInSelectionBox,
194 true,
195 out IReadOnlyList<IEntity> targets);
196
197 selectionMgr.Add(targets);
198 }
199
200 private ErrorMessage IsTargetInSelectionBox(SetTargetInputData data)
201 {
202 if (data.target.instance.IsDummy
203 || !data.target.instance.IsEntityTypeMatch(selectableTypes)
204 || (localPlayerOnly && !data.target.instance.IsLocalPlayerFaction()))
205 return ErrorMessage.invalid;
206
207 Vector3 unitScreenPosition = mainCameraController.MainCamera.WorldToScreenPoint(data.target.instance.Selection.transform.position);
208
209 // Make sure the unit screen position fits inside the current selection box
210 return unitScreenPosition.x >= lowerLeftCorner.x && unitScreenPosition.x <= upperRightCorner.x
211 && unitScreenPosition.y >= lowerLeftCorner.y && unitScreenPosition.y <= upperRightCorner.y
212 ? ErrorMessage.none : ErrorMessage.invalid;
213 }
