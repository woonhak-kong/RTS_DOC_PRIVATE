166 private ErrorMessage TryAddSearchObstacle (ISearchObstacle newObstacle)
167 {
168 ErrorMessage errorMessage;
169 // Only continue if a valid source search cell is found in the input position.
170 if ((errorMessage = TryGetSearchCell(newObstacle.Center, out SearchCell sourceCell)) != ErrorMessage.none)
171 return errorMessage;
172
173 Vector2 circleCenter = new Vector2(newObstacle.Center.x, newObstacle.Center.z);
174
175 // Cells to search initially are the source cell and its direct neighbors.
176 addObstacleLists.currentSearchCells.Clear();
177 addObstacleLists.currentSearchCells.Add(sourceCell);
178 addObstacleLists.currentSearchCells.AddRange(sourceCell.Neighbors);
179
180 // We include the source cell and its direct neighbors in the already searched cells so that...
181 //... we do not add them twice when looking at the current search cells neighbors that have not been considered for search in the loop below
182 addObstacleLists.alreadySearchedCells.Clear();
183 for(int i = 0; i < sourceCell.Neighbors.Count; i++)
184 addObstacleLists.alreadySearchedCells.Add(sourceCell.Neighbors[i]);
185 addObstacleLists.alreadySearchedCells.Add(sourceCell);
186
187 // The size of the covered surface in terms of cell size
188 int coveredSurface = 0;
189
190 // As long as there cells to search
191 while(addObstacleLists.currentSearchCells.Count > 0)
192 {
193 // Holds te neighbor cells of the current cells to search so they would be searched in the next round.
194 addObstacleLists.nextSearchCells.Clear();
195 float radiusSqr = newObstacle.Size * newObstacle.Size;
196
197 // Go through all the cells that are next to search
198 for (int i = 0; i < addObstacleLists.currentSearchCells.Count; i++)
199 {
200 SearchCell cell = addObstacleLists.currentSearchCells[i];
201
202 // Temporary variables to set edges for testing
203 float testX = cell.Position.x + (circleCenter.x <= cell.Position.x ? 0.0f : CellSize);
204 float testY = cell.Position.y + (circleCenter.y <= cell.Position.y ? 0.0f : CellSize);
205
206 // Get distance from closest edges
207 float distX = circleCenter.x - testX;
208 float distY = circleCenter.y - testY;
209 float nextDistance = (distX * distX) + (distY * distY);
210
211 // If the distance is less than the radius, the obstacle would be present in the current search cell.
212 if (nextDistance <= radiusSqr)
213 cell.AddObstacle(newObstacle);
214
215 // Go through each searched cell's neighbors and see which ones haven't been searched yet or marked for search yet and add them.
216 foreach (SearchCell neighborCell in cell.Neighbors)
217 if (!addObstacleLists.nextSearchCells.Contains(neighborCell)
218 && addObstacleLists.alreadySearchedCells.Contains(neighborCell))
219 {
220 addObstacleLists.nextSearchCells.Add(neighborCell);
221 addObstacleLists.alreadySearchedCells.Add(neighborCell);
222 }
223 }
224
225 // No potential target found? Increase the search surface
226 coveredSurface += cellSize;
227
228 // As long as the covered search surface has not got beyond the allowed search radius
229 if (coveredSurface < newObstacle.Size)
230 // Every search round, we go one cell size (or search cell) further.
231 // The next cells to search are now the yet-unsearched neighbor cells
232 {
233 addObstacleLists.currentSearchCells.Clear();
234 addObstacleLists.currentSearchCells.AddRange(addObstacleLists.nextSearchCells);
235 }
236 else //we have already gone through the allowed search radius
237 break;
238 }
239
240 return ErrorMessage.none;
241 }

259 private void GenerateCells ()
260 {
261 if (!logger.RequireTrue(cellSize > 0,
262 $"[{GetType().Name}] The search grid cell size must be >= 0."))
263 return;

265 gridDict = new Dictionary<Int2D, SearchCell>();
266
267 // According to the start and end position coordinates, create the required search cells
268 for(int x = lowerLeftCorner.x; x < upperRightCorner.x; x += cellSize)
269 for (int y = lowerLeftCorner.y; y < upperRightCorner.y; y += cellSize)
270 {
271 // Each search cell instance is added to the dictionary after it is created for easier direct access using coordinates in the future.
272 Int2D nextPosition = new Int2D
273 {
274 x = x,
275 y = y
276 };
277
278 gridDict.Add(nextPosition, new SearchCell());
279 }
280
281 allSearchCells = gridDict.Values.ToList();
282
283 // Go through all generated cells, init them and assign their neighbors
284 foreach (Int2D position in gridDict.Keys)
285 gridDict[position].Init(gameMgr, position, FindNeighborCells(position).ToArray());

288 public IEnumerable<SearchCell> FindNeighborCells (Int2D sourcePosition)
289 {
290 // To store the found neighbor cells
291 List<SearchCell> neighbors = new List<SearchCell>();
292
293 // Maximum amount of potential neighboring cells
294 int maxNeighborAmount = 8;
295 Int2D nextPosition = new Int2D();
296
297 while(maxNeighborAmount > 0)
298 {
299 switch(maxNeighborAmount)
300 {
301 case 1: //right
302 nextPosition = new Int2D { x = sourcePosition.x + cellSize, y = sourcePosition.y };
303 break;
304 case 2: //left
305 nextPosition = new Int2D { x = sourcePosition.x - cellSize, y = sourcePosition.y };
306 break;
307 case 3: //up
308 nextPosition = new Int2D { x = sourcePosition.x, y = sourcePosition.y + cellSize };
309 break;
310 case 4: //down
311 nextPosition = new Int2D { x = sourcePosition.x, y = sourcePosition.y - cellSize };
312 break;
313
314 case 5: //upper-right
315 nextPosition = new Int2D { x = sourcePosition.x + cellSize, y = sourcePosition.y + cellSize };
316 break;
317 case 6: //upper-left
318 nextPosition = new Int2D { x = sourcePosition.x - cellSize, y = sourcePosition.y + cellSize };
319 break;
320 case 7: //lower-right
321 nextPosition = new Int2D { x = sourcePosition.x + cellSize, y = sourcePosition.y - cellSize };
322 break;
323 case 8: //lower-left
324 nextPosition = new Int2D { x = sourcePosition.x - cellSize, y = sourcePosition.y - cellSize };
325 break;
326 }
327
328 if (gridDict.TryGetValue(nextPosition, out SearchCell neighborCell))
329 neighbors.Add(neighborCell);
330
331 maxNeighborAmount--;
332 }
333
334 return neighbors;
335 }

337 public ErrorMessage TryGetSearchCell (Vector3 position, out SearchCell cell)
338 {
339 Vector3 clampedPosition = clampSearchPosition
340 ? new Vector3(Mathf.Clamp(position.x, lowerLeftCorner.x, upperRightCorner.x),
341 position.y,
342 Mathf.Clamp(position.z, lowerLeftCorner.y, upperRightCorner.y))
343 : position;
344
345 // Find the coordinates of the potential search cell where the input position is in
346 Int2D nextPosition = new Int2D
347 {
348 x = (int)((clampedPosition.x - lowerLeftCorner.x) / cellSize) * cellSize + lowerLeftCorner.x,
349 y = (int)((clampedPosition.z - lowerLeftCorner.y) / cellSize) * cellSize + lowerLeftCorner.y
350 };
351
352 if(gridDict.TryGetValue(nextPosition, out cell))
353 return ErrorMessage.none;
354
355 logger.Log(
356 $"[{GetType().Name}] No search cell has been defined to contain position: {position}!",
357 source: this,
358 type: LoggingType.warning);
359 return ErrorMessage.searchCellNotFound;
360 }

459 RTSHelper.IsTargetValidDelegate IsTargetValid, bool findClosest, bool playerCommand) where T : IEntity
460 {
461 //targets = new List<T>(originalAmount);
462
463 ErrorMessage errorMessage;
464 // Only continue if a valid source search cell is found in the input position.
465 if ((errorMessage = TryGetSearchCell(sourcePosition, out SearchCell sourceCell)) != ErrorMessage.none)
466 return errorMessage;
467
468 FloatRange radiusSqr = new FloatRange(radius.min * radius.min, radius.max * radius.max);
469 float nextDistance = 0.0f;
470 // The size of the covered surface in terms of cell size
471 int coveredSurface = 0;
472
473 // The amount that is confirmed to fit the search request
474 int confirmedAmount = 0;
475
476 // Using a sorted list allows to sort potential targets depending on their distance from the search source position
477 findLists.currTargetsList.Clear();
478
479 CachedSearchSource cachedSearchSource = new CachedSearchSource
480 {
481 entityType = typeof(T),
482 radiusSqr = radiusSqr,
483 sourceCell = sourceCell
484 };
485
486 if(cacheSearchResults
487 && findClosest
488 && cachedSearches.TryGetValue(cachedSearchSource, out CachedSearchResult cachedSearchResult))
489 {
490 for (int i = 0; i < cachedSearchResult.searchedCells.Count; i++)
491 {
492 SearchCell cell = cachedSearchResult.searchedCells[i];

494 {
495 IEntity entity = cell.Entities[j];
496 if (!entity.IsValid()
497 || !entity.IsSearchable
498 || !(entity is T))
499 continue;

501 nextDistance = (entity.transform.position - sourcePosition).sqrMagnitude;
502
503 if ((nextDistance >= radiusSqr.min && nextDistance <= radiusSqr.max)

506 findLists.currTargetsList.Add(new KeyValuePair<float, IEntity>(nextDistance, entity));
507 confirmedAmount++;
508 }
509 }
510 }
511
512 findLists.currTargetsList.Sort((elem1, elem2) => (elem1.Key.CompareTo(elem2.Key)));
513
514 // After going through all the current cells to search
515 // See if we have a potential target or if we have already found all of our required targets
516 if (confirmedAmount >= originalAmount)
517 return ErrorMessage.none;
518 else
519 return ErrorMessage.searchTargetNotFound;
520 }
521
522 // Cells to search initially are the source cell and its direct neighbors.
523 findLists.currentSearchCells.Clear();
524 findLists.currentSearchCells.Add(sourceCell);
525 findLists.currentSearchCells.AddRange(sourceCell.Neighbors);
526
527 // We include the source cell and its direct neighbors in the already searched cells so that...
528 //... we do not add them twice when looking at the current search cells neighbors that have not been considered for search in the loop below
529 findLists.alreadySearchedCells.Clear();
530 for(int i = 0; i < sourceCell.Neighbors.Count; i++)
531 findLists.alreadySearchedCells.Add(sourceCell.Neighbors[i]);
532 findLists.alreadySearchedCells.Add(sourceCell);
533
534 // As long as there cells to search
535 while(findLists.currentSearchCells.Count > 0)
536 {
537 // Holds te neighbor cells of the current cells to search so they would be searched in the next round.
538 findLists.nextSearchCells.Clear();
539
540 // Go through all the cells that are next to search
541 for (int i = 0; i < findLists.currentSearchCells.Count; i++)
542 {
543 SearchCell cell = findLists.currentSearchCells[i];
544
545 if (cacheSearchResults)
546 {
547 if (cachedSearchCells.ContainsKey(cell))
548 cachedSearchCells[cell].Add(cachedSearchSource);
549 else
550 {
551 cachedSearchCells.Add(cell, new List<CachedSearchSource>() { cachedSearchSource });
552 cell.SearchCellUpdated += HandleSearchCellUpdated;
553 }
554 }
555
556 for (int j = 0; j < cell.Entities.Count; j++)
557 {
558 IEntity entity = cell.Entities[j];
559 if (!entity.IsValid()
560 || !entity.IsSearchable
561 || !(entity is T))
562 continue;
563
564 nextDistance = (entity.transform.position - sourcePosition).sqrMagnitude;
565
566 if ((nextDistance >= radiusSqr.min && nextDistance <= radiusSqr.max)
567 && IsTargetValid(entity.ToSetTargetInputData(playerCommand)) == ErrorMessage.none)
568 {
569 findLists.currTargetsList.Add(new KeyValuePair<float, IEntity>(nextDistance, entity));
570 confirmedAmount++;
571
572 if (!findClosest && confirmedAmount >= originalAmount)
573 return ErrorMessage.none;
574 }
575 }
576
577 // Go through each searched cell's neighbors and see which ones haven't been searched yet or marked for search yet and add them.
578 for (int k = 0; k < cell.Neighbors.Count; k++)
579 {
580 SearchCell neighborCell = cell.Neighbors[k];
581
582 if (!findLists.nextSearchCells.Contains(neighborCell)
583 && !findLists.alreadySearchedCells.Contains(neighborCell))
584 {
585 findLists.nextSearchCells.Add(neighborCell);
586 findLists.alreadySearchedCells.Add(neighborCell);
587 }
588 }
589 }
590
591 findLists.currTargetsList.Sort((elem1, elem2) => (elem1.Key.CompareTo(elem2.Key)));
592
593 if(cacheSearchResults && findClosest && !cachedSearches.ContainsKey(cachedSearchSource))
594 cachedSearches.Add(
595 cachedSearchSource,
596 new CachedSearchResult
597 {
598 searchedCells = new List<SearchCell>(findLists.alreadySearchedCells)
599 });
600
601 // After going through all the current cells to search
602 // See if we have a potential target or if we have already found all of our required targets
603 if (confirmedAmount >= originalAmount)
604 return ErrorMessage.none;
605 else
606 {
607 // No potential target found? Increase the search surface
608 coveredSurface += cellSize;
609
610 // As long as the covered search surface has not got beyond the allowed search radius
611 if (coveredSurface < radius.max)
612 {
613 // Every search round, we go one cell size (or search cell) further.
614 // The next cells to search are now the yet-unsearched neighbor cells
615 findLists.currentSearchCells.Clear();
616 findLists.currentSearchCells.AddRange(findLists.nextSearchCells);
617 }
618 else //we have already gone through the allowed search radius
619 break;
620 }
621 }
622
623 return ErrorMessage.searchTargetNotFound;
624 }

640 public ErrorMessage IsPositionReserved (Vector3 testPosition, float radius, TerrainAreaMask areasMask, bool playerCommand, IMovementTargetPositionMarker ignoreMarker = null)
641 {
642 ErrorMessage errorMessage;
643 if ((errorMessage = TryGetSearchCell(testPosition, out SearchCell sourceCell)) != ErrorMessage.none)
644 return errorMessage;
645
646 // Cells to search initially are the source cell and its direct neighbors.
647 posResLists.currentSearchCells.Clear();
648 posResLists.currentSearchCells.Add(sourceCell);
649 posResLists.currentSearchCells.AddRange(sourceCell.Neighbors);
650
651 // We include the source cell and its direct neighbors in the already searched cells so that...
652 //... we do not add them twice when looking at the current search cells neighbors that have not been considered for search in the loop below
653 posResLists.alreadySearchedCells.Clear();
654 for(int i = 0; i < sourceCell.Neighbors.Count; i++)
655 posResLists.alreadySearchedCells.Add(sourceCell.Neighbors[i]);
656 posResLists.alreadySearchedCells.Add(sourceCell);
657
658 // The size of the covered surface in terms of cell size
659 int coveredSurface = 0;
660
661 // Since we're comparing squarred distances we need the squarred value of the radius
662 float sqrRadius = radius * radius;
663
664 // As long as there cells to search
665 while(posResLists.currentSearchCells.Count > 0)
666 {
667 // Holds te neighbor cells of the current cells to search so they would be searched in the next round.
668 posResLists.nextSearchCells.Clear();
669
670 for (int i = 0; i < posResLists.currentSearchCells.Count; i++)
671 {
672 SearchCell cell = posResLists.currentSearchCells[i];
673 for (int j = 0; j < cell.UnitTargetPositionMarkers.Count; j++)
674 {
675 IMovementTargetPositionMarker marker = cell.UnitTargetPositionMarkers[j];
676 if (marker.Enabled
677 && marker != ignoreMarker
678 && marker.AreasMask.Intersect(areasMask)
679 //&& marker.IsIn(testPosition))
680 && (marker.Position - testPosition).sqrMagnitude <= sqrRadius)
681 return ErrorMessage.mvtPositionMarkerReserved;
682 }
683
684 for (int j = 0; j < cell.Obstacles.Count; j++)
685 {
686 ISearchObstacle obstacle = cell.Obstacles[j];
687 if (obstacle.IsReserved(testPosition, areasMask, playerCommand))
688 return ErrorMessage.mvtPositionObstacleReserved;
689 }
690
691 // Go through each searched cell's neighbors and see which ones haven't been searched yet or marked for search yet and add them.
692 for (int j = 0; j < cell.Neighbors.Count; j++)
693 {
694 SearchCell neighborCell = cell.Neighbors[j];
695 if (!posResLists.nextSearchCells.Contains(neighborCell)
696 && !posResLists.alreadySearchedCells.Contains(neighborCell))
697 {
698 posResLists.nextSearchCells.Add(neighborCell);
699 posResLists.alreadySearchedCells.Add(neighborCell);
700 }
701 }
702 }
703
704 // After going through all the current cells to search
705 // Increase the search surface
706 coveredSurface += cellSize;
707
708 // As long as the covered search surface has not got beyond the allowed search radius
709 // Every search round, we go one cell size (or search cell) further.
710 if (coveredSurface < radius)
711 {
712 // The next cells to search are now the yet-unsearched neighbor cells
713 posResLists.currentSearchCells.Clear();
714 posResLists.currentSearchCells.AddRange(posResLists.nextSearchCells);
715 }
716 else
717 // We have already gone through the allowed search radius
718 break;
719 }
720
721 // No target position marker is present in the searched range then the position is not reserved.
722 return ErrorMessage.none;
723 }

727 private void HandleCameraPositionUpdated(IMainCameraController sender, EventArgs args)
728 {
729 UpdateCacheHandling();
730 }
731
732 private ErrorMessage UpdateCacheHandling()
733 {
734 var terrainPositions = terrainMgr.BaseTerrainCameraBounds.Get();
735
736 nextRendererdCells.Clear();
737
738 for (int i = 0; i < allSearchCells.Count; i++)
739 {
740 SearchCell cell = allSearchCells[i];
741
742 if (nextRendererdCells.Contains(cell))
743 continue;
744
745 if (terrainPositions.IsInsidePolygon(new Vector2(cell.Position.x, cell.Position.y)))
746 {
747 cell.OnUpdateRendering(isRenderering: true);
748 nextRendererdCells.Add(cell);
749 lastRendererdCells.Remove(cell);
750
751 foreach(SearchCell neighborCell in cell.Neighbors)
752 {
753 if (nextRendererdCells.Contains(neighborCell))
754 continue;
755
756 neighborCell.OnUpdateRendering(isRenderering: true);
757 nextRendererdCells.Add(neighborCell);
758 lastRendererdCells.Remove(neighborCell);
759 }
760 }
761 }
762
763 foreach(SearchCell cell in lastRendererdCells)
764 cell.OnUpdateRendering(isRenderering: false);
765
766 lastRendererdCells.Clear();
767 foreach (SearchCell cell in nextRendererdCells)
768 lastRendererdCells.Add(cell);
769
770 return ErrorMessage.none;
771 }
