94 private void HandleBuildingDeadGlobal(IBuilding building, DeadEventArgs args)
95 {
96 if (!factionMgr.IsSameFaction(building)
97 || !building.BorderComponent.IsValid())
98 return;
99
100 // Consider building centers that are destroyed and remove all pending buildings that are supposed to be placed around them
101 defaultPlacementHandler.StopOnCondition(placementData =>
102 placementData.options.initialCenter.IsValid() && placementData.options.initialCenter == building);
103 }

107 private void HandleBuildingUpgradedGlobal(IBuilding building, UpgradeEventArgs<IEntity> args)
108 {
109 if (!factionMgr.IsSameFaction(args.FactionID))
110 return;
111
112 // In case a building type that is scheduled to be placed has been upraded then destroy it and allow the NPC faction to place the upgraded instance
113 defaultPlacementHandler.StopOnCondition(placementData =>
114 placementData.options.initialCenter.IsValid() && placementData.instance.Code == building.Code);
115 }

118 #region Requesting Building Placement
119 public bool OnBuildingPlacementRequest(BuildingCreationTask task, IBuilding buildingCenter, IReadOnlyList<BuildingPlaceAroundData> placeAroundDataSet, bool canRotate)
120 {
121 ErrorMessage errorMsg = defaultPlacementHandler.Add(task,
122 new BuildingPlacementOptions
123 {
124 initialCenter = buildingCenter?.BorderComponent,
125 placeAroundData = placeAroundDataSet,
126 disableRotation = !canRotate
127 });
128
129 if (errorMsg != ErrorMessage.none)
130 {
131 switch (errorMsg)
132 {
133 case ErrorMessage.placementBuildingCenterMissing:
134 LogEvent($"{task.Prefab.Code}: Building Center for building prefab hasn't been specified in the Building Placement Request!");
135 break;
136
137 case ErrorMessage.taskMissingResourceRequirements:
138 LogEvent($"{task.Prefab.Code}: Building creation task resource requirements missing!");
139 npcResourceMgr.OnIncreaseMissingResourceRequest(task.RequiredResources);
140 break;
141 default:
142 LogEvent($"'{task.Prefab.Code}': Placement Request Failure - Creation Tasks Requirements Not Met - Erorr: {errorMsg}");
143 break;
144 }
145
146 return false;
147 }
148
149 LogEvent($"'{task.Prefab.Code}': Placement Request Success");
150
151 return true;
152 }
