123
124 intValues = inputMgr.ToIntValues(targetFactionID)

126 source: this,

130 public sealed override ErrorMessage SetFactionLocal (IEntity source, int targetFactionID)
131 {
132 var eventArgs = new FactionUpdateArgs(source, targetFactionID);
133
134 RaiseFactionUpdateStart(eventArgs);
135 globalEvent.RaiseEntityFactionUpdateStartGlobal(this, eventArgs);
136
137 selectionMgr.Remove(this);
138
139 //to disable the entity components for its ex-faction
140 Disable(isUpgrade: false, isFactionUpdate: true);
141
142 if(RTSHelper.IsValidFaction(targetFactionID))
143 {
144 FactionMgr = gameMgr.GetFactionSlot(targetFactionID).FactionMgr;
145 FactionID = targetFactionID;
146 IsFree = false;
147 }
148 else
149 {
150 FactionMgr = null;
151 FactionID = RTSHelper.FREE_FACTION_ID;
152 IsFree = true;
153 }
154
155 //update colors.
156 UpdateColors();
157
158 RaiseFactionUpdateComplete(eventArgs);
159 globalEvent.RaiseEntityFactionUpdateCompleteGlobal(this, eventArgs);
160
161 OnFactionUpdated();
162
163 return ErrorMessage.none;
164 }
