234 public override ErrorMessage IsTargetValid(SetTargetInputData data)
235 {
236 TargetData<IFactionEntity> potentialTarget = data.target;
237
238 if (!factionEntity.CanLaunchTask)
239 return ErrorMessage.taskSourceCanNotLaunch;
240 // See if the attack entity can attack without target
241 else if (!potentialTarget.instance.IsValid())
242 return !requireTarget && attackMgr.CanLaunchTerrainAttack(data.ToLaunchAttackData<IEntity>(Entity))
243 ? ErrorMessage.none
244 : ErrorMessage.attackTargetRequired;
245
246 if (potentialTarget.instance == factionEntity)
247 return ErrorMessage.invalid;
248 else if (!potentialTarget.instance.IsInteractable)
249 return ErrorMessage.uninteractable;
250 else if (potentialTarget.instance.Health.IsDead)
251 return ErrorMessage.healthDead;
252 else if (!potentialTarget.instance.Health.CanBeAttacked)
253 return ErrorMessage.attackTargetDisabled;
254 else if (factionEntity.IsFriendlyFaction(potentialTarget.instance) && !engageOptions.engageFriendly)
255 return ErrorMessage.factionIsFriendly;
256 else if (gameMgr.InPeaceTime && !factionEntity.IsFree && !potentialTarget.instance.IsFree)
257 return ErrorMessage.gamePeaceTimeActive;
258 else if (!targetPicker.IsValidTarget(potentialTarget.instance))
259 return ErrorMessage.targetPickerUndefined;
260 // We leave checking for LOS as the last condition
261 // Since 'ErrorMessage IsTargetValid(SetTargetInputData testInput)' which calls this method when SetTargetLocal is called..
262 // ... has the flag for ignoring LOS which can override the decision made in this condition
263 else if (!data.ignoreLOS && !data.playerCommand)
264 {
265 ErrorMessage errorMessage = LineOfSight.IsInSight(data.target, ignoreAngle: engageOptions.autoIgnoreAngleLOS, ignoreObstacle: engageOptions.autoIgnoreObstacleLOS);
266 if (errorMessage != ErrorMessage.none)
267 return errorMessage;
268 }
269
270 return ErrorMessage.none;

446 private void Complete()
447 {
448 // Disable last terrain attack if one was active
449 if(!allowMultipleTerrainAttacks)
450 terrainAttackActive = false;
451
452 completeEvent.Invoke();
453
454 audioMgr.PlaySFX(factionEntity.AudioSourceComponent, attackCompleteAudio.Fetch(), false);
455 if (factionEntity.AnimatorController.IsValid())
456 factionEntity.AnimatorController.SetState(AnimatorState.idle);
457
458 OnComplete();
459
460 // This is not the attack type to revert, check if there is one and get back to it
461 if (!Revert)
462 {
463 for (int i = 0; i < factionEntity.AttackComponents.Count; i++)
464 {
465 if (factionEntity.AttackComponents[i].Revert)
466 {
467 factionEntity.AttackComponents[i].SwitchAttackAction(false);
468 break;
469 }
470 }
471 }
472
473 ResetAttack();
474
475 // Attack once? cancel attack to prevent source from attacking again
476 if (engageOptions.engageOnce == true)
477 Stop();
478 }

582 private ErrorMessage SwitchAttackActionLocal(bool playerCommand)
583 {
584 // Assuming that the method is called from its non local counterpart first, it already satisfies the conditions for SetActive
585 // Therefore, we call SetActiveLocal directly here
586
587 globalEvent.RaiseAttackSwitchStartGlobal(this);
588
589 // Deactivate currently active attack component
590 IAttackComponent lastActiveAttackComp = factionEntity.FirstActiveAttackComponent;
591 IFactionEntity lastActiveAttackCompTarget = null;
592 if (lastActiveAttackComp.IsValid())
593 {
594 if (lastActiveAttackComp.HasTarget)
595 lastActiveAttackCompTarget = lastActiveAttackComp.Target.instance;
596 lastActiveAttackComp.SetActiveLocal(false, playerCommand);
597 }
598
599 SetActiveLocal(true, playerCommand);
600
601 globalEvent.RaiseAttackSwitchCompleteGlobal(this);
602 if(lastActiveAttackCompTarget.IsValid())
603 SetTarget(lastActiveAttackCompTarget.ToTargetData(), playerCommand: false);
604
605 return ErrorMessage.none;
