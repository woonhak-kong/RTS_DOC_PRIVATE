54 #region Update/Apply Input
55 public override void UpdateInput()
56 {
57 currPanDirection = Vector3.zero;
58
59 // If the pan on screen edge is enabled and we either are ignoring UI elements on the edge of the screen or the player's mouse is not over one
60 if (screenEdgePanning.enabled && (screenEdgePanning.ignoreUI || !EventSystem.current.IsPointerOverGameObject()))
61 {
62 // If the mouse is in either one of the 4 edges of the screen then move it accordinly
63 if (Input.mousePosition.x <= screenEdgePanning.size && Input.mousePosition.x >= 0.0f)
64 currPanDirection.x = -1.0f;
65 else if (Input.mousePosition.x >= Screen.width - screenEdgePanning.size && Input.mousePosition.x <= Screen.width)
66 currPanDirection.x = 1.0f;
67
68 if (Input.mousePosition.y <= screenEdgePanning.size && Input.mousePosition.y >= 0.0f)
69 currPanDirection.z = -1.0f;
70 else if (Input.mousePosition.y >= Screen.height - screenEdgePanning.size && Input.mousePosition.y <= Screen.height)
71 currPanDirection.z = 1.0f;
72 }
73
74 // Camera pan on key input (overwrites the screen edge pan if it has been enabled and had effect on this frame)
75 if (keyPanning.enabled)
76 {
77 if(controls.Get(keyPanning.up))
78 currPanDirection.z = 1.0f;
79 else if(controls.Get(keyPanning.down))
80 currPanDirection.z = -1.0f;
81
82 if(controls.Get(keyPanning.right))
83 currPanDirection.x = 1.0f;
84 else if(controls.Get(keyPanning.left))
85 currPanDirection.x = -1.0f;
86 }
87
88 // Camera pan on axis input (overwrites the screen edge pan/key input axis if it has been enabled and had effect on this frame)
89 if (inputAxisPanning.enabled)
90 {
91 if (Mathf.Abs(Input.GetAxis(inputAxisPanning.horizontal)) > 0.25f)
92 currPanDirection.x = Mathf.Sign(Input.GetAxis(inputAxisPanning.horizontal)) * 1.0f;
93 if (Mathf.Abs(Input.GetAxis(inputAxisPanning.vertical)) > 0.25f)
94 currPanDirection.z = Mathf.Sign(Input.GetAxis(inputAxisPanning.vertical)) * 1.0f;
95 }
96
97 if (mouseWheelPanning.enabled && Input.GetMouseButton(2))
98 {
99 currPanDirection.x = cameraController.MousePositionDelta.x * mouseWheelPanning.smoothFactor;
100 currPanDirection.z = cameraController.MousePositionDelta.y * mouseWheelPanning.smoothFactor;
101 }
102
103 currPanDirection.y = 0.0f;
104 }
