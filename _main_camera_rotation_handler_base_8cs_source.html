102 public void Apply()
103 {
104 if (Mathf.Abs(currRotationValue.x) < minRotationTriggerValues.x)
105 currRotationValue.x = 0.0f;
106 if (Mathf.Abs(currRotationValue.y) < minRotationTriggerValues.y)
107 currRotationValue.y = 0.0f;
108
109 if (resetRotationOnPlacement && placementMgr.IsLocalPlayerPlacingBuilding)
110 {
111 if (!HasInitialRotation && !isResettingRotation)
112 {
113 Vector3 lookAtCenter = cameraController.ScreenToWorldPoint(RTSHelper.MiddleScreenPoint, applyOffset: false);
114 ResetRotation(smooth: false);
115 cameraController.PanningHandler.LookAt(lookAtCenter, smooth: false);
116 }
117 currRotationValue = Vector2.zero;
118 }
119
120 isRotatingAround = rotateAroundAlways || controls.Get(rotateAroundKey);
121
122 if (currRotationValue != Vector2.zero && cameraController.PanningHandler.IsFollowingTarget)
123 cameraController.PanningHandler.SetFollowTarget(null);
124
125 // If the player is moving the camera and the camera's rotation must be fixed during movement...
126 //... or if the camera is following a target, lock camera rotation to default value
127 if (isResettingRotation
128 || ((fixPanRotation && cameraController.PanningHandler.LastPanDirection.magnitude > allowedRotationPanSize) || cameraController.PanningHandler.IsFollowingTarget))
129 {
130 cameraController.MainCamera.transform.rotation = Quaternion.Lerp(cameraController.MainCamera.transform.rotation, initialRotation, rotationSpeed.smoothFactor);
131
132 if (HasInitialRotation)
133 isResettingRotation = false;
134 return;
135 }
136
137 if (!IsActive)
138 return;
139
140 // Smoothly update the last rotation value towards the current one
141 lastRotationValue = Vector2.Lerp(lastRotationValue, currRotationValue, rotationSpeed.smoothFactor);
142
143 Vector3 nextEulerAngles = cameraController.MainCamera.transform.rotation.eulerAngles;
144
145 if (isRotatingAround)
146 {
147 // The position that the camera will be rotating around is the world position of the middle of the screen.
148 // Only update it when the player is not actively rotating the camera..
149 if (currRotationValue == Vector2.zero)
150 rotateAroundCenter = cameraController.ScreenToWorldPoint(RTSHelper.MiddleScreenPoint, applyOffset: false);
151 cameraController.MainCamera.transform.RotateAround(rotateAroundCenter,
152 Vector3.up,
153 lastRotationValue.x * rotationSpeed.value * Time.deltaTime);
154
155 // added by Hak for rotating Y-Axis.
156 Vector3 directionToCenter = (rotateAroundCenter - cameraController.MainCamera.transform.position).normalized;
157 Vector3 rightDirection = Vector3.Cross(cameraController.MainCamera.transform.up, directionToCenter);
158 cameraController.MainCamera.transform.RotateAround(rotateAroundCenter,
159 rightDirection,
160 lastRotationValue.y * rotationSpeed.value * Time.deltaTime);
161
162
163
164
165 nextEulerAngles = cameraController.MainCamera.transform.eulerAngles;
166 }
167 else
168 {
169 nextEulerAngles.y -= rotationSpeed.value * Time.deltaTime * lastRotationValue.x;
170 nextEulerAngles.x -= rotationSpeed.value * Time.deltaTime * lastRotationValue.y;
171 }
172
173 // Limit the y/x euler angless if that's enabled
174 if (rotationLimit.enableXLimit)
175 nextEulerAngles.x = Mathf.Clamp(nextEulerAngles.x, rotationLimit.xLimit.min, rotationLimit.xLimit.max);
176 if (rotationLimit.enableYLimit)
177 nextEulerAngles.y = Mathf.Clamp(nextEulerAngles.y, rotationLimit.yLimit.min, rotationLimit.yLimit.max);
178
179 cameraController.MainCamera.transform.rotation = Quaternion.Euler(nextEulerAngles);
180 if (lastRotationValue != Vector2.zero)
181 cameraController.RaiseCameraTransformUpdated();
182 }
