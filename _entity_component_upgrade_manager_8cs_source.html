90 public bool TryGet(IEntity entity, int factionID, out List<UpgradeElement<IEntityComponent>> componentUpgrades)
91 {
92 componentUpgrades = null;
93
94 if (!logger.RequireTrue(RTSHelper.IsValidFaction(factionID),
95 $"[{GetType().Name}] Attempting to get entity component upgrade elements for faction ID: {factionID} is not allowed!",
96 source: this))

102 public bool IsLaunched(EntityComponentUpgrade upgrade, EntityComponentUpgradeElementSource upgradeSource, int factionID)
103 {
104 if (!logger.RequireValid(RTSHelper.IsValidFaction(factionID),
105 $"[{GetType().Name}] Attempting to get entity component upgrade elements for faction ID: {factionID} is not allowed!"))
106 return false;
107
108 if (elements[factionID].TryGetValue(upgradeSource.GetSourceCode(upgrade.SourceEntity), out List<UpgradeElement<IEntityComponent>> componentUpgrades))
109 return componentUpgrades
110 .Where(element => element.sourceCode == upgradeSource.GetSourceCode(upgrade.SourceEntity)
111 && element.target == upgradeSource.UpgradeTarget)
112 .Any();
113
114 return false;
115 }

118 #region Launching Entity Component Upgrades
119 public ErrorMessage LaunchLocal(EntityComponentUpgrade upgrade, EntityComponentUpgradeElementSource upgradeSource, int factionID)
120 {
121 if (!logger.RequireTrue(RTSHelper.IsValidFaction(factionID),
122 $"[{GetType().Name}] Attempting to launch upgrade for invalid faction ID: {factionID} is not allowed!")
123 || !logger.RequireValid(upgrade.SourceEntity,
124 $"[{GetType().Name}] Attempting to launch upgrade for invalid source entity is not allowed!")
125 || !logger.RequireValid(upgradeSource.UpgradeTarget,
126 $"[{GetType().Name}] Attempting to launch upgrade for invalid target entity component is not allowed!"))
127 return ErrorMessage.invalid;
128
129 else if (!logger.RequireTrue(!IsLaunched(upgrade, upgradeSource, factionID),
130 $"[{GetType().Name}] Attempting to launch entity component upgrade for entity of code '{(upgrade.SourceEntity.IsValid() ? upgrade.SourceEntity.Code : "DESTROYED")}' that has been already launched!"))
131 return ErrorMessage.upgradeLaunched;
132
133 UpgradeElement<IEntityComponent> newElement = new UpgradeElement<IEntityComponent>
134 {
135 sourceCode = upgradeSource.GetSourceCode(upgrade.SourceEntity),
136 target = upgradeSource.UpgradeTarget
137 };
138
139 if (upgrade.SourceInstanceOnly)
140 {
141 effectObjPool.Spawn(upgrade.UpgradeEffect, upgrade.SourceEntity.transform);
142 upgrade.SourceEntity.UpgradeComponent(newElement);
143
144 if (!sourceOnlyElements.ContainsKey(upgrade.SourceEntity))
145 sourceOnlyElements.Add(upgrade.SourceEntity, new List<UpgradeElement<IEntityComponent>>());
146 sourceOnlyElements[upgrade.SourceEntity].Add(newElement);
147
148 // When the entity is destroyed, we make sure to remove it from the source only upgrade elments list
149 upgrade.SourceEntity.Health.EntityDead += HandleSourceOnlyUpgradeDead;
150 }
151 else
152 {
153 if (elements[factionID].TryGetValue(upgrade.SourceEntity.Code, out List<UpgradeElement<IEntityComponent>> componentUpgrades))
154 componentUpgrades.Add(newElement);
155 else
156 elements[factionID].Add(upgrade.SourceEntity.Code, new List<UpgradeElement<IEntityComponent>> { newElement });
157
158 // Upgrade all spawned elements of the same type of the upgrade source?
159 if (upgrade.UpdateSpawnedInstances)
160 foreach (IFactionEntity factionEntity in gameMgr.GetFactionSlot(factionID).FactionMgr.FactionEntities)
161 if (factionEntity.Code == upgrade.SourceEntity.Code)
162 {
163 factionEntity.UpgradeComponent(newElement);
164 effectObjPool.Spawn(upgrade.UpgradeEffect, factionEntity.transform);
165 }
166
167 globalEvent.RaiseEntityComponentUpgradedGlobal(
168 upgrade.SourceEntity,
169 new UpgradeEventArgs<IEntityComponent>(newElement, factionID, null));
170 }
171
172 // If there are upgrades that get triggerd from this one, launch them
173 foreach (TriggerUpgrade triggerUpgrade in upgradeSource.TriggerUpgrades)
174 triggerUpgrade.upgradeComp.LaunchLocal(gameMgr, triggerUpgrade.upgradeIndex, factionID);
175
176 return ErrorMessage.none;
177 }

186 #region Reseting Upgrades
187 public void ResetUpgrades(IEnumerable<IReadOnlyDictionary<string, IEnumerable<UpgradeElement<IEntityComponent>>>> newElements,
188 IReadOnlyDictionary<IEntity, IEnumerable<UpgradeElement<IEntityComponent>>> newSourceOnlyElements)
189 {
190 if (newElements.Count() != gameMgr.FactionCount)
191 {
192 logger.LogError("[EntityComponentUpgradeManager] The upgrades elements count must match the faction slots count!", source: this);
193 return;
194 }
195
196 elements = newElements
197 .Select(factionSlotUpgrades =>
198 {
199 var nextDic = new Dictionary<string, List<UpgradeElement<IEntityComponent>>>();
200 foreach (var upgradeElement in factionSlotUpgrades)
201 {
202 nextDic.Add(upgradeElement.Key, new List<UpgradeElement<IEntityComponent>>());
203 nextDic[upgradeElement.Key].AddRange(upgradeElement.Value);
204 }
205 return nextDic;
206 })
207 .ToArray();
208
209 sourceOnlyElements.Clear();
210 foreach (var kvp in sourceOnlyElements)
211 {
212 sourceOnlyElements.Add(kvp.Key, kvp.Value.ToList());
213 // When source only component upgrades are reset, we launch them again
214 foreach(var nextUpgradeElement in kvp.Value)
215 {
216 kvp.Key.UpgradeComponent(nextUpgradeElement);
217 }
218 }
219 }
220 #endregion
