261 public abstract bool IsLobbyGameDataMaster();

263 public void UpdateLobbyGameData (LobbyGameData newLobbyGameData)
264 {
265 LobbyGameData prevLobbyGameData = CurrentLobbyGameData;
266 CurrentLobbyGameData = newLobbyGameData;
267
268 // Only change the drop down menu values of the lobby game data in case this is not the lobby game data master
269 // Because the lobby game data master needs to change these values through the UI interface in order for this method to be called on their local end
270 // Resetting the drop down menu values for the lobby game data master means that the drop down value change event will be triggered
271 // When that event is triggered again, an update to the lobby game data will be called and an endless loop will occur
272 if (!IsLobbyGameDataMaster())
273 {
274 DefeatConditionSelector.SetOption(newLobbyGameData.defeatConditionID);
275 TimeModifierSelector.SetOption(newLobbyGameData.timeModifierID);
276 InitialResourcesSelector.SetOption(newLobbyGameData.initialResourcesID);
277 }
278
279 RaiseLobbyGameDataUpdated(prevLobbyGameData);
280 OnLobbyGameDataUpdated(prevLobbyGameData);
281 }
