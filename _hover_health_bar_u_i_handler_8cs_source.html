112 public bool CanEnable(IEntity source)
113 {
114 return isActive
115 && source.IsValid()
116 && !source.IsDummy
117 && source.IsEntityTypeMatch(allowedEntityTypes)
118 && (!playerFactionOnly || source.IsLocalPlayerFaction())
119 && source.Health.HoverHealthBarData.enabled
120 && !activeHealthBars.ContainsKey(source);
121 }
122
123 private void Enable(IEntity source)
124 {
125 if (!CanEnable(source))
126 return;
127
128 HoverHealthBarData nextData = source.Health.HoverHealthBarData;
129 if (nextData.offset.y == -1.0f)
130 nextData.offset = new Vector3(0.0f, source.Health.HoverHealthBarY, 0.0f);
131
132 HoverHealthBar nextHealthBar = Spawn(prefab, new HoverHealthBarSpawnInput(source, nextData));
133
134 activeHealthBars.Add(source, nextHealthBar);
135 }
136
137 private void Hide (IEntity source)
138 {
139 if (!activeHealthBars.TryGetValue(source, out HoverHealthBar nextHealthBar))
140 return;
141
142 Despawn(nextHealthBar);
143 activeHealthBars.Remove(source);
144 }
