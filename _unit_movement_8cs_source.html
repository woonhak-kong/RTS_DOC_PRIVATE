219 void FixedUpdate()
220 {
221 if (Unit.Health.IsDead) //if the unit is already dead
222 return; //do not update movement
223
224 if (isMoving == false)
225 {
226 UpdateIdleRotation();
227 return;
228 }
229
230 //to sync the unit's movement with its animation state, only handle movement if the unit is in its mvt animator state.
231 if (!Unit.AnimatorController.IsInMvtState == true)
232 return;
233
234 UpdateMovementRotation();
235
236 if (Controller.LastSource.targetAddableUnit.IsValid() //we have an addable target
237 //and it moved further away from the fetched addable position when the path was calculated and movement started.
238 && Vector3.Distance(Controller.LastSource.targetAddableUnitPosition, Controller.LastSource.targetAddableUnit.GetAddablePosition(Unit)) > mvtMgr.StoppingDistance)
239 {
240 OnHandleAddableUnitStop();
241 return;
242 }
243
244 if (DestinationReached == false) //check if the unit has reached its target position or not
245 if ((DestinationReached = IsPositionReached(Destination)))
246 OnHandleAddableUnitStop();
247 }
248
249 public bool IsPositionReached(Vector3 position)
250 => Vector3.Distance(Unit.transform.position, position) <= mvtMgr.StoppingDistance;

252 public void OnHandleAddableUnitStop()
253 {
254 MovementSource lastSource = Controller.LastSource;
255 Stop(); //stop the unit mvt
256
257 if (lastSource.targetAddableUnit.IsValid()) //unit is supposed to be added to this instance.
258 {
259 //so that the unit does not look at the IAddableUnit entity after it is added.
260 Target = new TargetData<IEntity> { opPosition = Target.opPosition };
261 lastSource.targetAddableUnit.Add(
262 Unit,
263 lastSource.sourceTargetComponent.IsValid() && lastSource.sourceTargetComponent != Unit.CarriableUnit
264 ? new AddableUnitData(lastSource.sourceTargetComponent, playerCommand: false)
265 : Unit.CarriableUnit.GetAddableData(playerCommand: false));
266 }
267 }

272 private void UpdateIdleRotation ()
273 {
274 if (!canIdleRotate || NextRotationTarget == Quaternion.identity) //can the unit rotate when idle + there's a valid rotation target
275 return;
276
277 if (Target.instance.IsValid()) //if there's a target object to look at.
278 NextRotationTarget = RTSHelper.GetLookRotation(Unit.transform, Target.instance.transform.position); //keep updating the rotation target as the target object might keep changing position
279
280 if (smoothIdleRotation)
281 Unit.transform.rotation = Quaternion.Slerp(Unit.transform.rotation, NextRotationTarget, Time.deltaTime * idleAngularSpeed.Value);
282 else
283 Unit.transform.rotation = NextRotationTarget;
284 }
285
289 private void EnableMovementRotation ()
290 {
291 facingNextCorner = false; //to trigger checking for correct rotation properties
292 Controller.IsActive = false; //stop handling rotation using the movement controller
293
294 NextCorner = Controller.NextPathTarget; //assign new corner in path
295 //set the rotation target to the next corner.
296 NextRotationTarget = RTSHelper.GetLookRotation(Unit.transform, NextCorner);
297 }
298
303 private void UpdateMovementRotation()
304 {
305 if (canMoveRotate) //can move and rotate? do not proceed.
306 return;
307
308 if (NextCorner != Controller.NextPathTarget) //if the next corner/destination on path has been updated
309 EnableMovementRotation();
310
311 if (facingNextCorner) //facing next corner? we good
312 return;
313
314 if (Controller.IsActive) //stop movement it if it's not already stopped
315 Controller.IsActive = false;
316
317 //keep checking if the angle between the unit and its next destination
318 Vector3 IdleLookAt = NextCorner - Unit.transform.position;
319 IdleLookAt.y = 0.0f;
320
321 //as long as the angle is still over the min allowed movement angle, then do not proceed to keep moving
322 //allow the controller to retake control of the movement if we're correctly facing the next path corner.
323 if(facingNextCorner = Vector3.Angle(Unit.transform.forward, IdleLookAt) <= minMoveAngle)
324 {
325 Controller.IsActive = true;
326 return;
327 }
328
329 //update the rotation as long as the unit is attempting to look at the next target in the path before it the Controller takes over movement (and rotation)
330 Unit.transform.rotation = Quaternion.Slerp(
331 Unit.transform.rotation,
332 NextRotationTarget,
333 Time.deltaTime * idleAngularSpeed.Value);
334 }
