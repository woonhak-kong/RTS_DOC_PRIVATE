298 private bool TryFindValidInitialGotoPosition()
299 {
300 Vector3 originalPosition = factionEntity.transform.position + initialDirection;
301 if (IsTargetValid(originalPosition.ToSetTargetInputData(playerCommand: false)) == ErrorMessage.none)
302 {
303 if (factionEntity.IsDummy)
304 gotoTransform.position = originalPosition;
305 else
306 SetTarget(originalPosition, false);
307
308 return true;
309 }
310
311 if (repositionToValidTerrainArea)
312 {
313 int expectedPositionCount = Mathf.FloorToInt(2.0f * Mathf.PI * factionEntity.Radius / (repositionSize * 2.0f));
314
315 // If no expected positions are to be found and the radius offset is zero then set the expected position count to 1 to test the actual target position if it is valid
316 if (expectedPositionCount == 0 && factionEntity.Radius == 0.0f)
317 expectedPositionCount = 1;
318
319 // Represents increment value of the angle inside the current circle with the above perimeter
320 float angleIncValue = 360f / expectedPositionCount;
321 float currentAngle = initialAngle;
322
323 Vector3 nextDestination = factionEntity.transform.position + factionEntity.Radius * new Vector3(Mathf.Cos(Mathf.Deg2Rad * currentAngle), 0.0f, Mathf.Sin(Mathf.Deg2Rad * currentAngle));
324
325 int counter = 0;
326
327 Vector3 closestPosition = Vector3.zero;
328 float closestDistance = Mathf.Infinity;
329 bool foundPosition = false;
330
331 // As long as we haven't inspected all the expected free positions inside this cirlce
332 while (counter < expectedPositionCount)
333 {
334 if (IsTargetValid(nextDestination.ToSetTargetInputData(playerCommand: false)) == ErrorMessage.none
335 && Vector3.Distance(nextDestination, originalPosition) < closestDistance)
336 {
337 foundPosition = true;
338 closestDistance = Vector3.Distance(nextDestination, originalPosition);
339 closestPosition = nextDestination;
340 }
341
342 // Increment the angle value to find the next position on the circle
343 currentAngle += angleIncValue;
344
345 // Rotate the nextDestination vector around the y axis by the current angle value
346 nextDestination = factionEntity.transform.position + factionEntity.Radius * new Vector3(Mathf.Cos(Mathf.Deg2Rad * currentAngle), 0.0f, Mathf.Sin(Mathf.Deg2Rad * currentAngle));
347
348 counter++;
349 }
350
351 if (foundPosition)
352 {
353 if (factionEntity.IsDummy)
354 gotoTransform.position = closestPosition;
355 else
356 SetTarget(closestPosition, false);
357
358 return true;
359 }
360 }
361
362 return false;
363 }
