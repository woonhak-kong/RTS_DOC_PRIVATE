81 #region Updating Amount
82 public void UpdateAmount(ResourceTypeValue updateValue, out int restAmount)
83 {
84 restAmount = 0;
85
86 if (Type.HasCapacity && Type.CapacityType == ResourceCapacityType.simpleLimit
87 && Amount + updateValue.amount > Capacity + updateValue.capacity)
88 {
89 int allowedAmount = Capacity + updateValue.capacity - Amount;
90 restAmount = updateValue.amount - allowedAmount;
91 updateValue.amount = allowedAmount;
92 }
93
94 Capacity += updateValue.capacity;
95 Amount += updateValue.amount;
96
97 ResourceUpdateEventArgs eventArgs = new ResourceUpdateEventArgs(
98 Type,
99 updateValue);
100
101 globalEvent.RaiseFactionSlotResourceAmountUpdatedGlobal(FactionID.ToFactionSlot(), eventArgs);
102 RaiseFactionResourceAmountUpdated(eventArgs);
103
104 OnAmountUpdated();
105 }

217 {
218 if (!args.Entity.IsUnit())
