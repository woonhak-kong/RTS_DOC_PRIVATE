162 private void PrepareMovement()
163 {
164 // Movement inputs:
165 mvtDirection = (TargetPosition - transform.position).normalized;
166
167 // Only consider the X and Z axis to calculate the distance because the heightMultiplier will affect the position on the Y axis.
168 initialDistance = Vector3.Distance(
169 new Vector3(transform.position.x, 0.0f, transform.position.z),
170 new Vector3(TargetPosition.x, 0.0f, TargetPosition.z));
171 incrementalDistance = 0.0f;
172
173 nextMvtCurve = initialDistance >= mainMvtCurve.minDistance && Mathf.Abs(transform.position.y - TargetPosition.y) <= mainMvtCurve.maxHeightDifference
174 ? mainMvtCurve
175 : altMvtCurve;
176
177 nextPosition = transform.position;
178 nextHeight = transform.position.y;
179 }

199 #region Handling State/Movement
200 protected override void OnActiveUpdate()
201 {
202 // Update the follow transform in case the attack object is "parented" into another game object that it needs to follow.
203 followTransform.Update();
204
205 // Allow the attack object to keep moving if it is being disabled but it does not have a parent object.
206 if (State == EffectObjectState.inactive
207 || (State == EffectObjectState.disabling && followTransform.HasTarget))
208 return;
209
210 if (lookAtPosition != Vector3.zero)
211 transform.rotation = Quaternion.LookRotation(lookAtPosition);
212
213 // Handle delay mode:
214 if (InDelay)
215 {
216 // Delay mode is done.
217 if (delayTimer.ModifiedDecrease())
218 {
219 followTransform.ResetTarget();
220
221 PrepareMovement();
222
223 if (triggerEffectsPostDelay)
224 TriggerEffectAudio();
225 }
226
227 // The attack object doesn't move while in delay
228 return;
229 }
230
231 // Following the target
232 if (followTarget
233 && Data.target.IsValid()
234 && Vector3.Distance(transform.position, Data.target.Selection.transform.position) <= followTargetMaxDistance)
235 {
236 mvtDirection = (Data.target.Selection.transform.position - transform.position).normalized;
237 }
238
239 // Handle movement
240 transform.position = new Vector3(nextPosition.x, nextHeight, nextPosition.z);
241
242 nextPosition += mvtDirection * speed.Value * Time.deltaTime;
243 incrementalDistance += speed.Value * Time.deltaTime;
244
245 float evalTime = Mathf.Clamp(incrementalDistance / initialDistance, 0.0f, Mathf.Infinity);
246 nextHeight = nextPosition.y + nextMvtCurve.curve.Evaluate(evalTime) * nextMvtCurve.heightMultiplier;
247
248 // Update rotation target
249 lookAtPosition = new Vector3(nextPosition.x, nextHeight, nextPosition.z) - transform.position;
250 }
251 #endregion

264 #region Dealing Damage
265 private void OnTriggerEnter(Collider other)
266 {
267 var entitySelection = other.gameObject.GetComponent<EntitySelectionCollider>();
268
269 if (State != EffectObjectState.running
270 || (InDelay && !Data.damageInDelay)
271 || (didDamage && damageOnce)
272 || RTSHelper.IsInLayerMask(terrainMgr.BaseTerrainLayerMask, other.gameObject.layer)
273 || (Data.source.Entity.IsValid() && entitySelection?.Entity == Data.source.Entity))
274 return;
275
276 // Does the collider belong to an obstacle?
277 if (RTSHelper.IsInLayerMask(obstacleLayerMask, other.gameObject.layer))
278 {
279 ApplyDamage(other.gameObject, null, transform.position);
280 return;
281 }
282
283 // Does the collider belong to a IFactionEntity?
284 if (!(entitySelection?.Entity is IFactionEntity hitFactionEntity)
285 || !hitFactionEntity.IsInitialized)
286 return;
287
288 // If this is a valid target to deal damage to, apply the damage.
289 if ((hitFactionEntity.FactionID != Data.sourceFactionID || Data.damageFriendly) && !hitFactionEntity.Health.IsDead)
290 ApplyDamage(other.gameObject, hitFactionEntity, hitFactionEntity.transform.position);
291 }

294 private void ApplyDamage(GameObject targetObject, IFactionEntity target, Vector3 targetPosition)
295 {
296 // Deal damage
297 damage.Trigger(target, targetPosition);
298 didDamage = true;
299
300 // Hit effect and audio
301
302 if (target.IsValid())
303 {
304 foreach (FactionEntityDependantHitEffectData hitEffectData in hitEffects)
305 if (hitEffectData.picker.IsValidTarget(target))
306 {
307 effectObjPool.Spawn(
308 hitEffectData.effect.Output,
309 new EffectObjectSpawnInput(
310 parent: null,
311
312 useLocalTransform: false,
313 spawnPosition: transform.position,
314 spawnRotation: new RotationData(
315 hitEffectData.faceSource ? RotationType.lookAtPosition : RotationType.free,
316 Data.spawnPosition,
317 fixYRotation: true)));
318
319 audioMgr.PlaySFX(target?.AudioSourceComponent,
320 hitEffectData.audio.Fetch(),
321 false);
322 break;
323 }
324 }
325 else
326 {
327 effectObjPool.Spawn(
328 obstacleHitEffect.effect.Output,
329 new EffectObjectSpawnInput(
330 parent: null,
331
332 useLocalTransform: false,
333 spawnPosition: transform.position,
334 spawnRotation: new RotationData(
335 obstacleHitEffect.faceSource ? RotationType.lookAtPosition : RotationType.free,
336 Data.spawnPosition,
337 fixYRotation: true)));
338
339 audioMgr.PlaySFX(target?.AudioSourceComponent,
340 obstacleHitEffect.audio.Fetch(),
341 false);
342 }
343
344 // Child object of the target?
345 if (targetObject.IsValid() && childOnDamage == true)
346 followTransform.SetTarget(
347 target.IsValid() ? target.transform : targetObject.transform,
348 offset: (transform.position - targetObject.transform.position),
349 enableCallback: true
350 );
351
352 // Disable on damage? Handle it through the effect object.
353 if (disableOnDamage == true)
354 Deactivate();
355 }
