157 private ErrorMessage AddLocal(SetTargetInputData input, bool launchOnEmpty = true)
158 {
159 input.fromTasksQueue = true;
160 queue.Add(input);
161
162 bool canStopCollectingResource = resourceCollector.IsValid() && resourceCollector.HasTarget && excludeResourceCollectionInKeepActiveTask;
163
164 if (launchOnEmpty && (!IsRunningQueueTask || canStopCollectingResource) && queue.Count == 1)
165 {
166 IEntityTargetComponent activeTargetComponent = Entity
167 .EntityTargetComponents
168 .Values
169 .FirstOrDefault(comp => comp.HasTarget && !comp.IsIdle);
170
171 // If there is no active target component or it is the resource collector one and we are allowed to bypass that
172 // Then launch the newly added task to the empty queue directly
173 if (!activeTargetComponent.IsValid() || !keepActiveTask || canStopCollectingResource || Entity.IsIdle)
174 TryLaunchNext(directLaunch: true);
175 else
176 {
177 SetRunningComponent(activeTargetComponent.Code, force: true);
178 }
179 }
180
181 return ErrorMessage.none;
182 }

184 private bool TryLaunchNext(bool directLaunch = false)
185 {
186 if (queue.Count == 0)
187 return false;
188
189 SetTargetInputData nextInput = queue[0];
190 // Mark the task as a player command only if it is marked as the first one to launch in the queue.
191 nextInput.playerCommand = directLaunch;
192
193 queue.RemoveAt(0);
194
195 if (!Entity.EntityTargetComponents[nextInput.componentCode].IsTargetValid(nextInput, out ErrorMessage errorMessage))
196 return false;
197
198 if((nextInput.playerCommand && Entity.HasAuthority())
199 || (!nextInput.playerCommand && RTSHelper.IsMasterInstance()))
200 Entity.EntityTargetComponents[nextInput.componentCode].SetTarget(nextInput);
201
202 SetRunningComponent(nextInput.componentCode, force: true);
203
204 return true;
205 }
