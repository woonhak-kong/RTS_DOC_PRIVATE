142 public bool TryGetCellPosition (Vector3 position, out Int2D cellPosition)
143 {
144 Vector3 clampedPosition = new Vector3(
145 Mathf.Clamp(position.x, terrainMgr.HeightCacheLowerLeftCorner.x, terrainMgr.HeightCacheUpperRightCorner.x),
146 position.y,
147 Mathf.Clamp(position.z, terrainMgr.HeightCacheLowerLeftCorner.y, terrainMgr.HeightCacheUpperRightCorner.y));
148
149 // Find the coordinates of the potential search cell where the input position is in
150 cellPosition = new Int2D
151 {
152 x = ( ((int)clampedPosition.x - terrainMgr.HeightCacheLowerLeftCorner.x) / CellSize) * CellSize + terrainMgr.HeightCacheLowerLeftCorner.x,
153 y = ( ((int)clampedPosition.z - terrainMgr.HeightCacheLowerLeftCorner.y) / CellSize) * CellSize + terrainMgr.HeightCacheLowerLeftCorner.y
154 };
155
156 return gridDict.ContainsKey(cellPosition);
