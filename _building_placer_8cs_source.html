295 private void OnTriggerEnter(Collider other)
296 {
297 // Ignore colliders that belong to this building (its selection colliders namely) and ones attached to effect objects
298 if (!IsPlacementStarted
299 || Placed
300 || ignoreCollisionLayerMask == (ignoreCollisionLayerMask | (1 << other.gameObject.layer))
301 || Building.Selection.IsSelectionCollider(other)
302 || other.gameObject.GetComponent<IEffectObject>().IsValid())
303 return;
304
305 IUnit collidedUnit = other.gameObject.GetComponent<EntitySelectionCollider>()?.Entity as IUnit;
306 if (collidedUnit.IsValid() && collidedUnit.IsSameFaction(Building))
307 overlappedLocalUnits.Add(collidedUnit);
308 else
309 overlappedColliders.Add(other);
310
311 OnPlacementUpdate();
312 }
313
314 private void OnTriggerExit(Collider other)
315 {
316 // Ignore colliders that belong to this building (its selection colliders namely) and ones attached to effect objects
317 if (!IsPlacementStarted
318 || Placed
319 || Building.Selection.IsSelectionCollider(other)
320 || other.gameObject.GetComponent<IEffectObject>().IsValid())
321 return;
322
323 IUnit collidedUnit = other.gameObject.GetComponent<EntitySelectionCollider>()?.Entity as IUnit;
324 if (collidedUnit.IsValid() && collidedUnit.IsSameFaction(Building))
325 overlappedLocalUnits.Remove(collidedUnit);
326 else
327 overlappedColliders.Remove(other);
328
329 OnPlacementUpdate();

332 public bool IsBuildingInBorder()
333 {
334 bool inRange = false; //true if the building is inside its faction's territory
335
336 if (PlacementCenter.IsValid()) //if the building is already linked to a building center
337 {
338 //check if the building is still inside this building center's territory
339 if (PlacementCenter.IsInBorder(transform.position)) //still inside the center's territory
340 inRange = true; //building is in range
341 else
342 {
343 inRange = false; //building is not in range
344 PlacementCenter = null; //set the current center to null, so we can find another one
345 }
346 }
347
348 if (!PlacementCenter.IsValid()) //if at this point, the building doesn't have a building center.
349 {
350 foreach (IBuilding center in Building.FactionMgr.BuildingCenters)
351 {
352 if (!center.BorderComponent.IsActive) //if the border of this center is not active yet
353 continue;
354
355 if (center.BorderComponent.IsInBorder(Building.transform.position) && center.BorderComponent.IsBuildingAllowedInBorder(Building)) //if the building is inside this center's territory and it's allowed to have this building around this center
356 {
357 inRange = true; //building center found
358 PlacementCenter = center.BorderComponent;
359 break; //leave the loop
360 }
361 }
362 }
363
364 if (canPlaceOutsideBorder)
365 inRange = true;
366
367 if ((PlacementCenter.IsValid() || canPlaceOutsideBorder) && inRange) //if, at this point, the building has a center assigned
368 {
369 //Sometimes borders collide with each other but the priority of the borders is determined by the order of the creation of the borders
370 //That's why we need to check for other factions' borders and make sure the building isn't inside one of them:
371
372 foreach(IBorder border in buildingMgr.AllBorders)
373 {
374 //if the border is not active or it belongs to the building's faction
375 if (!border.IsActive || border.Building.IsFriendlyFaction(Building) || canPlaceInEnemyBorder)
376 continue; //off to the next one
377
378 if (border.IsInBorder(Building.transform.position)
379 && (!PlacementCenter.IsValid() || border.SortingOrder > PlacementCenter.SortingOrder)) //if the building is inside this center's territory
380 {
381 //and if the border has a priority over the one that the building belongs to:
382 return false;
383 }
384
385 }
386 }
387
388 return inRange; //return whether the building is in range a building center or not
389 }

391 public bool IsBuildingOnMap()
392 {
393 Ray ray = new Ray(); //create a new ray
394 RaycastHit[] hits; //this will hold the registerd hits by the above ray
395
396 BoxCollider boxCollider = boundaryCollider.GetComponent<BoxCollider>();
397
398 //Start by checking if the middle point of the building's collider is over the map.
399 //Set the ray check source point which is the center of the collider in the game world:
400 ray.origin = new Vector3(transform.position.x + boxCollider.center.x, transform.position.y + 0.5f, transform.position.z + boxCollider.center.z);
401
402 ray.direction = Vector3.down; //The direction of the ray is always down because we want check if there's terrain right under the building's object:
403
404 int i = 4; //we will check the four corners and the center
405 while (i > 0) //as long as the building is still on the map/terrain
406 {
407 hits = Physics.RaycastAll(ray, placerMgr.TerrainMaxDistance); //apply the raycast and store the hits
408
409 bool hitTerrain = false; //did one the hits hit the terrain?
410 foreach(RaycastHit rh in hits) //go through all hits
411 if (terrainMgr.IsTerrainArea(rh.transform.gameObject, placableTerrainAreas))
412 hitTerrain = true;
413
414 if (hitTerrain == false) //if there was no registerd terrain hit
415 return false; //stop and return false
416
417 i--;
418
419 //If we reached this stage, then applying the last raycast, we successfully detected that there was a terrain under it, so we'll move to the next corner:
420 switch (i)
421 {
422 case 0:
423 ray.origin = new Vector3(transform.position.x + boxCollider.center.x + boxCollider.size.x / 2, transform.position.y + 0.5f, transform.position.z + boxCollider.center.z + boxCollider.size.z / 2);
424 break;
425 case 1:
426 ray.origin = new Vector3(transform.position.x + boxCollider.center.x + boxCollider.size.x / 2, transform.position.y + 0.5f, transform.position.z + boxCollider.center.z - boxCollider.size.z / 2);
427 break;
428 case 2:
429 ray.origin = new Vector3(transform.position.x + boxCollider.center.x - boxCollider.size.x / 2, transform.position.y + 0.5f, transform.position.z + boxCollider.center.z - boxCollider.size.z / 2);
430 break;
431 case 3:
432 ray.origin = new Vector3(transform.position.x + boxCollider.center.x - boxCollider.size.x / 2, transform.position.y + 0.5f, transform.position.z + boxCollider.center.z + boxCollider.size.z / 2);
433 break;
434 }
435 }
436
437 return true; //at this stage, we're sure that the center and all corners of the building are on the map, so return true
438 }
