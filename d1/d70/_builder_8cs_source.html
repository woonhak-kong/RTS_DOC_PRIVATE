250 public override ErrorMessage IsTargetValid (SetTargetInputData data)
251 {
252 TargetData<IBuilding> potentialTarget = data.target;
253
254 if (!IsActive || !potentialTarget.instance.IsValid())
255 return ErrorMessage.invalid;
256 else if (!potentialTarget.instance.IsInteractable)
257 return ErrorMessage.uninteractable;
258 else if (!RTSHelper.IsSameFaction(potentialTarget.instance, factionEntity))
259 return ErrorMessage.factionMismatch;
260 else if (!canConstruct || !constructionTargetPicker.IsValidTarget(potentialTarget.instance))
261 return ErrorMessage.targetPickerUndefined;
262 else if (potentialTarget.instance.Health.IsDead)
263 return ErrorMessage.targetDead;
264 else if (potentialTarget.instance.Health.HasMaxHealth)
265 return ErrorMessage.targetHealthtMaxReached;
266 else if (!factionEntity.CanMove() && !IsTargetInRange(transform.position, potentialTarget))
267 return ErrorMessage.targetOutOfRange;
268 else if (Target.instance != potentialTarget.instance && potentialTarget.instance.WorkerMgr.HasMaxAmount)
269 return ErrorMessage.workersMaxAmountReached;
270
271 // Repair Costs check
272 if (potentialTarget.instance.IsBuilt)
273 {
274 switch (buildingMgr.ConstructionType)
275 {
276 case ConstructionType.health:
277
278 // Unable to afford repair costs
279 if (!potentialTarget.instance.Health.CanRepair(healthPerProgress, takeResources: false))
280 return ErrorMessage.constructionResourcesMissing;
281 break;
282
283 case ConstructionType.time:
284 int maxHealth = potentialTarget.instance.Health.MaxHealth;
285 int currHealth = potentialTarget.instance.Health.CurrHealth;
286 float expectedBuildTime = ((maxHealth - currHealth) * potentialTarget.instance.Health.BuildTime) / (maxHealth - 1);
287 int expectedNextHealthToAdd = Mathf.CeilToInt(maxHealth - (1 + ((maxHealth - 1) / expectedBuildTime) * (expectedBuildTime - Time.deltaTime * TimeModifier.CurrentModifier)));
288 if (!potentialTarget.instance.Health.CanRepair(expectedNextHealthToAdd, takeResources: false))
289 return ErrorMessage.constructionResourcesMissing;
290 break;
291 }
292 }
293
294 return potentialTarget.instance.WorkerMgr.CanMove(unit);
295 }
