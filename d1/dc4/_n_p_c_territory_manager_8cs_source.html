44 protected override void OnPreInit()

58 protected override void OnPostInit()
59 {
60 IsActive = true;
61
62 // Go through the spawned building centers and add their territory size to the faction's territory count
63 foreach (IBuilding buildingCenter in factionMgr.BuildingCenters)
64 UpdateCurrentTerritory(buildingCenter.BorderComponent.Surface);

66 globalEvent.BuildingPlacementStartGlobal += HandleBuildingPlacementStartGlobal;

82 private void HandleFirstBuildingCenterInitialized(INPCBuildingCreator npcBuildingCreator, EventArgs args)
83 {
84 foreach (GameObject building in buildingCenters)
85 {
86 IBorder border = building?.GetComponentInChildren<IBorder>();
87
88 if (!logger.RequireValid(border,
89 $"[{GetType().Name} - {factionMgr.FactionID}] 'Building Centers' field has some unassigned/invalid elements that do not include the '{typeof(IBorder).Name}' component."))
90 return;
91
92 NPCBuildingRegulator nextRegulator;
93 if ((nextRegulator = npcBuildingCreator.ActivateBuildingRegulator(
94 building.GetComponent<IBuilding>(),
95 npcBuildingCreator.FirstBuildingCenter)).IsValid())
96 centerMonitor.AddCode(nextRegulator.Prefab.Code);
97 }
98
99 if (centerMonitor.Count <= 0)
100 {
101 logger.LogWarning($"[{GetType().Name} - Faction ID: {factionMgr.FactionID}] No building center regulators have been assigned!");
102 IsActive = false;
103 }
104 }

127 #region Handling Territory Ratio
128 private void UpdateCurrentTerritory(float value)
129 {
130 currentTerritoryRatio += (value / terrainMgr.MapSize);

132 IsActive = !HasReachedMaxTerritory && !HasReachedMinTerritory;
