104 private void UpdateBuildTime()
105 {
106 hasBuildersInProgress = false;
107 float multiplierTotal = 0;
108 for (int i = 0; i < Building.WorkerMgr.Workers.Count; i++)
109 {
110 IUnit worker = Building.WorkerMgr.Workers[i];
111 if (worker.BuilderComponent.InProgress)
112 {
113 hasBuildersInProgress = true;
114 multiplierTotal += worker.BuilderComponent.TimeMultiplier;
115 }
116 }
117
118 currBuildTimeStartingHealth = CurrHealth;
119 currBuildTime = ((MaxHealth - CurrHealth) * buildTime) / (MaxHealth - 1);
120 currBuildTime /= (1+multiplierTotal);
121
122 buildTimer = new TimeModifiedTimer(currBuildTime);
123 }
124
125 private void Update()
126 {
127 if (!RTSHelper.IsMasterInstance()
128 || Building.WorkerMgr.Amount == 0
129 || !hasBuildersInProgress
130 || buildingMgr.ConstructionType != ConstructionType.time)
131 return;
132
133 if (buildTimer.ModifiedDecrease())
134 Add(new HealthUpdateArgs(value: MaxHealth, source: Building.WorkerMgr.Workers[0]));
135 else
136 {
137 int nextCurrHealth = currBuildTimeStartingHealth + Mathf.CeilToInt(MaxHealth - (1 + ((MaxHealth - 1) / currBuildTime) * buildTimer.CurrValue));
138 if (nextCurrHealth < CurrHealth)
139 return;
140
141 // Unable to afford repair costs
142 if (Building.IsBuilt && !CanRepair(nextCurrHealth - CurrHealth))
143 {
144 if (Building.IsLocalPlayerFaction())
145 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
146 {
147 message = ErrorMessage.constructionResourcesMissing,
148
149 source = Building.WorkerMgr.Workers[0],
150 target = Building
151 });
152
153 // stop all builders
154 while(Building.WorkerMgr.Amount > 0)
155 Building.WorkerMgr.Workers[0].BuilderComponent.Stop();
156 return;
157 }
158
159
160 Add(new HealthUpdateArgs(value: nextCurrHealth - CurrHealth, source: Building.WorkerMgr.Workers[0]));
161 }
162 }

165 #region Handling Repair Costs
166 public bool CanRepair(int healthToAdd, bool takeResources = true)
167 {
168 for (int i = 0; i < adjustedCosts.Length; i++)
169 {
170 adjustedCosts[i].value = new ResourceTypeValue
171 {
172 amount = Mathf.FloorToInt((healthToAdd * repairCosts[i].value.amount) / (MaxHealth - 1)),
173 capacity = Mathf.FloorToInt((healthToAdd * repairCosts[i].value.capacity) / (MaxHealth - 1)),
174 };
175
176 if(!resourceMgr.HasResources(adjustedCosts[i], Building.FactionID))
177 return false;
178 }
179
180 if(takeResources)
181 resourceMgr.UpdateResource(Building.FactionID, adjustedCosts, add: false);
182 return true;
183 }
