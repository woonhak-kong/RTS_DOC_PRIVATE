159 public IReadOnlyList<IUnit> CreateUnitLocal(IUnit unitPrefab, Vector3 spawnPosition, Quaternion spawnRotation, InitUnitParameters initParams)
160 {
161 int unitCount = initParams.isSquad ? initParams.squadCount : 1;
162 IUnit[] createdUnits = new IUnit[unitCount];
163
164 while (unitCount > 0)
165 {
166 IUnit nextUnit = Instantiate(unitPrefab.gameObject, spawnPosition, spawnRotation).GetComponent<IUnit>();
167
168 nextUnit.gameObject.SetActive(true);
169 nextUnit.Init(gameMgr, initParams);
170
171 createdUnits[createdUnits.Length - unitCount] = nextUnit;
172
173 unitCount--;
174 }
175
176 if (initParams.isSquad && squadMgr.IsValid())
177 {
178 squadMgr.JoinToNewSquad(createdUnits);
179 }
180
181 return createdUnits;
