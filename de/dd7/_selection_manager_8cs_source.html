88 public IEnumerable<IEntity> GetEntitiesList(EntityType requiredType, bool exclusiveType, bool localPlayerFaction)
89 {
90 List<IEntity> entities = new List<IEntity>();
91
92 foreach (string entityCode in selectionDic.Keys)
93 {
94 // Reference entity to test.
95 IEntity entity = selectionDic[entityCode][0];
96
97 if (localPlayerFaction && !entity.IsLocalPlayerFaction())
98 return Enumerable.Empty<IEntity>();
99
100 // If this matches the type we're looking for.
101 if (entity.IsEntityTypeMatch(requiredType))
102 {
103 // Use the Linq.Select method when adding entities range to create a new IEnumerable instance and not tie this one with the one in the selection dictionary
104 entities.AddRange(selectionDic[entityCode].Select(nextEntity => nextEntity));
105 continue;
106 }
107
108 // Entity type mismatch and we need all selected entities to be of the same type.
109 if (exclusiveType == true)
110 return Enumerable.Empty<IEntity>();
111 }
112
113 return entities;
114 }

174 private SelectedType AddInternal(IEntity entity, SelectionType type)
175 {
176 if (!entity.IsValid())
177 return SelectedType.notSelected;
178
179 // Overwrite the selection type if the player is forcing multiple selection
180 if (selector.MultipleSelectionModeEnabled == true)
181 type = SelectionType.multiple;
182
183 // If the last selection type was exclusive and this doesn't match the last selected entity type
184 // Force single selection now (so that all currently selected entities will be deselected)
185 if (isCurrSelectionExclusive && entity.Type != lastSelectedEntityType)
186 type = SelectionType.single;
187
188 // Will the selection be marked as exclusive in case the entity is successfully selected? by default no.
189 bool exclusiveOnSuccess = false;
190
191 // Find the current entity type selection options and update accordingly
192 foreach (EntitySelectionOptions options in selectionOptions)
193 if (entity.Type == options.entityType)
194 {
195 if (options.exclusive == true)
196 {
197 exclusiveOnSuccess = true;
198
199 // If the last selected entity does not match with the current type
200 // Force selection type to single (All previous selected entities will be deselected)
201 if (entity.Type != lastSelectedEntityType)
202 type = SelectionType.single;
203 }
204
205 // If the selection type is multiple but that's not allowed for this entity type
206 // Force selection type to single (All previous selected entities will be deselected)
207 if (type == SelectionType.multiple && options.allowMultiple == false)
208 type = SelectionType.single;
209
210 break;
211 }
212
213 switch (type)
214 {
215 case SelectionType.single:
216
217 // Remove all currently selected entities
218 RemoveAll();
219
220 break;
221
222 /*
223 case SelectionType.multiple: //multiple selection:
224
225 // If the multiple selection key is down & entity is already selected -> remove it from the already selected group and stop here.
226 if (mouseSelector.MultipleSelectionKeyDown && IsSelected(entity))
227 {
228 Remove(entity);
229 return false;
230 }
231
232 break;
233 */
234 }
235
236 bool isSelected = IsSelected(entity);
237 if (isSelected)
238 return SelectedType.alreadySelected;
239
240 if (entity.Selection.CanSelect)
241 {
242 // If there's at least another entity of the same type that is already selected then add this entity to the same list
243 if (selectionDic.TryGetValue(entity.Code, out List<IEntity> targetList))
244 targetList.Add(entity);
245 else
246 selectionDic.Add(entity.Code, new List<IEntity> { entity });
247
248 Count++;
249 if (entity.IsLocalPlayerFaction())
250 LocalFactionCount++;
251
252 // Set firstSelected entity
253 if (Count == 1)
254 FirstSelected = entity;
255
256 lastSelectedEntityType = entity.Type;
257 isCurrSelectionExclusive = exclusiveOnSuccess;
258
259 return SelectedType.newlySelected;
260 }
261 else
262 {
263 return SelectedType.notSelected;
264 }
265
266 // Even if the entity is already selected, we do not add it to the selection dictionaries again
267 // But we still return true in order to trigger the corresponding selection events upon which...
268 // ... other mechnacis like range double click selection depend.
269 }

280 public bool Remove(IEntity entity)
281 {
282 if (selectionDic.TryGetValue(entity.Code, out List<IEntity> selectedList))
283 {
284 if (!selectedList.Remove(entity))
285 return false;
286
287 Count--;
288 if (entity.IsLocalPlayerFaction())
289 LocalFactionCount--;
290
291 // If the selection list is now empty, remove the whole entry from the dictionary
292 if (selectedList.Count == 0)
293 selectionDic.Remove(entity.Code);
294
295 if (Count == 1) //if only one entity is left selected, assign it as the single selected entity
296 FirstSelected = GetEntitiesList(EntityType.all, false, false).FirstOrDefault();
297
298 entity.Selection.OnDeselected(new EntityDeselectionEventArgs(DeselectedType.single));
299
300 return true;
301 }
302
303 return false;
304 }
