45 {
46 public EntityItemSource(T source, bool isCreationInstance = false)
47 {
48 Source = source;
49 IsCreationInstance = isCreationInstance;
50 SerializedObj = Source is Entity
51 ? new SerializedObject(Source as Entity)
52 : (Source is StaticBuilding ? new SerializedObject(Source as StaticBuilding) : null);
53 }
54
55 public T Source { get; }
56 public GameObject Obj => Source.gameObject;
57 public bool IsCreationInstance { get; }
58
59 public string ListViewItemTitle => Source.Code;
60 public string UniqueName => Source.Code;
61 public string SearchComparer => Source.Code;
62 public string Title => $"{Source.Name}\n({Source.Code})";
63 public Texture Icon => Source.Icon.IsValid() ? Source.Icon.texture : null;
64 public SerializedObject SerializedObj { private set; get; }
65 public IEnumerable<SerializedProperty> Properties
66 {
67 get
68 {
69 if (!SerializedObj.IsValid())
70 return Enumerable.Empty<SerializedProperty>();
71
72 List<SerializedProperty> props = new List<SerializedProperty>();
73
74 if (Source is Entity)
75 props.Add(SerializedObj.FindProperty("_name"));
76 props.Add(SerializedObj.FindProperty("code"));
77 props.Add(SerializedObj.FindProperty("category"));
78 if (Source is Entity)
79 {
80 props.Add(SerializedObj.FindProperty("description"));
81 props.Add(SerializedObj.FindProperty("icon"));
82 }
83
84 if (Source is Resource || Source is ResourceBuilding)
85 {
86 props.Add(SerializedObj.FindProperty("resourceType"));
87 }
88
89 return props;
90 }

217 private void LoadListViewSourceList()
218 {
219 var nextEntities = RTSEditorHelper
220 .GetEntities()
221 .Values
222 .Where(entity => entity.IsValid() && !excludedCodes.Contains(entity.Code));
223
224 switch(SortBy)
225 {
226 case SortOption.Code:
227 nextEntities = nextEntities.OrderBy(entity => entity.Code);
228 break;
229 case SortOption.Category:
230 nextEntities = nextEntities.OrderBy(entity => entity.Category.FirstOrDefault());
231 break;
232 }
233
234 switch (MenuType)
235 {
236 case NavigationMenuType.Units:
237 ListViewSourceList = nextEntities
238 .Where(entity => entity.IsUnit())
239 .Select(entity => new EntityItemSource<IUnit>(entity as IUnit))
240 .Cast<IItemSource>()
241 .ToList();
242 break;
243 case NavigationMenuType.Buildings:
244 ListViewSourceList = nextEntities
245 .Where(entity => entity.IsBuilding())
246 .Select(entity => new EntityItemSource<IBuilding>(entity as IBuilding))
247 .Cast<IItemSource>()
248 .ToList();
249 break;
250 case NavigationMenuType.Resources:
251 ListViewSourceList = nextEntities
252 .Where(entity => entity.IsResource())
253 .Select(entity => new EntityItemSource<IResource>(entity as IResource))
254 .Cast<IItemSource>()
255 .ToList();
256 break;
257 case NavigationMenuType.ResourceBuildings:
258 ListViewSourceList = nextEntities
259 .Where(entity => entity.IsResource() && entity.IsBuilding())
260 .Select(entity => new EntityItemSource<IBuilding>(entity as IBuilding))
261 .Cast<IItemSource>()
262 .ToList();
263 break;
264 }
265
266 string[] searchResults = RTSEditorHelper.GetMatchingStrings(
267 currSearchPattern,
268 ListViewSourceList.Select(item => item.SearchComparer).ToArray(),
269 new string[0]
270 );
271
272 ListViewSourceList = ListViewSourceList
273 .Where(item => searchResults.Contains(item.SearchComparer))
274 .ToList();
275 }
