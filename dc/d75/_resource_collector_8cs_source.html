237 public override ErrorMessage IsTargetValid (SetTargetInputData data)
238 {
239 TargetData<IResource> potentialTarget = data.target;
240
241 if (!potentialTarget.instance.IsValid())
242 return ErrorMessage.invalid;
243 if (!factionEntity.IsInteractable)
244 return ErrorMessage.invalid;
245 if (!potentialTarget.instance.IsInteractable)
246 return ErrorMessage.uninteractable;
247 if (!potentialTarget.instance.CanCollect)
248 return ErrorMessage.resourceNotCollectable;
249 if (!IsResourceTypeCollectable(potentialTarget.instance.ResourceType))
250 return ErrorMessage.targetPickerUndefined;
251 if (potentialTarget.instance.Health.IsDead)
252 return ErrorMessage.targetDead;
253 // Check if this is not the same resource that is being collected before checking if it has max collectors (in case player is asking the unit to collect the resource it is actively collecting).
254 if (!factionEntity.CanMove() && !IsTargetInRange(transform.position, potentialTarget))
255 return ErrorMessage.targetOutOfRange;
256
257 if (!potentialTarget.instance.CanCollectOutsideBorder && !potentialTarget.instance.IsFriendlyFaction(factionEntity))
258 return ErrorMessage.resourceOutsideTerritory;
259 if (resourceMgr.HasResourceTypeReachedLimitCapacity(potentialTarget.instance.ResourceType, unit.FactionID))
260 return ErrorMessage.resourceTypeLimitCapacityReached;
261 if (!potentialTarget.instance.CanAutoCollect)
262 {
263 if (!unit.DropOffSource.IsValid())
264 return ErrorMessage.dropOffComponentMissing;
265 if (!isNextAutoSearchConditionActive && unit.DropOffSource.HasReachedMaxCapacity(potentialTarget.instance.ResourceType))
266 {
267 unit.DropOffSource.AttemptStartDropOff(force: true, potentialTarget.instance.ResourceType);
268 return ErrorMessage.dropOffMaxCapacityReached;
269 }
270 }
271 if (OnNextAutoSearchCondition(potentialTarget, out ErrorMessage errorMessage))
272 return errorMessage;
273
274 return potentialTarget.instance.WorkerMgr.CanMove(
275 unit,
276 new AddableUnitData
277 {
278 allowDifferentFaction = true
279 });
280 }

282 protected override void OnTargetPostLocked(SetTargetInputData input, bool sameTarget)
283 {
284 if (!logger.RequireTrue(Target.instance.CanAutoCollect || unit.DropOffSource.IsValid(),
285 $"[{GetType().Name} - {Entity.Code}] A component that extends {typeof(IDropOffSource).Name} interface must be attached to this resource collector since resources can not be auto collected!"))
286 return;
287
288 var handler = collectableResourcesDic[Target.instance.ResourceType];
289 // In this component, the in progress object depends on the type of resource that is being collected.
290 inProgressObject = handler.enableObject;
291 progressOverrideController = handler.animatorOverrideController;
292 progressEnabledAudio = handler.enableAudio;
293 sourceEffect = handler.sourceEffect;
294 targetEffect = handler.targetEffect;
295
296 if (Target.instance.WorkerMgr.Move(
297 unit,
298 new AddableUnitData(
299 sourceTargetComponent: this,
300 input,
301 allowDifferentFaction: Target.instance.CanCollectOutsideBorder)) != ErrorMessage.none)
302 {
303 unit.DropOffSource?.Cancel();
304 Stop();
305 return;
306 }
307
308 // If the movement component is unable to calculate the path towards the target, it will set the unit back to idle
309 // And in this case, we do not continue
310 if (unit.IsIdle)
311 return;
312
313 if(input.playerCommand && unit.IsLocalPlayerFaction())
314 audioMgr.PlaySFX(handler.orderAudio, loop: false);
315
316 // For the worker component manager, make sure that enough worker positions is available even in the local method.
317 // Since they are only updated in the local method, meaning that the SetTarget method would always relay the input in case a lot of consecuive calls are made...
318 //... on the same resource from multiple collectors.
319
320 if (sameTarget)
321 {
322 AttemptDropOff();
323 return;
324 }
325
326 if (!Target.instance.Health.IsValid())
327 logger.LogError($"[ResourceCollector] Target resource of code '{Target.instance.Code}' does not have a valid health component.", source: Target.instance);
328
329 Target.instance.Health.EntityHealthUpdated += HandleTargetHealthUpdated;
330 Target.instance.Health.EntityDead += HandleTargetResourceDead;
331
332 globalEvent.RaiseEntityComponentTargetLockedGlobal(this, new TargetDataEventArgs(Target));
333
334 AttemptDropOff();
335 }

345 private bool EnableNextAutoSearchCondition(IsTargetValid<IResource> condition, SetTargetInputData lastInput)
346 {
347 if (!condition.IsValid())
348 return false;
349
350 nextAutoSearchCondition = condition;
351 isNextAutoSearchConditionActive = true;
352
353 if(gridSearch.Search(lastInput.target.instance.transform.position, TargetFinder.Range, IsTargetValid, playerCommand: false, out IResource potentialTarget) == ErrorMessage.none)
354 {
355 SetTarget(new SetTargetInputData
356 {
357 target = new TargetData<IEntity>
358 {
359 instance = potentialTarget,
360 position = potentialTarget.transform.position
361 },
362 playerCommand = lastInput.playerCommand,
363 fromRallypoint = lastInput.fromRallypoint,
364 fromTasksQueue = lastInput.fromTasksQueue,
365 });
366 }
367
368 DisableNextAutoSearchCondition();
369
370 return true;
371 }

373 private bool OnNextAutoSearchCondition(TargetData<IResource> data, out ErrorMessage errorMsg)
374 {
375 errorMsg = ErrorMessage.none;
376 if (!isNextAutoSearchConditionActive)
377 return false;
378
379 errorMsg = nextAutoSearchCondition(data);
380 return errorMsg != ErrorMessage.none;

383 protected override void OnSetTargetError(SetTargetInputData input, ErrorMessage errorMsg)
384 {
385 if (!input.playerCommand && !input.fromRallypoint)
386 return;
387
388 switch (errorMsg)
389 {
390 case ErrorMessage.workersMaxAmountReached:
391 switch(onTargetResourceFullSearch)
392 {
393 case ResourceCollectorSearchBehaviour.prioritizeLastResourceType:
394 IResource lastResource = (input.target.instance as IResource);
395
396 EnableNextAutoSearchCondition((data =>
397 data.instance.ResourceType == lastResource.ResourceType
398 ? ErrorMessage.none
399 : ErrorMessage.resourceTypeMismatch),
400 input);
401
402 break;
403 }
404 break;
405 }
