94 #region Starting
95 protected override void OnStart()
96 {
97 currentPlaceAroundHandler = new BuildingPlaceAroundHandler(gameMgr,current.instance, current.options.placeAroundData);
98
99 // If we are unable to set the place around data for the next pending building then we discard it and move to the next one.
100 // And make sure that the building is allowed inside its currently assigned border
101 if (!TrySetNextPlaceAroundData()
102 || !current.instance.PlacerComponent.IsBuildingInBorder())
103 {
104 EventLogger?.Invoke($"'{current.instance.Code}': Active Placement Stop - Unable to set place around data.");
105 Stop();
106 return;
107 }
108
109 current.instance.gameObject.SetActive(true);
110 ResetMovementFields();
111
112 // Start the height check coroutine to keep the building always on top of the terrain.
113 // This coroutine is only active when this component is placing a pending building.
114 if (heightCheckCoroutine.IsValid())
115 return;
116
117 heightCheckCoroutine = HeightCheck(heightCheckReload);
118 npcBuildingPlacer.StartCoroutine(heightCheckCoroutine);
119 }
120
121 public bool TrySetNextPlaceAroundData ()
122 {
123 while(currentPlaceAroundHandler.TrySetNextData())
124 {
125 IEnumerable<IEntity> nextPlaceAroundEntities = current.options.initialCenter.EntitiesInRange
126 .Where(entity => currentPlaceAroundHandler.CurrData.IsValidType(entity.ToSetTargetInputData(playerCommand: false)) == ErrorMessage.none);
127
128 IEntity nextPlaceAroundEntity = nextPlaceAroundEntities
129 .ElementAtOrDefault(UnityEngine.Random.Range(0, nextPlaceAroundEntities.Count()));
130
131 if(nextPlaceAroundEntity.IsValid())
132 {
133 // Get a suitable position for the new building on the build around position
134 terrainMgr.GetTerrainAreaPosition(
135 nextPlaceAroundEntity.transform.position,
136 current.instance.PlacerComponent.PlacableTerrainAreas,
137 out currentPlaceAroundPosition
138 );
139 currentPlaceAroundPosition.y += placerMgr.BuildingPositionYOffset;
140
141 // Offset the building by the place around data minimum distance and the build around entity radius
142 Vector3 nextBuildingPosition = currentPlaceAroundPosition;
143 nextBuildingPosition.x += nextPlaceAroundEntity.Radius + currentPlaceAroundHandler.CurrData.range.min;
144
145 current.instance.transform.position = nextBuildingPosition;
146
147 // Pick a random starting position for building by randomly rotating it around its build around positio
148 current.instance.transform.RotateAround(currentPlaceAroundPosition, Vector3.up, UnityEngine.Random.Range(0.0f, 360.0f));
149 // Keep initial rotation (because the RotateAround method will change the building's rotation as well which we do not want)
150 current.instance.transform.rotation = current.task.Prefab.transform.rotation;
151
152 return true;
153 }
154 }
155
156 return false;
157 }

187 private IEnumerator HeightCheck(float waitTime)
188 {
189 while (true)
190 {
191 yield return new WaitForSeconds(waitTime);
192
193 if (current.instance.IsValid())
194 {
195 terrainMgr.SampleHeight(current.instance.transform.position, placerMgr.PlacableTerrainAreas, out float height);
196 //Debug.Log("HeightCheck(" + waitTime + ") = " + height);
197 current.instance.transform.position = new Vector3(
198 current.instance.transform.position.x,
199 height + placerMgr.BuildingPositionYOffset,
200 current.instance.transform.position.z);
201
202 //Debug.Log("HeightCheck(" + waitTime + ") = " + current.instance.transform.position);
203 }
204 }

208 #region Handling Movement
209 protected override void OnActiveUpdate()
210 {
211 placementDelayTimer -= Time.deltaTime;
212 placementMoveTimer -= Time.deltaTime;
213
214 // Invalid pending building instance
215 if (!current.instance.IsValid())
216 {
217 EventLogger?.Invoke($"'{current.instance.Code}': Active Placement Stop - Invalid Pending Building Instance");
218 Stop();
219 return;
220 }
221
222 // If the pending building leaves the allowed maximum range to the build around position, then try if we can get another place around data to be active for this pending building
223 // If none is found then we can discard this pending building and move to the next one.
224 if(Vector3.Distance(current.instance.transform.position, CurrentPlaceAroundPosition) >= CurrentMaxPlacementRange)
225 {
226 if (TrySetNextPlaceAroundData())
227 {
228 ResetMovementFields();
229
230 EventLogger?.Invoke($"'{current.instance.Code}': Active Placement Update - Next Place Around Set");
231 }
232 else
233 {
234 EventLogger?.Invoke($"'{current.instance.Code}': Active Placement Stop - Place Around Data No Longer Met");
235 Stop();
236 }
237
238 return;
239 }
240
241 // Keep rotating the building around its build around position
242 // Make sure to cache the building's local rotation as the 'RotateAround' method will affect that while we need it unchanged
243 Quaternion nextBuildingRotation = current.options.disableRotation
244 ? current.instance.transform.rotation
245 : RTSHelper.GetLookRotation(current.instance.transform, CurrentPlaceAroundPosition, true);
246
247 current.instance.transform.RotateAround(CurrentPlaceAroundPosition, Vector3.up, rotationMultiplier * rotationSpeed * Time.deltaTime);
248
249 current.instance.transform.rotation = nextBuildingRotation;
250
251 if (placementMoveTimer <= 0.0f)
252 {
253 // Reset timer
254 placementMoveTimer = placementMoveReload.RandomValue + (placementMoveReloadInc.RandomValue * placementMoveReloadIncCount);
255 placementMoveReloadIncCount++;
256
257 // Move building away from build around position by the defined movement distance
258 Vector3 mvtDir = (current.instance.transform.position - CurrentPlaceAroundPosition).normalized;
259 mvtDir.y = 0.0f;
260 if (mvtDir == Vector3.zero)
261 mvtDir = new Vector3(1.0f, 0.0f, 0.0f);
262 Vector3 nextPlacementPos = current.instance.transform.position + mvtDir * moveDistance.RandomValue;
263
264 if (IsGridPlacementActive)
265 {
266 placerMgr.GridHandler.TryGetCellPosition(nextPlacementPos, out nextPlacementPos);
267 }
268
269 current.instance.PlacerComponent.OnPlacementUpdate(nextPlacementPos);
270 }
271
272
273 if (placementDelayTimer <= 0.0f
274 && current.instance.PlacerComponent.CanPlace
275 && currentPlaceAroundHandler.IsPlaceAroundValid())
276 Complete();
277 }
278 #endregion
