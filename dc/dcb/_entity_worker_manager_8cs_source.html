136 public ErrorMessage CanMove(IUnit worker, AddableUnitData addableData = default)
137 {
138 if (!worker.IsValid())
139 return ErrorMessage.invalid;
140 else if (!worker.IsInteractable)
141 return ErrorMessage.uninteractable;
142 else if (worker.Health.IsDead)
143 return ErrorMessage.healthDead;
144
145 else if (!addableData.allowDifferentFaction && !RTSHelper.IsSameFaction(worker, Entity))
146 return ErrorMessage.factionMismatch;
147
148 // Already reached maxmimum amount and this is a new worker attempting to be added
149 else if (HasMaxAmount && !workers.Contains(worker))
150 {
151 // If no possible destination is available then stop all entity target components of the worker
152 worker.SetIdle();
153 return ErrorMessage.workersMaxAmountReached;
154 }
155
156 return ErrorMessage.none;

159 public ErrorMessage Move(IUnit worker, AddableUnitData addableData)
160 {
161 ErrorMessage errorMsg;
162 if ((errorMsg = CanMove(worker, addableData)) != ErrorMessage.none)
163 {
164 if (addableData.playerCommand && RTSHelper.IsLocalPlayerFaction(worker))
165 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
166 {
167 message = errorMsg,
168
169 source = Entity,
170 target = worker
171 });
172
173 return errorMsg;
174 }
175
176 // Getting the addable position is handled here and not in the GetAddablePosition() method
177 // This is because when an addable position is determined, the worker entity gets added to the workers list of this worker manager
178 // The reason behind that is that we want the worker to reserve a spot in this worker manager before they get to their addable position
179
180 if (!workerToPositionIndex.TryGetValue(worker, out int positionIndex))
181 {
182 // If no specific transforms are assigned to the worker positions then get the first free slot
183 if (!workerPositions[freePositionIndexes[0]].IsValid())
184 {
185 positionIndex = 0;
186 }
187 // Else look for the closest worker position to the next worker
188 else
189 {
190 float closestDistance = Mathf.Infinity;
191 float nextDistance;
192
193 for (int i = 0; i < freePositionIndexes.Count; i++)
194 {
195 nextDistance = Vector3.Distance(workerPositions[freePositionIndexes[i]].position, worker.transform.position);
196 if (nextDistance < closestDistance)
197 {
198 positionIndex = freePositionIndexes[i];
199 closestDistance = nextDistance;
200 }
201 }
202 }
203
204 freePositionIndexes.Remove(positionIndex);
205
206 workers.Add(worker);
207 workerToPositionIndex.Add(worker, positionIndex);
208
209 RaiseWorkerAdded(Entity, new EntityEventArgs<IUnit>(worker));
210 }
211
212 Vector3 destination = workerPositions[positionIndex].IsValid() ? workerPositions[positionIndex].position : Entity.transform.position;
213 float radius = workerPositions[positionIndex].IsValid() ? 0.0f : Entity.Radius;
214
215 errorMsg = mvtMgr.SetPathDestinationLocal(
216 new SetPathDestinationData<IEntity>
217 {
218 source = worker,
219 destination = destination,
220 offsetRadius = radius,
221 target = Entity,
222 mvtSource = new MovementSource
223 {
224 playerCommand = addableData.playerCommand,
225
226 sourceTargetComponent = addableData.sourceTargetComponent,
227
228 targetAddableUnit = this,
229 targetAddableUnitPosition = destination,
230
231 isMoveAttackRequest = addableData.isMoveAttackRequest
232 }
233 });
234
235 if (errorMsg != ErrorMessage.none
236 && addableData.playerCommand && RTSHelper.IsLocalPlayerFaction(worker))
237 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
238 {
239 message = errorMsg,
240
241 source = Entity,
242 target = worker
243 });
244
245 return errorMsg;
246 }
