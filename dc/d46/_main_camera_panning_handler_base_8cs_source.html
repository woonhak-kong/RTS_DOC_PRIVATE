94 public void Apply()
95 {
96 // Prioritize following target
97 if (followTarget.IsValid())
98 {
99 OnFollowTargetPanning();
100 return;
101 }
102 else
103 {
104 if (!IsActive)
105 return;
106
107 // Smoothly update the last panning direction towards the current one
108 LastPanDirection = Vector3.Lerp(LastPanDirection, currPanDirection, panningSpeed.smoothFactor);
109
110 OnNonFolloTargetPanning();

113 cameraController.RaiseCameraTransformUpdated();

144 public void LookAt(Vector3 targetPosition, bool smooth, float smoothFactor = 0.1f)
145 {
146 cameraController.RotationHandler.ResetRotation(smooth: true);
147
148 // Adding half of the target position height to the X and Z axis of the camera's target position allows it to center over the target position for any variying height
149 targetPosition = new Vector3(
150 targetPosition.x + CurrOffsetX + targetPosition.y / 2.0f,
151 // Enforce minimum look at camera height in case FOV is not used, which is the height before which pivoting to reach minimum height is enabled
152 cameraController.ZoomHandler.UseFOV
153 ? cameraController.MainCamera.transform.position.y
154 : Mathf.Max(cameraController.MainCamera.transform.position.y, cameraController.ZoomHandler.LookAtTargetMinHeight),
155 targetPosition.z + CurrOffsetZ + targetPosition.y / 2.0f);
156
157 SetPosition(
158 smooth
159 ? Vector3.Lerp(cameraController.MainCamera.transform.position, targetPosition, smoothFactor)
160 : targetPosition);
161
162 cameraController.RaiseCameraTransformUpdated();
163 }
