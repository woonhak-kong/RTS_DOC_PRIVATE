31 #region Handling Mouse Selection
32 protected override void OnActiveUpdate()
33 {
34 bool leftButtonDown = Input.GetMouseButtonUp(0);
35 bool rightButtonDown = Input.GetMouseButtonUp(1);
36
37 // If the mouse pointer is over a UI element or the minimap, we will not detect entity selection
38 // In addition, we make sure that one of the mouse buttons are down
39 if (!leftButtonDown && !rightButtonDown)
40 return;
41
42 if (raycast.Hit(mainCameraController.ScreenPointToRay(Input.mousePosition), out RaycastHit hit))
43 {
44
45 // Get the entity that the player clicked in or see if the player clicked on a terrain area.
46 IEntity hitEntity = hit.transform.gameObject.GetComponent<EntitySelectionCollider>()?.Entity;
47 bool hitTerrain = terrainMgr.IsTerrainArea(hit.transform.gameObject);
48 Debug.Log("Right1 = " + hitTerrain);
49 if (rightButtonDown)
50 {
51 // If there was an active awaiting task then disable it since it can only be completed with a left mouse button click
52 if (taskMgr.AwaitingTask.IsEnabled)
53 {
54 taskMgr.AwaitingTask.Disable();
55 Debug.Log("Right2");
56 return;
57 }
58
59 if (hitEntity.IsValid())
60 {
61 Debug.Log("Right3");
62 hitEntity.Selection.OnDirectAction();
63 }
64 else if (hitTerrain)
65 {
66 Debug.Log("Right4");
67 selectionMgr.GetEntitiesList(EntityType.all, false, true)
68 .SetTargetFirstMany(new SetTargetInputData
69 {
70 target = hit.point,
71 playerCommand = true,
72 includeMovement = true
73 });
74 }
75 }
76 else if (leftButtonDown)
77 {
78 if (!hitEntity.IsValid())
79 {
80 // Complete awaiting task on terrain click.
81 if (hitTerrain && taskMgr.AwaitingTask.IsEnabled)
82 foreach (IEntityTargetComponent sourceComponent in taskMgr.AwaitingTask.Current.sourceTracker.EntityTargetComponents)
83 sourceComponent.OnAwaitingTaskTargetSet(taskMgr.AwaitingTask.Current, hit.point);
84 // No awaiting task and terrain click = deselecting currently selected entities
85 else
86 selectionMgr.RemoveAll();
87
88 taskMgr.AwaitingTask.Disable();
89 return;
90 }
91
92 // Awaiting task is active with a valid hit entity
93 if (taskMgr.AwaitingTask.IsEnabled)
94 {
95 hitEntity.Selection.OnAwaitingTaskAction(taskMgr.AwaitingTask.Current);
96 taskMgr.AwaitingTask.Disable();
97 }
98 // If no awaiting task is active then proceed with regular selection
99 else
100 selectionMgr.Add(hitEntity, SelectionType.single, isLocalPlayerClickSelection: true);
101 }
102 }
103 }
104 #endregion

106 }
