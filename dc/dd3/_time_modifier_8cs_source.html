232 #region Handling Global Timers
233 private void Update()
234 {
235 int i;
236 if (nextTimers.Count > 0)
237 {
238 for(i = 0; i < nextTimers.Count; i++)
239 {
240 globalTimers.Add(nextTimers[i]);
241 globalTimersDic.Add(nextTimers[i], nextTimersDic[nextTimers[i]]);
242 }
243
244 nextTimers.Clear();
245 nextTimersDic.Clear();
246 }
247
248 globalTimersCount = globalTimers.Count;
249
250 if (globalTimersCount == 0)
251 return;
252
253 i = 0;
254 currFrameTimerRemovals = 0;
255 while(i < globalTimers.Count)
256 {
257 if(globalTimers[i].ModifiedDecrease())
258 {
259 currFrameTimerRemovals++;
260 RemoveTimer(i);
261
262 if (RTSOptimizations.LIMIT_TIMER_REMOVALS_PER_FRAME && currFrameTimerRemovals >= RTSOptimizations.MAX_TIMER_REMOVALS_PER_FRAME)
263 break;
264 else
265 continue;
266 }
267
268 i++;
269 }
270 }

281 private void RemoveTimer(int i)
282 {
283 globalTimersDic.TryGetValue(globalTimers[i], out Action value);
284
285 globalTimersDic.Remove(globalTimers[i]);
286 globalTimers.RemoveAt(i);
287
288 if (value.IsValid())
289 value();
290 }

292 public void RemoveTimer(GlobalTimeModifiedTimer timer, bool triggerRemovalCallback = true)
