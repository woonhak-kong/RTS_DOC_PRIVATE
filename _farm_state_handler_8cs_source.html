152 private void TryAddNextState(bool force = false)
153 {
154 int nextStateID = currStateID.GetNextIndex(farmStates);
155
156 if (farmStates.Length < 2
157 || (!force && lastCollectedResources < farmStates[nextStateID].requiredResources))
158 return;
159
160 if(!force)
161 lastCollectedResources -= farmStates[nextStateID].requiredResources;
162
163 // Add the next farm state as the state that the farm is attempting to reach
164 targetStateIDs.AddLast(currStateID.GetNextIndex(farmStates));
