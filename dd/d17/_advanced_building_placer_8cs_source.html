116 #region Handling Placement Status
117 protected override void OnPlacementStarted(BuildingPlacementUpdateData data)
118 {
119 segmentIndex = data.sourceSegmentIndex;
120
121 // If current placement is not in segmentation mode, nothing to do.
122 if (data.segmentCount < minSegmentCountForReplacement)
123 return;
124
125 if (data.sourceSegmentIndex == 0) // First segment
126 {
127 if (firstSegmentReplacement.IsInitialized)
128 {
129 placerMgr.Add(
130 Building.FactionID,
131 firstSegmentReplacement,
132 new BuildingPlacementOptions
133 {
134 isReplacingExisting = true,
135 targetIndex = 0
136 });
137 return;
138 }
139 }
140 else if (data.sourceSegmentIndex == data.segmentCount - 1) // Last segment
141 {
142 if (lastSegmentReplacement.IsInitialized)
143 {
144 placerMgr.Add(
145 Building.FactionID,
146 lastSegmentReplacement,
147 new BuildingPlacementOptions
148 {
149 isReplacingExisting = true,
150 targetIndex = data.segmentCount - 1
151 });
152 return;
153 }
154 }
155 // Periodic segment replacement
156 else if (segmentPeriodReplacement.enabled
157 && segmentPeriodReplacement.targetSegment.IsInitialized
158 && data.sourceSegmentIndex % segmentPeriodReplacement.period == 0)
159 {
160 if (!segmentPeriodReplacement.disableNextToEdges
161 || (data.sourceSegmentIndex != 1 && data.sourceSegmentIndex != data.segmentCount - 2))
162 placerMgr.Add(
163 Building.FactionID,
164 segmentPeriodReplacement.targetSegment,
165 new BuildingPlacementOptions
166 {
167 isReplacingExisting = true,
168 targetIndex = data.sourceSegmentIndex
169 });
170 }
171 }
172
173 protected override void OnPlacementStopped(BuildingPlacementUpdateData data)
174 {
175 // To replace segments back to the default ones
176 if (data.segmentCount == 0
177 || !defaultSegmentReplacement.IsInitialized)
178 return;
179
180 // Minimum required segment count for replacement not reached -> back to default segments.
181 if (data.segmentCount < minSegmentCountForReplacement)
182 {
183 // First building in the segmentation mode and there is a valid replacement
184 if (data.sourceSegmentIndex == 0)
185 {
186 placerMgr.Add(Building.FactionID,defaultSegmentReplacement,
187 new BuildingPlacementOptions
188 {
189 isReplacingExisting = true,
190 targetIndex = 0
191 });
192
193 return;
194 }
195 // Last segment
196 else if (data.sourceSegmentIndex == data.segmentCount - 1)
197 {
198 placerMgr.Add(Building.FactionID,defaultSegmentReplacement,
199 new BuildingPlacementOptions
200 {
201 isReplacingExisting = true,
202 targetIndex = data.segmentCount - 1
203 });
204
205 return;
206 }
207 }
208
209 // Making sure segments next to the edges are the default ones, in case that is enabled.
210 if (segmentPeriodReplacement.disableNextToEdges
211 && data.sourceSegmentIndex % segmentPeriodReplacement.period == 0
212 && (data.sourceSegmentIndex == 1 || data.sourceSegmentIndex == data.segmentCount - 2))
213 {
214 placerMgr.Add(Building.FactionID,defaultSegmentReplacement,
215 new BuildingPlacementOptions
216 {
217 isReplacingExisting = true,
218 targetIndex = data.sourceSegmentIndex
219 });
220
221 return;
222 }
223 }

226 #region Handling Position Update
227 protected override void OnPositionUpdate(Vector3 position)
228 {
229 if (segmentSnapping.enabled)
230 {
231 IBuilding snappableBuilding = RTSHelper.GetClosestEntity(
232 Building.transform.position,
233 Building.FactionMgr.Buildings,
234 target =>
235 segmentSnapping.targetOptions.Contains(target)
236 && Vector3.Distance(target.transform.position, Building.transform.position) <= segmentSnapping.range);
237
238 if (snappableBuilding.IsValid())
239 {
240 if(placerMgr.Add(Building.FactionID, null,
241 new BuildingPlacementOptions
242 {
243 isSnappingSegment = true,
244 snapTarget = snappableBuilding,
245 targetIndex = segmentIndex
246 }) == true)
247 return;
248 }
249 }
250
251 if (!placerMgr.GridHandler.IsValid()
252 || !placerMgr.GridHandler.IsEnabled)
253 {
254 base.OnPositionUpdate(position);
255 return;
256 }
257
258 // Grid Placement position update
259 Building.transform.position = gridOptions.ApplyPivotPoint(position);
260 }
