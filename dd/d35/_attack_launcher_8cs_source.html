54 #region Triggering Launch
55 public void Trigger(Action launchCompleteCallback, IReadOnlyCollection<AttackObjectLaunchLogInput> nextLaunchLogInput)
56 {
57 this.launchCompleteCallback = launchCompleteCallback;
58 // Direct attack? apply damage to target and complete attack.
59 if (!useAttackObjects)
60 {
61 SourceAttackComp.Damage.Trigger(SourceAttackComp.Target.instance, RTSHelper.GetAttackTargetPosition(SourceAttackComp.Target));
62 Complete();
63 return;
64 }
65
66 lastCompletedLaunchLogIndex = -1;
67
68 // Pre-assigned launch log
69 if (nextLaunchLogInput != null && nextLaunchLogInput.Count > 0)
70 {
71 foreach(var nextInput in nextLaunchLogInput)
72 launchLog.Add(new AttackObjectLaunchLog(sources, nextInput));
73 return;
74 }
75
76 // Non direct attack? start coroutine to handle attack objects.
77 switch(launchType)
78 {
79 case AttackObjectLaunchType.random:
80
81 // Random attack object launch? mark attack as complete after launching one attack object.
82 launchLog.Add(new AttackObjectLaunchLog(sources, sourceIndex: UnityEngine.Random.Range(0, sources.Length), isLastLaunch: true));
83 break;
84
85 case AttackObjectLaunchType.inOrder:
86
87 launchLog.Add(new AttackObjectLaunchLog(sources, sourceIndex: 0, isLastLaunch: false));
88 break;
89
90 case AttackObjectLaunchType.simultaneous:
91
92 for (int sourceIndex = 0; sourceIndex < sources.Length; sourceIndex++)
93 launchLog.Add(new AttackObjectLaunchLog(sources, sourceIndex, isLastLaunch: false));
94 break;
95 }
96 }
97 #endregion

99 #region Handling Launch Progress
100 public void Update()
101 {
102 if (launchLog.Count == 0)
103 return;
104
105 for (int launchID = 0; launchID < launchLog.Count; launchID++)
106 {
107 if (launchID <= lastCompletedLaunchLogIndex)
108 continue;
109 AttackObjectLaunchLog nextLaunch = launchLog[launchID];
110
111 if (nextLaunch.preDelayTimer.CurrValue >= 0.0f)
112 {
113 if (!nextLaunch.preDelayTimer.ModifiedDecrease())
114 return;
115
116 nextLaunch.attackObject = sources[nextLaunch.sourceIndex].Launch(attackMgr, SourceAttackComp);
117
118 RaiseAttackLaunched(new AttackLaunchEventArgs(nextLaunch));
119 }
120 else
121 {
122 if (!nextLaunch.postDelayTimer.ModifiedDecrease())
123 return;
124
125 switch (launchType)
126 {
127 case AttackObjectLaunchType.inOrder:
128
129 // Already launched the last attack object in the sources? stop here
130 // If not, move to the next one by creating a new coroutine for it.
131 if (nextLaunch.sourceIndex < sources.Length - 1)
132 {
133 launchLog.Add(new AttackObjectLaunchLog(sources, sourceIndex: nextLaunch.sourceIndex + 1, isLastLaunch: false));
134 // mark as complete as soon as it the current launch adds the next launch
135 lastCompletedLaunchLogIndex = launchID;
136 }
137 break;
138
139 default:
140 break;
141 }
142
143 // Marked as the last attack object to launch or this is indeed the last one in the array.
144 if (nextLaunch.isLastLaunch || nextLaunch.sourceIndex == sources.Length - 1)
145 Complete();
146 }
147 }
148 }
149 #endregion
150
151 #region Completing Launch

153 {
154 launchCompleteCallback();
155 Reset();
156 }

158 public void Reset()
159 {
160 foreach(AttackObjectLaunchLog nextLaunch in launchLog)
161 {
162 // If the reset occurs while the attack object is still in delay
163 // ... this would call the need for disabling the attack object entirely
164 // Example case is when the source attack entity is destroyed and attack objects are still in delay.
165 if (nextLaunch.attackObject.IsValid() && nextLaunch.attackObject.InDelay)
166 nextLaunch.attackObject.Deactivate(useDisableTime: false);
167 }
168
169 launchLog.Clear();
170 }
