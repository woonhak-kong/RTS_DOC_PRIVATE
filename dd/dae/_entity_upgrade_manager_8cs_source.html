99 #region Launching Entity Upgrades
100 public ErrorMessage LaunchLocal(EntityUpgrade upgrade, EntityUpgradeElementSource upgradeSource, int factionID)
101 {
102 if (!RTSHelper.IsValidFaction(factionID))
103 {
104 logger.LogError($"[EntityUpgradeManager - {upgrade}] Attempting to launch entity upgrade for invalid faction ID: {factionID} is not allowed!", source: this);
105 return ErrorMessage.invalid;
106 }
107 else if (!upgradeSource.UpgradeTarget.IsValid())
108 {
109 logger.LogError($"[EntityUpgradeManager - {upgrade}] Attempting to launch upgrade for invalid target entity is not allowed!", source: this);
110 return ErrorMessage.invalid;
111 }
112 else if (IsLaunched(upgrade, upgradeSource, factionID))
113 {
114 logger.LogError($"[EntityUpgradeManager] Attempting to launch entity upgrade for entity of code '{upgrade.SourceCode}' (Faction ID: {factionID}) that has been already launched!", source: this);
115 return ErrorMessage.upgradeLaunched;
116 }
117 else if (upgrade.SourceEntity.IsValid() && upgrade.SourceEntity.Type != upgradeSource.UpgradeTarget.Type)
118 {
119 logger.LogError($"[EntityUpgradeManager] Upgrade source entity ('{upgrade.SourceEntity?.Type}') and target entity ('{(upgradeSource.UpgradeTarget).Type}') have different types!", source: this);
120 return ErrorMessage.upgradeTypeMismatch;
121 }
122
123 UpgradeElement<IEntity> newElement = new UpgradeElement<IEntity>
124 {
125 sourceCode = upgrade.SourceCode,
126 target = upgradeSource.UpgradeTarget
127 };
128
129 if (upgrade.SourceInstanceOnly)
130 {
131 if(upgrade.SourceEntity.IsValid())
132 UpgradeInstance(upgrade.SourceEntity, newElement, factionID, upgradeSource.EntityComponentMatcher, upgrade.UpgradeEffect);
133 }
134 else
135 {
136 elements[factionID].Add(newElement);
137
138 // Upgrade all spawned instances of the same type of the upgrade source?
139 if (upgrade.UpdateSpawnedInstances && upgrade.SourceEntity.IsValid())
140 // Why use ToArray() here? because we do not want to update the FactionManager's FactionEntities collection directly since we will be destroying/creating new entities
141 foreach (IFactionEntity factionEntity in gameMgr.GetFactionSlot(factionID).FactionMgr.FactionEntities.ToArray())
142 if (factionEntity.Code == upgrade.SourceCode)
143 UpgradeInstance(factionEntity, newElement, factionID, upgradeSource.EntityComponentMatcher, upgrade.UpgradeEffect);
144
145 if(upgradeSource.UpgradeTarget.IsBuilding())
146 globalEvent.RaiseBuildingUpgradedGlobal(
147 upgrade.SourceEntity as IBuilding,
148 new UpgradeEventArgs<IEntity>(newElement, factionID, null));
149 else if(upgradeSource.UpgradeTarget.IsUnit())
150 globalEvent.RaiseUnitUpgradedGlobal(
151 upgrade.SourceEntity as IUnit,
152 new UpgradeEventArgs<IEntity>(newElement, factionID, null));
153 else
154 logger.LogError($"[EntityUpgradeManager] Upgrading 'resource' instances is currently not allowed!");
155
156 globalEvent.RaiseEntityUpgradedGlobal(
157 upgrade.SourceEntity,
158 new UpgradeEventArgs<IEntity>(newElement, factionID, null));
159 }
160
161 //if there are upgrades that get triggerd from this one, launch them
162 foreach (TriggerUpgrade triggerUpgrade in upgradeSource.TriggerUpgrades)
163 triggerUpgrade.upgradeComp.LaunchLocal(gameMgr, triggerUpgrade.upgradeIndex, factionID);
164
165 return ErrorMessage.none;
166 }

168 // Upgrades a faction entity instance locally
169 private void UpgradeInstance(IFactionEntity sourceInstance, UpgradeElement<IEntity> upgradeElement, int factionID, IEnumerable<EntityUpgradeComponentMatcherElement> entityComponentMatcher, IEffectObject upgradeEffect)
170 {
171 // Upgraded instances get the same curr health to max health ratio when they are created as the instnaces that they were upgraded from
172 float healthRatio = sourceInstance.Health.CurrHealth / (float)sourceInstance.Health.MaxHealth;
173
174 // We want to re-select the upgraded instance after creating if this is the case
175 bool wasSelected = sourceInstance.Selection.IsSelected;
176
177 IEntity upgradedInstance = null;
178
179 if (sourceInstance.IsBuilding())
180 {
181 IBuilding currBuilding = sourceInstance as IBuilding;
182
183 // Get the current builders of this building if there are any
184 // And make them stop building the instance of the building since it will be destroyed.
185 IEnumerable<IUnit> currBuilders = currBuilding.WorkerMgr.Workers.ToArray();
186 foreach (IUnit unit in currBuilders)
187 unit.BuilderComponent.Stop();
188
189 // If the source building has Border component, then remove all resources under its territory
190 // The resource removal would occur normally when the source building is destroyed
191 // however, by that time, the same resources would be added to the upgraded building instance
192 // and removing them later will mean that they be removed from the new upgraded building instance border as well
193 if (currBuilding.BorderComponent.IsValid())
194 currBuilding.BorderComponent.Disable();
195
196 // Create upgraded instance of the building
197 upgradedInstance = buildingMgr.CreatePlacedBuildingLocal(
198 upgradeElement.target as IBuilding,
199 sourceInstance.transform.position,
200 sourceInstance.transform.rotation,
201 new InitBuildingParameters
202 {
203 free = sourceInstance.IsFree,
204 factionID = factionID,
205
206 setInitialHealth = true,
207 initialHealth = (int)(healthRatio * upgradeElement.target.gameObject.GetComponent<IEntityHealth>().MaxHealth),
208
209 buildingCenter = currBuilding.CurrentCenter,
210
211 giveInitResources = true,
212 });
213
214 foreach (IUnit unit in currBuilders)
215 unit.SetTargetFirst(new SetTargetInputData
216 {
217 target = RTSHelper.ToTargetData<IEntity>(upgradedInstance),
218 playerCommand = false
219 });
220 }
221 else if (sourceInstance.IsUnit())
222 {
223 IUnit unitPrefab = upgradeElement.target as IUnit;
224
225 // Create upgraded instance of the unit
226 upgradedInstance = unitMgr.CreateUnitLocal(
227 unitPrefab,
228 sourceInstance.transform.position,
229 sourceInstance.transform.rotation,
230 new InitUnitParameters
231 {
232 free = sourceInstance.IsFree,
233 factionID = factionID,
234
235 setInitialHealth = true,
236 initialHealth = (int)(healthRatio * upgradeElement.target.gameObject.GetComponent<IEntityHealth>().MaxHealth),
237
238 rallypoint = unitPrefab.SpawnRallypoint,
239 gotoPosition = sourceInstance.transform.position,
240
241 giveInitResources = true,
242 })[0];
243 }
244 else
245 {
246 logger.LogError($"[EntityUpgradeManager] Upgrading 'resource' instances is currently not allowed!");
247 }
248
249 foreach (EntityUpgradeComponentMatcherElement matcherElement in entityComponentMatcher)
250 if (sourceInstance.EntityComponents.TryGetValue(matcherElement.sourceComponentCode, out IEntityComponent sourceComponent)
251 && upgradedInstance.EntityComponents.TryGetValue(matcherElement.targetComponentCode, out IEntityComponent targetComponent))
252 targetComponent.HandleComponentUpgrade(sourceComponent);
253
254 globalEvent.RaiseEntityInstanceUpgradedGlobal(
255 sourceInstance,
256 new UpgradeEventArgs<IEntity>(upgradeElement, factionID, upgradedInstance));
257
258 effectObjPool.Spawn(upgradeEffect, upgradedInstance.transform);
259
260 // Destroy the upgraded instance
261 sourceInstance.Health.DestroyLocal(true, null);
262
263 if (wasSelected)
264 selectionMgr.Add(upgradedInstance, SelectionType.multiple, isLocalPlayerClickSelection: false);
265 }
