69
70 if(updateOptionButton)
71 updateOptionButton.interactable = RTSHelper.IsMasterInstance();
72
73 // As this is a post run game service, it is initialized after the ITimeModifier component so after the initial time modifier value is set.
74 // In case the game started with a build instance with time modifier game data and we are okay with overriding the options defined here by the ones in the game build data
75 if (overrideOptionsByGameData && gameMgr.CurrBuilder.IsValid())
76 options = timeModifier.Options.values;

84 // In case, the current local player is not allowed to update the time

87 if (!currOptionID.IsValidIndex(options))
88 currOptionID = 0;
