101 public void Trigger (IFactionEntity target, Vector3 targetPosition)
102 {
103 if (areaAttackEnabled == true)
104 TriggerArea(target.IsValid() ? target.transform.position : targetPosition, sourceFactionID: SourceAttackComp.Entity.FactionID);
105 // Apply damage directly
106 else if(target.IsValid())

110 private void TriggerArea (Vector3 center, int sourceFactionID)
111 {
112 gridSearch.Search(
113 center,
114 areaAttackData[areaAttackData.Length - 1].range,
115 -1,
116 SourceAttackComp.IsTargetValid,
117 // Set to true because we do not want to tie the target to the LOS parameters.
118 playerCommand: true,
119 out IReadOnlyList<IFactionEntity> targetsInRange);
120
121 for (int i = 0; i < targetsInRange.Count; i++)
122 {
123 IFactionEntity target = targetsInRange[i];
124 float distance = Vector3.Distance(target.transform.position, center);
125
126 for (int j = 0; j < areaAttackData.Length; j++)
127 {
128 // As long as the right range for this faction entity isn't found, move to the next one
129 if (distance > areaAttackData[j].range)
130 continue;
131
132 Deal(target, areaAttackData[j].data.Get(target));
133
134 // Area attack range found, move to the next target.
135 break;
136 }
137 }
138 }

141 #region Dealing Damage
142 private void Deal (IFactionEntity target, int value)
143 {
144 if (enabled == false || !target.IsValid()) // Can't deal damage then stop here
145 return;
146
147 foreach (FactionEntityDependantHitEffectData hitEffectData in hitEffects)
148 if (hitEffectData.picker.IsValidTarget(target))
149 {
150 effectObjPool.Spawn(
151 hitEffectData.effect.Output,
152 new EffectObjectSpawnInput(
153 parent: null,
154
155 useLocalTransform: false,
156 spawnPosition: target.transform.position,
157 spawnRotation: new RotationData(
158 hitEffectData.faceSource ? RotationType.lookAtPosition : RotationType.free,
159 SourceAttackComp.Entity.transform.position,
160 fixYRotation: true)));
161
162 audioMgr.PlaySFX(target?.AudioSourceComponent,
163 hitEffectData.audio.Fetch(),
164 false);
165 break;
166 }
167
168 if (dotEnabled == true)
169 target.Health.AddDamageOverTime(dotData, value, SourceAttackComp.Entity);
170 else
171 target.Health.Add(new HealthUpdateArgs(-value, SourceAttackComp.Entity));
172
173 // If this is a new target to deal damage to and the damage dealt is only logged for a single target
174 if (resetDamageDealt && target != LastTarget)
175 DamageDealt = 0;
176
177 DamageDealt += value;
178
179 LastTarget = target;
180
181 RaiseAttackDamageDealt(new HealthUpdateArgs(value, target));
182 damageDealtEvent.Invoke();
183 }
184 #endregion
