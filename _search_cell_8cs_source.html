168 #region Tracking Moving Entities
169 WaitForSeconds trackingUnitWaitFor = new WaitForSeconds(0.1f);
174 private IEnumerator EntityPositionCheck()
175 {
176 while (true)
177 {
178 yield return trackingUnitWaitFor;
179
180 int i = 0;
181 while (i < movingEntities.Count)
182 {
183 if (!movingEntities[i].IsValid())
184 {
185 movingEntities.RemoveAt(i);
186 continue;
187 }
188
189 if (!IsIn(movingEntities[i].transform.position))
190 {
191 IEntity nextEntity = movingEntities[i];
192
193 // Find a new cell for the unit
194
195 if (gridSearch.TryGetSearchCell(nextEntity.transform.position, out SearchCell newCell) != ErrorMessage.none)
196 {
197 logger.LogError($"[SearchCell] Unable to find a new search cell for unit of code {nextEntity.Code} at position {nextEntity.transform.position}!");
198 continue;
199 }
200
201 newCell.Add(nextEntity);
202
203 Remove(nextEntity);
204
205 continue;
206 }
207
208 i++;
209 }
210 }
