212 private void AddFactionEntity(IFactionEntity factionEntity)
213 {
214 if (factionEntity.IsDummy)
215 return;
216
217 factionEntities.Add(factionEntity);
218 if(!factionEntityCodeDic.TryGetValue(factionEntity.Code, out List<IFactionEntity> entityTypeList))
219 {
220 entityTypeList = new List<IFactionEntity>();
221 factionEntityCodeDic.Add(factionEntity.Code, entityTypeList);
222 }
223 entityTypeList.Add(factionEntity);
224
225 if (factionEntity.IsMainEntity)
226 mainEntities.Add(factionEntity);
227
228 if (factionEntity.DropOffTarget.IsValid())
229 dropOffTargets.Add(factionEntity);
230
231 UpdateLimit(factionEntity.Code, factionEntity.Category, increment: true);
232 }
233
234 private void RemoveFactionEntity (IFactionEntity factionEntity)
235 {
236 if (factionEntity.IsDummy)
237 return;
238
239 factionEntities.Remove(factionEntity);

241
242 if (factionEntity.IsMainEntity)
243 mainEntities.Remove(factionEntity);
244
245 if (factionEntity.DropOffTarget.IsValid())
246 dropOffTargets.Remove(factionEntity);
247
248 UpdateLimit(factionEntity.Code, factionEntity.Category, increment:false);
249
250 // Check if the faction doesn't have any buildings/units anymore and trigger the faction defeat in that case
251 CheckFactionDefeat();
252 }

312 #region Faction Entity Code/Category Amount Updating
313 private void AddFactionEntityAmount(IFactionEntity factionEntity)
314 {
315 if (factionEntity.IsDummy
316 || !factionEntity.CanLaunchTask)
317 return;
318
319 foreach (string category in factionEntity.Category)
320 {
321 if (!factionEntityCategoryToAmount.ContainsKey(category))
322 factionEntityCategoryToAmount.Add(category, 0);
323
324 factionEntityCategoryToAmount[category] += 1;
325 }
326 if (!factionEntityToAmount.ContainsKey(factionEntity.Code))
327 factionEntityToAmount.Add(factionEntity.Code, 0);
328 factionEntityToAmount[factionEntity.Code] += 1;
329
330 }
331
332 private void RemoveFactionEntityAmount(IFactionEntity factionEntity)
333 {
334 if (factionEntity.IsDummy
335 || (!factionEntity.CanLaunchTask && !factionEntity.Health.IsDead))
336 return;
337
338
339 foreach (string category in factionEntity.Category)
340 factionEntityCategoryToAmount[category] = factionEntityCategoryToAmount[category] - 1;
341
342 factionEntityToAmount[factionEntity.Code] = factionEntityToAmount[factionEntity.Code] - 1;
343 }
