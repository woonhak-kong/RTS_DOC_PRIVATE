105 private IMinimapIcon Spawn(IEntity source)
106 {
107 if (activeIcons.ContainsKey(source.Key))
108 return activeIcons[source.Key];
109
110 IMinimapIcon nextIcon = base.Spawn(prefab.Output);
111
112 if (!nextIcon.IsValid())
113 return null;
114
115 MinimapIconSpawnInput input = new MinimapIconSpawnInput(
116 sourceEntity: source,
117 height,
118 source.MinimapIconHandler.IsValid()
119 ? source.MinimapIconHandler.Data
120 : new IEntityMinimapIconData { changeTransparency = false, changeDarkness = false },
121 prefab.Output.transform.rotation);
122
123 nextIcon.OnSpawn(input);
124 activeIcons.Add(source.Key, nextIcon);
125
126 return nextIcon;
127 }
128
129 private void Despawn(IEntity source)
130 {
131 if (activeIcons.TryGetValue(source.Key, out IMinimapIcon icon))
132 {
133 base.Despawn(icon);
134
135 activeIcons.Remove(source.Key);
136 }
137 }
