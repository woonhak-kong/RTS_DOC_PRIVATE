121 #region Generating/Updating Mesh Data
122 private void GenerateMeshData()
123 {
124 int cellSize = placerMgr.GridHandler.CellSize;
125
126 positions = new List<Vector3>();
127
128 for (int x = terrainMgr.HeightCacheLowerLeftCorner.x; x < terrainMgr.HeightCacheUpperRightCorner.x; x += cellSize)
129 for (int y = terrainMgr.HeightCacheLowerLeftCorner.y; y < terrainMgr.HeightCacheUpperRightCorner.y; y += cellSize)
130 {
131 Vector3 position = new Vector3(x, 0.0f, y);
132 if (!terrainMgr.SampleHeight(position, placerMgr.PlacableTerrainAreas, out float height))
133 continue;
134
135 height += placerMgr.BuildingPositionYOffset;
136 position.y = height;
137
138 positions.Add(position);
139 }
140 }

142 private void GenerateGridMesh()
143 {
144 var mesh = new Mesh();
145
146 verticies = new List<Vector3>();
147
148 int cellSize = placerMgr.GridHandler.CellSize;
149
150 freeIndices = new List<int>();
151 occupiedIndices = new List<int>();
152
153 for (int i = 0; i < positions.Count; i++)
154 {
155 Vector3 position = positions[i];
156
157 verticies.Add(new Vector3(position.x, position.y, position.z + cellSize));
158 verticies.Add(new Vector3(position.x + cellSize, position.y, position.z + cellSize));
159 verticies.Add(new Vector3(position.x, position.y, position.z));
160 verticies.Add(new Vector3(position.x + cellSize, position.y, position.z));
161
162 var nextIndices = placerMgr.GridHandler.IsOccupied(position)
163 ? occupiedIndices
164 : freeIndices;
165
166 nextIndices.Add(4 * i + 0);
167 nextIndices.Add(4 * i + 1);
168 nextIndices.Add(4 * i + 2);
169 nextIndices.Add(4 * i + 2);
170 nextIndices.Add(4 * i + 1);
171 nextIndices.Add(4 * i + 3);
172 }
173
174 mesh.subMeshCount = 2;
175 mesh.vertices = verticies.ToArray();
176
177 mesh.SetIndices(occupiedIndices.ToArray(), MeshTopology.Lines, 0);
178 mesh.SetIndices(freeIndices.ToArray(), MeshTopology.Lines, 1);
179
180 meshFilter.mesh = mesh;
181
182 meshRenderer.materials = new Material[2];
183
184 meshRenderer.materials[0] = occupiedMaterial.IsValid() ? occupiedMaterial : new Material(Shader.Find("Sprites/Default"));
185 meshRenderer.materials[0].color = occupiedColor;
186
187 meshRenderer.materials[1].color = freeColor;
188 meshRenderer.materials[1] = freeMaterial.IsValid() ? freeMaterial : new Material(Shader.Find("Sprites/Default"));
189 }
190 #endregion
