144 private void ShowSquad(IUnitSquad squad)
145 {
146 //either valid selected entity or one that is not currently being displayed in the panel
147 if (!squad.IsValid()
148 || squad.CurrentCount == 0)
149 return;

151 IEntity entity = squad.Units[0];
152 panel.SetActive(true);
153
154 if (nameUIText && textDisplayer.EntityNameToText(entity, out string entityName))
155 nameUIText.text = $"{entityName} {squad.CurrentCount}/{squad.SpawnCount}"; //display that the entity belongs to a faction or free one?
156
157 if(descriptionUIText && textDisplayer.EntityDescriptionToText(entity, out string entityDescription))
158 descriptionUIText.text = entityDescription;
159
160 if(icon)
161 icon.sprite = entity.Icon;
162
163 if(squad == currSquad)
164 return;
165
166 squad.SquadUpdated += HandleSquadUpdated;
167
168 ShowSquadHealthUI(squad);
169 squad.SquadHealthUpdated += HandleSquadHealthUpdated;
170 squad.SquadMaxHealthUpdated += HandleSquadHealthUpdated;
171
172 //ShowDropOffResources(entity);
173
174 currEntity = entity;
175 currSquad = squad;

309 #region Handling Single Entity Drop Off Resources
310 private void ShowDropOffResources(IEntity entity)
311 {
312 if (!dropOffResourcePanelEnabled
313 || !dropOffResourcePanel.entityPicker.IsValidTarget(entity))
314 return;
315
316 // Resource collector handling
317 if(entity.IsUnit())
318 {
319 IDropOffSource dropOffComp = (entity as IUnit).DropOffSource;
320 if(dropOffComp.IsValid())
321 {
322 foreach(var elem in dropOffComp.CollectedResources)
323 {
324 if(dropOffResourceTasks.TryGetValue(elem.Key, out DropOffResourceTaskUI resourceTaskUI))
325 {
326 resourceTaskUI.Reload(new DropOffResourceTaskUIAttributes
327 {
328 dropOffSource = dropOffComp,
329 resourceType = elem.Key,
330 maxCapacityColor = dropOffResourcePanel.maxCapacityColor
331 }) ;
332 }
333 }
334 }
335 }
336 }
