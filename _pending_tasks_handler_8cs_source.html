126 private bool TryCompleteCurrent()
127 {
128 RaisePendingTaskStateUpdated(new PendingTaskEventArgs(
129 data: queue[0],
130 pendingQueueID: 0,
131 state: PendingTaskState.preCompleted
132 ));
133 queue[0].sourceComponent.OnPendingTaskPreComplete(queue[0]);
134
135 ErrorMessage launchError = queue[0].sourceTaskInput.CanComplete();
136 if(launchError == ErrorMessage.none)
137 {
138 CompleteCurrent();
139 return true;
140 }
141
142 // Cancel task is it belongs to a NPC faction
143 if(RTSHelper.IsNPCFaction(Entity))
144 CancelByQueueID(0);
145
146 else if (!isLocked)
147 {
148 if (Entity.IsLocalPlayerFaction())
149 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
150 {
151 message = launchError,
152
153 source = Entity
154 });
155
156 globalEvent.RaiseEntityComponentPendingTaskUIReloadRequestGlobal(Entity);
157
158 isLocked = true;
159 }
160
161 return false;
162 }

164 public void CompleteCurrent()
165 {
166 if (queue.Count == 0)
167 return;
168
169 PendingTask completedTask = queue[0];
170 queue.RemoveAt(0);
171
172 completedTask.sourceTaskInput.OnComplete();
173 completedTask.sourceComponent.OnPendingTaskCompleted(completedTask);
174
175 RaisePendingTaskStateUpdated(new PendingTaskEventArgs(
176 data: completedTask,
177 pendingQueueID: -1,
178 state: PendingTaskState.completed
179 ));
180
181 StartNext();
182
183 globalEvent.RaiseEntityComponentPendingTaskUIReloadRequestGlobal(Entity);
184 }

194 public void CancelByQueueID(int pendingTaskIndex)
195 {
196 if (pendingTaskIndex < 0 || pendingTaskIndex >= QueueCount)
197 return;
198
199 CancelInternal(queue[pendingTaskIndex]);
200 queue.RemoveAt(pendingTaskIndex);
201
202 globalEvent.RaiseEntityComponentPendingTaskUIReloadRequestGlobal(Entity);
203
204 if (pendingTaskIndex == 0)
205 StartNext();
206 }
207
208 public void CancelBySourceID(IPendingTaskEntityComponent sourceComponnet, int sourceID)
209 {
210 //in case the RemoveAll removes the first pending task in the queue, the queue must start the next one in case there is one.
211 PendingTask lastFirst = First;
212
213 queue.RemoveAll(pendingTask =>
214 {
215 if (pendingTask.sourceComponent == sourceComponnet && pendingTask.sourceTaskInput.ID == sourceID)
216 {
217 CancelInternal(pendingTask);
218 return true;
219 }
220 return false;
221 });
222
223 globalEvent.RaiseEntityComponentPendingTaskUIReloadRequestGlobal(Entity);
224
225 PendingTask newFirst = First;
226 if (!lastFirst.Equals(newFirst))
227 StartNext();
228 }
229
230 private void CancelInternal(PendingTask task)
231 {
232 task.sourceTaskInput.OnCancel();
233 task.sourceComponent.OnPendingTaskCancelled(task);
234
235 RaisePendingTaskStateUpdated(new PendingTaskEventArgs(
236 data: task,
237 pendingQueueID: -1,
238 state: PendingTaskState.completed
239 ));
240 }

243 #region Handling UI
244 public bool OnPendingTaskUIRequest(out IReadOnlyList<EntityComponentPendingTaskUIAttributes> taskUIAttributes)
245 {
246 taskUIAttributes = null;
247
248 if (!Entity.CanLaunchTask
249 || !RTSHelper.IsLocalPlayerFaction(Entity))
250 return false;
251
252 taskUIAttributesCache.Clear();
253 for (int i = 0; i < Queue.Count; i++)
254 {
255 PendingTask pendingTask = Queue[i];
256
257 if (!pendingTask.sourceComponent.IsActive)
258 continue;
259
260 taskUIAttributesCache.Add(
261 new EntityComponentPendingTaskUIAttributes
262 {
263 data = pendingTask.sourceTaskInput.Data,
264
265 pendingData = new EntityComponentPendingTaskUIData
266 {
267 queueIndex = i,
268 handler = this
269 },
270
271 locked = pendingTask.sourceTaskInput.CanComplete() != ErrorMessage.none,
272 lockedData = pendingTask.sourceTaskInput.MissingRequirementData,
273 });
274 }
275
276 taskUIAttributes = taskUIAttributesCache;
277
278 return true;
279 }
