66 public void Init(FactionSlotData data, int ID, IGameManager gameMgr)
67 {
68 this.gameMgr = gameMgr;
69 this.logger = this.gameMgr.GetService<IGameLoggingService>();
70 this.mainCameraController = gameMgr.GetService<IMainCameraController>();
71
72 if (!logger.RequireTrue(State == FactionSlotState.inactive,
73 $"[FactionSlot] Slot state must be set to inactive in order to initialize it, current state is '{State}'!",
74 source: gameMgr))
75 return;
76
77 this.data = data;
78 this.ID = ID;
79 if(data.forceID && data.forcedID != this.ID)
80 {
81 logger.LogError($"[FactionSlot] faction slot was supposed to initialize with forced ID {data.forcedID} but was assigned ID {this.ID} instead!");
82 return;
83 }
84
85 // Check for valid input
86 if (logger.RequireValid(initialCamLookAtPosition,
87 $"[FactionSlot - ID: {ID}] The 'Initial Camera Look At Position' field has not been assigned and therefore the initial camera position is set to (0,0,0)",
88 source: gameMgr,
89 type: LoggingType.warning))
90 FactionSpawnPosition = initialCamLookAtPosition.position;
91 else
92 FactionSpawnPosition = Vector3.zero;
93
94 // Check for validity of assigned initial faction entities
95 foreach (IFactionEntity instance in initialFactionEntities.GetAll())
96 if (!logger.RequireValid(instance,
97 $"[FactionSlot - ID: {ID}] 'Initial Faction Entities' has some invalid elements assigned!"))
98 return;
99 initialFactionEntitiesInitialized = false;
100
101 UpdateState(FactionSlotState.active);
102 }

104 private void InitNPC()
105 {
106 if (!logger.RequireValid(data.npcType,
107 $"[FactionSlot - ID: {ID}] No NPC type has been assigned! Faction will be initiated as dummy faction.",
108 type: LoggingType.warning))
109 return;
110
111 INPCManager npcMgrPrefab = data.npcType.GetNPCManagerPrefab(data.type);
112
113 if (!npcMgrPrefab.IsValid())
114 {
115 logger.LogWarning($"[FactionSlot - ID: {ID}] No NPC manager defined for the faction slot's NPC type {data.npcType.Key} and faction type '{data.type.GetKey()}'. Faction will be initiated as dummy faction.", source: gameMgr);
116 return;
117 }
118
119 // We only create NPC managers for the master instance
120 // In case of a singleplayer game, the master instance is the only available instance
121 // In case of a multiplayer game, the master instance is where the server is running (either on the host machine or headless server)
122 // This guarantees that there is only one instance of the NPC manager in one of the machines in the multiplayer game (always where the server is).
123 if (RTSHelper.IsMasterInstance())
124 {
125 CurrentNPCMgr = UnityEngine.Object.Instantiate(npcMgrPrefab.gameObject).GetComponent<INPCManager>();
126 CurrentNPCMgr.Init(data.npcType, gameMgr, FactionMgr);
127 }

130 public void InitDefaultFactionEntities ()
131 {
132 if (!logger.RequireTrue(!initialFactionEntitiesInitialized,
133 $"[FactionSlot - ID: {ID}] The initial faction entities have been already initialized. Unable to initialize them again!"))
134 return;
135
136 IReadOnlyList<IFactionEntity> usedEntities = initialFactionEntities.GetFiltered(data.type, out IReadOnlyList<IFactionEntity> unusedEntities);
137
138 if (gameMgr.ClearDefaultEntities)
139 foreach (IFactionEntity instance in usedEntities)
140 UnityEngine.Object.DestroyImmediate(instance.gameObject);
141 else
142 RTSHelper.InitFactionEntities(usedEntities, ID);
143
144 // Destroy the unusued entities (ones that do not fit the current faction's type)
145 foreach (IFactionEntity instance in unusedEntities)
146 UnityEngine.Object.DestroyImmediate(instance.gameObject);
147
148 if (Data.isLocalPlayer)
149 mainCameraController.PanningHandler.LookAt(FactionSpawnPosition, smooth: false);
150 }

163 }

167 #region Handling Faction Slot State/Role
168 public void UpdateState (FactionSlotState newState)
169 {
170 if (State == newState)
171 return;
172
173 switch(State)
174 {
175 case FactionSlotState.inactive:
176 if (!logger.RequireTrue(newState == FactionSlotState.active,
177 $"[FactionSlot - ID: {ID}] Unable to update the faction slot state from '{this.State}' to '{newState}'!."))
178 return;
179
180 this.FactionMgr = new FactionManager();
181 FactionMgr.Init(gameMgr, this);
182
183 if (data.role == FactionSlotRole.npc)
184 InitNPC();
185
186 break;
187
188 case FactionSlotState.active:
189
190 if (!logger.RequireTrue(newState == FactionSlotState.eliminated,
191 $"[FactionSlot - ID: {ID}] Unable to update the faction slot state from '{this.State}' to '{newState}'!."))
192 return;
193
194 // Disable NPC
195 if (CurrentNPCMgr.IsValid())
196 GameObject.Destroy(CurrentNPCMgr.gameObject);
197
198 break;
199
200 default:
201
202 logger.LogError($"[FactionSlot - ID: {ID}] Unable to update the faction slot state from '{this.State}' to '{newState}'!.");
203
204 return;
205 }
206
207 this.State = newState;
208 RaiseFactionSlotStateUpdated();
209 }
