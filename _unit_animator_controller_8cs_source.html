130 public void SetState(AnimatorState newState)
131 {
132 if (LockState == true)
133 return;
134
135 if (CurrState == AnimatorState.dead
136 // If the damage animation is active, only allow to change the animation if the next one is a death animation.
137 || (CurrState == AnimatorState.takeDamage && newState != AnimatorState.dead))
138 return;
139
140 CurrState = newState;
141
142 animator.SetBool(UnitAnimator.Parameters[AnimatorState.takeDamage], CurrState == AnimatorState.takeDamage);
143
144 // Stop the idle animation in case take damage animation is played since the take damage animation is broken by the idle anim
145 animator.SetBool(UnitAnimator.Parameters[AnimatorState.idle], CurrState == AnimatorState.idle);
146
147 // If the new animator state is the taking damage one then do not disable the rest of animations since as soon as the take damage animation is disabled, we want to get back to the last active state
148 if (CurrState == AnimatorState.takeDamage)
149 {
150 animator.SetBool(UnitAnimator.Parameters[AnimatorState.inProgress], false);
151 return;
152 }
153
154 foreach (AnimatorState state in UnitAnimator.States)
155 if (state != AnimatorState.movingState)
156 animator.SetBool(UnitAnimator.Parameters[state], state == CurrState);
157
158 animator.SetBool(UnitAnimator.Parameters[AnimatorState.movingState], CurrState == AnimatorState.moving);
159 }

169 #region Updating Animator Override Controller
170 public void SetOverrideController(AnimatorOverrideController newOverrideController)
171 {
172 // Only if the unit is not in its dead animation state do we reset the override controller
173 // And since all parameters reset when the unit is dead and the unit is locked in its death state
174 // Reseting the controller makes it start from its "entry state" back to "idle" state, this makes the unit leave its death state while still marked as dead in the currAnimatorState
175 if (!newOverrideController.IsValid()
176 || CurrState == AnimatorState.dead)
177 return;
178
179 this.currController = newOverrideController;
180 animator.runtimeAnimatorController = newOverrideController;
181
182 // Since changing the override controller resets all parameters, we need to re-set the current animator state
183 SetState(CurrState);
184 }
