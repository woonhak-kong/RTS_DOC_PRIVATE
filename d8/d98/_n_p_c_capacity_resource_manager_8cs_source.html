87 private void HandleFirstBuildingCenterInitialized(INPCBuildingCreator sender, EventArgs args)
88 {
89 // Activate the faction entities that influence the capacity resources
90 // Only activate them at this point because if the NPC is using building centers to regulate building creation then the centers have to be initialized first.
91 TargetCapacityResource = capacityResource.GetFiltered(gameMgr.GetFactionSlot(factionMgr.FactionID).Data.type);
92 if (!logger.RequireValid(TargetCapacityResource,
93 $"[{GetType().Name} - {factionMgr.FactionID}] No capacity resource that matches the faction type has been found, disabling component!",
94 source: this,
95 LoggingType.warning))
96 {
97 IsActive = false;
98 return;
99 }
100
101 if (!logger.RequireTrue(TargetCapacityResource.HasCapacity,
102 $"[{GetType().Name} - {factionMgr.FactionID}] Resource '{TargetCapacityResource.Key}' Must be a capacity resource to be handled by this component!"))
103 return;
104
105 // Capacity resource events
106 capacityResourceHandler = resourceMgr.FactionResources[factionMgr.FactionID].ResourceHandlers[TargetCapacityResource];
107 capacityResourceHandler.FactionResourceAmountUpdated += HandleFactionResourceAmountUpdated;
108
109 // Add the faction entities that can influence the capacity resource managed by this component
110 for (int i = 0; i < factionEntities.Count; i++)
111 {
112 IFactionEntity nextEntity = factionEntities[i].IsValid() ? factionEntities[i].GetComponent<IFactionEntity>() : null;
113 if (!logger.RequireValid(nextEntity,
114 $"[{GetType().Name} - {factionMgr.FactionID}] 'Faction Entities' list has some unassigned or invalid elements.",
115 source: this))
116 continue;
117
118 bool hasTargetCapacityResource = false;
119 foreach(ResourceInput resourceInput in nextEntity.InitResources)
120 {
121 if (resourceInput.type == TargetCapacityResource)
122 {
123 hasTargetCapacityResource = true;
124 break;
125 }
126 }
127 if (!hasTargetCapacityResource)
128 continue;
129
130 if (nextEntity.IsUnit())
131 {
132 NPCUnitRegulator nextUnitRegulator;
133 if ((nextUnitRegulator = npcUnitCreator.ActivateUnitRegulator(
134 nextEntity as IUnit)).IsValid())
135 unitsMonitor.AddCode(nextUnitRegulator.Prefab.Code);
136 }
137 else if (nextEntity.IsBuilding())
138 {
139 NPCBuildingRegulator nextBuildingRegulator;
140 if ((nextBuildingRegulator = npcBuildingCreator.ActivateBuildingRegulator(
141 nextEntity as IBuilding,
142 npcBuildingCreator.FirstBuildingCenter)).IsValid())
143 buildingsMonitor.AddCode(nextBuildingRegulator.Prefab.Code);
144 }
145 }
146
147 if (unitsMonitor.Count <= 0 && buildingsMonitor.Count <= 0)
148 logger.LogWarning($"{GetType().Name} - {factionMgr.FactionID}] No regulators are found for faction entities that can increase the capacity resource when created.", source: this);
149 }
150
151 protected override void OnDestroyed()
152 {
153 npcBuildingCreator.FirstBuildingCenterInitialized -= HandleFirstBuildingCenterInitialized;
154
155 if (TargetCapacityResource.IsValid())
156 {
157 IFactionResourceHandler capacityResourceHandler = resourceMgr.FactionResources[factionMgr.FactionID].ResourceHandlers[TargetCapacityResource];
158 capacityResourceHandler.FactionResourceAmountUpdated -= HandleFactionResourceAmountUpdated;
159 }
160
161 globalEvent.BuildingPlacementStartGlobal -= HandleBuildingPlacementStartGlobal;
162 globalEvent.BuildingPlacementStopGlobal -= HandleBuildingPlacementStopOrBuiltGlobal;
163 globalEvent.BuildingBuiltGlobal -= HandleBuildingPlacementStopOrBuiltGlobal;
164
165 buildingsMonitor.Disable();
166 unitsMonitor.Disable();
167 }

171 private void HandleFactionResourceAmountUpdated(IFactionResourceHandler capacityResourceHandler, ResourceUpdateEventArgs args)
172 {
173 // If the amount of the capacity resources is less than required and has not reached the target capacity value
174 // then activate this component to make the NPC component attempt to raise the capacity amount.
175 if (autoCreate
176 && capacityResourceHandler.FreeAmount + pendingCapacity < minFreeAmount
177 && capacityResourceHandler.Capacity < targetCapacityRange.RandomValue)
178 IsActive = true;
179 }
180 #endregion

182 #region Handling Events: IBuilding
183 private void HandleBuildingPlacementStartGlobal(IBuilding building, EventArgs e)
184 {
185 if (building.FactionID != factionMgr.FactionID)
186 return;

188 // Handle the potential of increasing population capactiy when the building is later placed by tracking its population resource increase
189 pendingCapacity += GetResourceCapacityValue(building.InitResources);
190 }

192 //when the building is built (its init resources will be added to faction
193 private void HandleBuildingPlacementStopOrBuiltGlobal(IBuilding building, EventArgs e)
194 {
195 if (building.FactionID != factionMgr.FactionID)
196 return;
197
198 pendingCapacity -= GetResourceCapacityValue(building.InitResources);
199 }

203 protected override void OnActiveUpdate()
204 {
205 base.OnActiveUpdate();
206
207 timer.ModifiedDecrease();
208
209 if (timer.CurrValue <= 0.0f)
210 {
211 timer = new TimeModifiedTimer(reloadRange.RandomValue);
212
213 if (IsTargetCapacityReached
214 || OnIncreaseCapacityRequest())
215 IsActive = false;
216 }
217 }
218
219 public bool OnIncreaseCapacityRequest()
220 {
221 // Prioritize placing buildings to increase capacity resource.
222
223 if (buildingsMonitor.Count > 0
224 && npcBuildingCreator.OnCreateBuildingRequest(buildingsMonitor.RandomCode))
225 return true;
226
227 if (unitsMonitor.Count > 0)
228 return npcUnitCreator.OnCreateUnitRequest(unitsMonitor.RandomCode, requestedAmount: 1, out _);
229
230 return false;
231 }
