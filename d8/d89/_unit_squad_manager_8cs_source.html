88 #region Creating Unit Squads
89 private bool JoinToNewSquadWithID(IReadOnlyList<IUnit> units, int nextID, int spawnCount)
90 {
91 if(!units.IsValid() || units.Count == 0)
92 {
93 logger.LogError($"[{GetType().Name}] Inputs units list to form a squad is not valid or is empty!");
94 return false;
95 }
96
97 string unitCode = units[0].Code;
98 int factionID = units[0].FactionID;
99 foreach (IUnit unit in units)
100 {
101 if (unitCode != unit.Code)
102 {
103 logger.LogError($"[{GetType().Name}] All units to join to new squad must have the same type/code!");
104 return false;
105 }
106 else if (factionID != unit.FactionID)
107 {
108 logger.LogError($"[{GetType().Name}] All units to join to new squad must belong to the same faction!");
109 return false;
110 }
111 else if (unit.Squad.IsValid())
112 {
113 logger.LogError($"[{GetType().Name}] One or more units already belong to an existing squad!");
114 return false;
115 }
116
117 if(IsInSquad(unit))
118 {
119 logger.LogError($"[{GetType().Name}] One or more input units is already in a squad! Can not add to new squad!");
120 return false;
121 }
122 }
123
124 if(squadTracker.ContainsKey(nextID))
125 {
126 logger.LogError($"[{GetType().Name}] Squad ID: {nextID} has been already used!");
127 return false;
128 }
129
130 UnitSquad nextSquad = new UnitSquad(gameMgr, nextID, unitCode, factionID, spawnCount);
131 squadTracker.Add(
132 nextID,
133 nextSquad);
134
135 foreach (IUnit unit in units)
136 {
137 if (!nextSquad.Add(unit))
138 {
139 logger.LogError($"[{GetType().Name}] Unable to add unit {unit} to new squad! Stopping and removing squad!");
140 nextSquad.Destroy();
141 return false;
142 }
143 unitToSquadDic.Add(unit, nextSquad);
144 unit.Health.EntityDead += HandleSquadUnitDead;
145 }
146
147 nextSquad.SquadDead += HandleSquadDead;
148
149 return true;
150 }
151
152 public bool JoinToNewSquad(IReadOnlyList<IUnit> units)
153 {
154 int spawnCount = units.Count;
155 units = units.Distinct().ToArray();
156
157 if(units.Count != spawnCount)
158 {
159 logger.LogError($"[{GetType().Name}] Unique unit instances in the input list does not match with the spawn count!");
160 return false;
161 }
162
163 if(JoinToNewSquadWithID(units, nextSquadID, spawnCount))
164 {
165 nextSquadID++;
166 return true;
167 }
168
169 return false;
170 }

225 public void Load(string data)
226 {
227 UnitSquadManagerSaveData saveData = saveFormatter.FromSaveFormat<UnitSquadManagerSaveData>(data);
228
229 foreach(var nextSquad in saveData.squads)
230 {
231 IUnit[] units = nextSquad.unitKeys
232 .Select(key =>
233 {
234 if (!inputMgr.TryGetEntityInstanceWithKey(key, out IEntity unit))
235 logger.LogError($"[{GetType().Name}] Unable to find unit with key: {key}!");
236
237 return unit as IUnit;
238 })
239 .ToArray();
240
241 if (JoinToNewSquadWithID(units, nextSquad.ID, nextSquad.spawnCount))
242 {
243
244 if (nextSquad.isSelected)
245 squadTracker[nextSquad.ID].Select();
246 }
247 }
248
249 nextSquadID = saveData.nextSquadID;
250 }
