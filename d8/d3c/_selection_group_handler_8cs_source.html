81 #region Adding/Removing
82 public bool Update(SelectionGroupAction action, bool requireKey = true)
83 {
84 if (requireKey && (!HasKey || !controls.Get(key)))
85 return false;
86
87 switch(action)
88 {
89 case SelectionGroupAction.assign:
90
91 current.Clear();
92
93 Add(selectionMgr.GetEntitiesList(allowedEntityTypes, exclusiveType: false, localPlayerFaction: true));
94 break;
95
96 case SelectionGroupAction.append:
97
98 Add(selectionMgr.GetEntitiesList(allowedEntityTypes, exclusiveType: false, localPlayerFaction: true));
99 break;
100
101 case SelectionGroupAction.substract:
102
103 Remove(selectionMgr.GetEntitiesList(allowedEntityTypes, exclusiveType: false, localPlayerFaction: true));
104 break;
105
106 case SelectionGroupAction.reset:
107 RemoveAll();
108 break;
109
110 default:
111
112 if (current.Count > 0)
113 {
114 selectionMgr.RemoveAll();
115 selectionMgr.Add(current);
116 }
117 break;
118 }
119
120 if(action != SelectionGroupAction.select)
121 RaiseGroupUpdated();
122
123 return true;
124 }

126 public bool Add(IEnumerable<IEntity> entities)
127 {
128 foreach (IEntity entity in entities)
129 {
130 if (current.Contains(entity)
131 || (maxAmountEnabled && current.Count == maxAmount)
132 || (forceSameType && current.Count > 0 && entity.Code != forcedEntityCode))
133 continue;
134
135 current.Add(entity);
136 entity.Health.EntityDead += HandleEntityDead;
137 entity.FactionUpdateComplete += HandleFactionUpdateComplete;
138
139 if (forceSameType && current.Count == 1)
140 forcedEntityCode = entity.Code;
141 }
142
143 return true;
144 }

236 SelectionGroupAction nextAction = SelectionGroupAction.select;
237
238 if(controls.Get(assignKey))
239 nextAction = SelectionGroupAction.assign;
240 else if (appendKey.IsValid() && controls.Get(appendKey))
241 nextAction = SelectionGroupAction.append;
242 else if (substractKey.IsValid() && controls.Get(substractKey))
243 nextAction = SelectionGroupAction.substract;
244 else if (resetKey.IsValid() && controls.Get(resetKey))
245 nextAction = SelectionGroupAction.reset;
246
247 for (int i = 0; i < groups.Length; i++)
248 {
249 if (groups[i].Update(nextAction))
