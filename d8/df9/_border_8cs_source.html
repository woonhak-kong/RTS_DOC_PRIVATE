174 private void AddResource (IResource resource)
175 {
176 if (resourcesInRange.Contains(resource)
177 || (!resource.IsSameFaction(Building) && resource.FactionID != -1)
178 || !IsInBorder(resource.transform.position))

184 // Only update the resource's "owner" faction ID if it is a resource building that already only belongs to one faction

277 public virtual bool IsBuildingAllowedInBorder(IBuilding building)
278 {
279 foreach(BuildingAmount ba in buildingLimits)

281 {
282 buildingTypeTracker.TryGetValue(building.Code, out int currValue);
