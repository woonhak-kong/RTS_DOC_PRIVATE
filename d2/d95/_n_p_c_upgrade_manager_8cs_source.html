95 #region Launching Upgrades
96 protected override void OnActiveUpdate()
97 {
98 if (!upgradeTimer.ModifiedDecrease())
99 return;
100
101 upgradeTimer.Reload(upgradeReloadRange);
102
103 // Assume that there are no upgrade tasks left to launch before going through the cached ones
104 IsActive = false;
105
106 foreach (IUpgradeLauncher upgradeLauncher in npcTracker.UpgradeLauncherTracker.Components)
107 {
108 if (upgradeLauncher.Tasks.Count <= 0)
109 continue;
110
111 IsActive = true;
112
113 for (int upgradeTaskID = 0; upgradeTaskID < upgradeLauncher.Tasks.Count; upgradeTaskID++)
114 OnUpgradeLaunchRequestInternal(upgradeLauncher, upgradeTaskID);
115 }
116 }
117
118 public bool OnUpgradeLaunchRequest(IUpgradeLauncher upgradeLauncher, int upgradeTaskID)
119 {
120 if (!upgradeOnDemand)
121 return false;
122
123 return OnUpgradeLaunchRequestInternal(upgradeLauncher, upgradeTaskID);
124 }
125
126 private bool OnUpgradeLaunchRequestInternal(IUpgradeLauncher upgradeLauncher, int upgradeTaskID)
127 {
128 if (!upgradeLauncher.IsValid()
129 || !upgradeTaskID.IsValidIndex(upgradeLauncher.Tasks)
130 || UnityEngine.Random.value > acceptanceRange.RandomValue)
131 return false;
132
133 ErrorMessage errorMessage = upgradeLauncher.LaunchTaskAction(upgradeTaskID, false);
134
135 switch (errorMessage)
136 {
137 case ErrorMessage.taskMissingResourceRequirements:
138 npcResourceMgr.OnIncreaseMissingResourceRequest(upgradeLauncher.Tasks[upgradeTaskID].RequiredResources);
139 break;
140 }
141
142 LogEvent($"Launching upgrade on {upgradeLauncher.Tasks[upgradeTaskID].Entity.Code}: Success? {errorMessage == ErrorMessage.none} - Error: {errorMessage}");
143
144 return errorMessage == ErrorMessage.none;
145 }
