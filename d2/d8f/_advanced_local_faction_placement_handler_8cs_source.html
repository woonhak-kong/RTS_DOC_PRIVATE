99 #region Handling Movement
100 private void MoveBuilding(bool forceMovementUpdate = false)
101 {
102 // Using a raycheck, we will make the current building follow the mouse position and stay on top of the terrain.
103 if (!Physics.Raycast(
104 mainCameraController.MainCamera.ScreenPointToRay(Input.mousePosition),
105 out RaycastHit hit,
106 Mathf.Infinity,
107 placerMgr.PlacableLayerMask))
108 return;
109
110 // Depending on the height of the terrain, we will place the building on it
111 Vector3 nextBuildingPos = hit.point;
112 // Make sure that the building position on the y axis stays inside the min and max height interval
113 nextBuildingPos.y += placerMgr.BuildingPositionYOffset;
114
115 if (forceMovementUpdate
116 || current.lastInstance.transform.position != nextBuildingPos)
117 {
118 // Segmentation mode inactive? Simple single building placement handling
119 if (!SegmentationActive)
120 {
121 if (IsGridPlacementActive)
122 {
123 placerMgr.GridHandler.TryGetCellPosition(nextBuildingPos, out nextBuildingPos);
124 }
125
126 current.firstInstance.PlacerComponent.OnPlacementUpdate(nextBuildingPos);
127 return;
128 }
129
130 // Segmentation mode active
131 OnSegmentationUpdate(finalPosition: nextBuildingPos);
132 }
133 }

136 #region Handling Rotation
137 private void RotateBuilding()
138 {
139 // Can not rotate buildings when segmentation mode is active
140 if (!canRotate
141 || SegmentationActive
142 || current.mainOptions.disableRotation)
143 return;
144
145 for (int i = 0; i < current.SegmentCount; i++)
146 {
147 IBuilding instance = current.instances[i];
148
149 float rotationMultiplier;
150 // For grid placement, only allow for 4 direction rotations
151 if (IsGridPlacementActive)
152 {
153 rotationMultiplier = (controls.GetDown(positiveRotationKey) ? 1.0f : (controls.GetDown(negativeRotationKey) ? -1.0f : 0.0f));
154 instance.transform.Rotate(new Vector3(0.0f, rotationMultiplier * 90.0f, 0.0f));
155 }
156 // Non grid placement allows for free rotation
157 else
158 {
159 Vector3 nextEulerAngles = instance.transform.rotation.eulerAngles;
160 // Only rotate if one of the keys is pressed down (check for direction) and rotate on the y axis only.
161 rotationMultiplier = (controls.Get(positiveRotationKey) ? 1.0f : (controls.Get(negativeRotationKey) ? -1.0f : 0.0f));
162 nextEulerAngles.y += rotationSpeed * rotationMultiplier;
163
164 instance.transform.rotation = Quaternion.Euler(nextEulerAngles);
165 }
166
167 if (rotationMultiplier != 0.0f)
168 instance.PlacerComponent.OnPlacementUpdate();
169 }
170 }
