155 public void UpdateResource(int factionID, ResourceInput resourceInput, bool add, out int restAmount)
156 {
157 restAmount = 0;
158
159 if (!IsResourceTypeValidInGame(resourceInput, factionID)
160 || resourceInput.nonConsumable)
161 return;
162
163 IFactionResourceHandler resourceHandler = FactionResources[factionID].ResourceHandlers[resourceInput.type];
164
165 ResourceTypeValue updateValue = add
166 ? resourceInput.value
167 : new ResourceTypeValue { amount = -resourceInput.value.amount, capacity = -resourceInput.value.capacity };
168
169 resourceHandler.UpdateAmount(updateValue, out restAmount);
170 }

210 public bool HasResources(ResourceInput resourceInput, int factionID)
211 {
212 if (!IsResourceTypeValidInGame(resourceInput, factionID))
213 return false;
214
215 var resourceHandler = FactionResources[factionID].ResourceHandlers[resourceInput.type];
216
217 return
218 // Make sure the amount value is available
219 (resourceHandler.Type.HasCapacity && resourceHandler.Type.CapacityType == ResourceCapacityType.slotBased
220 ? (resourceHandler.FreeAmount - resourceHandler.ReservedCapacity) >= resourceInput.value.amount
221 : resourceHandler.Amount - resourceHandler.ReservedAmount >= resourceInput.value.amount * FactionResources[factionID].ResourceNeedRatio)
222 // Make sure the capacity value is available
223 && resourceHandler.Capacity >= resourceInput.value.capacity;
224 }
