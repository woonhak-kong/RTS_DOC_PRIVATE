81 public bool CanSpawn(Vector3 spawnPosition)
82 {
83 return effectObjPool.ActiveDic.TryGetValue(prefab.Output.Code, out IEnumerable<IEffectObject> currActiveSet)
84 ? currActiveSet
85 .All(activeEffect => Vector3.Distance(activeEffect.GetComponent<RectTransform>().localPosition, spawnPosition) > minDistance)

92 public void Spawn(Vector3 targetPosition)
93 {
94 if (!logger.RequireTrue(minimapCameraController.WorldPointToLocalPointInMinimapCanvas(
95 targetPosition, out Vector3 spawnPosition, height),
96 $"[{GetType().Name}] Unable to find the target position of '{targetPosition}' for the new attack warning effect on the minimap canvas!"))
97 return;
98
99 if (!CanSpawn(spawnPosition))
100 return;
101
102 effectObjPool.Spawn(
103 prefab.Output,
104 new EffectObjectSpawnInput(
105 parent: minimapCameraController.MinimapCanvas.transform,
106
107 useLocalTransform: true,
108 spawnPosition: spawnPosition,
109 spawnRotation: prefab.Output.transform.localRotation));
110
111 audioMgr.PlaySFX(audioClip.Fetch(), false);
112
113 if (sendPlayerMessage)
114 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
115 {
116 message = ErrorMessage.factionUnderAttack,
117 });
118 }
