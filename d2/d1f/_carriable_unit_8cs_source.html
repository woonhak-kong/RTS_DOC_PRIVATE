168 private void HandleTargetCarrierUnitAdded(IUnitCarrier carrier, UnitCarrierEventArgs args)
169 {
170 if (args.Unit != unit)
171 return;
172
173 CurrCarrier = carrier;
174 CurrSlot = args.Slot;
175 CurrSlotID = args.SlotID;
176
177 if (CurrSlot.IsValid())
178 {
179 unit.MovementComponent.SetPosition(CurrSlot.position);
180 slotFollowHandler.SetTarget(CurrSlot, enableCallback: false);
181 }
182
183 CurrCarrier.UnitAdded -= HandleTargetCarrierUnitAdded;
184 CurrCarrier.UnitRemoved += HandleTargetCarrierUnitRemoved;
185 }
186
187 private void HandleTargetCarrierUnitRemoved(IUnitCarrier carrier, UnitCarrierEventArgs args)
188 {
189 if (carrier != CurrCarrier)
190 return;
191
192 // If this unit is confirmed to be added to its target carrier
193 if (args.Unit == unit)
194 {
195 Stop();
196
197 CurrCarrier.UnitRemoved -= HandleTargetCarrierUnitRemoved;
198
199 CurrCarrier = null;
200 CurrSlot = null;
201 CurrSlotID = -1;
202 slotFollowHandler.ResetTarget();
203 }
