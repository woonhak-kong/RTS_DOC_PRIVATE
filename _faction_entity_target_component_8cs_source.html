239 /*
240 * SetTargetLocal calls 'bool IsTargetValid(SetTargetInputData testInput, out ErrorMessage errorMessage)'
241 * by default, that method calls 'ErrorMessage IsTargetValid(SetTargetInputData testInput)'
242 * which then calls the 'ErrorMessage IsTargetValid(TargetData<IEntity> testTarget, bool playerCommand)' method

291 public virtual ErrorMessage SetTargetLocal(SetTargetInputData input)
292 {
293 if (!factionEntity.CanLaunchTask)
294 return ErrorMessage.taskSourceCanNotLaunch;
295
296 IsTargetValid(input, out ErrorMessage errorMsg);
297 if (errorMsg != ErrorMessage.none)
298 {
299 OnSetTargetError(input, errorMsg);
300 if (input.playerCommand && RTSHelper.IsLocalPlayerFaction(factionEntity))
301 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
302 {
303 message = errorMsg,
304
305 source = Entity,
306 target = input.target.instance
307 });
308
309 return errorMsg;
310 }
311
312 if (HasTarget)
313 Stop();
314
315 bool sameTarget = input.target.instance == Target.instance as IEntity && input.target.instance.IsValid();
316
317 // If this component requires the entity to be idle to run then set the entity to idle before assigning the new target
318 if(RequireIdleEntity)
319 factionEntity.SetIdle(sameTarget ? this : null);
320
321 OnTargetPreLocked(input.playerCommand, input.target, sameTarget);
322
323 TargetInputData = input;
324 Target = input.target;
325
326 if (input.playerCommand && Target.instance.IsValid() && factionEntity.IsLocalPlayerFaction())
327 selector.FlashSelection(Target.instance, factionEntity.IsFriendlyFaction(Target.instance));
328
329 OnTargetPostLocked(input, sameTarget);
330
331 RaiseTargetUpdated();
332
333 return ErrorMessage.none;
334 }
