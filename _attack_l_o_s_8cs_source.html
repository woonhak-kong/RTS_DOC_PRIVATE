59 public bool IsAngleBlocked (Vector3 sourcePosition, Quaternion sourceRotation, Vector3 targetPosition)
60 {
61 if (!enabled)
62 return true;
63
64 Vector3 lookAt = targetPosition - sourcePosition;
65
66 // Which axis to ignore when checking for LOS?
67 if (ignoreRotationX == true)
68 lookAt.x = 0.0f;
69 if (ignoreRotationY == true)
70 lookAt.y = 0.0f;
71 if (ignoreRotationZ == true)
72 lookAt.z = 0.0f;
73
74 // if the angle is below the allowed LOS Angle then the attacker is in line of sight of the target
75 return Vector3.Angle(sourceRotation * Vector3.forward, lookAt) >= angle;
76 }
