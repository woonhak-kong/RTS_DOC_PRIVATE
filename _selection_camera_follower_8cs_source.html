67 public void FollowNextEntity()
68 {
69 if (!isActive

71 || selectionMgr.Count == 0

75 // First follow order?

77 ResetTarget();
78
79 // Since a new entity will be followed, unsub to the last entity's deselection event
80 if(currFollowedEntity.IsValid())
81 currFollowedEntity.Selection.Deselected -= HandleCurrentFollowedEntityDeselected;
82
83 // Handling the index of the selected entities to follow
84 if (nextFollowIndex >= selectionMgr.Count)
85 nextFollowIndex = 0;
86
87 followedEntities = selectionMgr.GetEntitiesList(EntityType.all, exclusiveType: false, localPlayerFaction: false).ToList();
88
89 currFollowedEntity = followedEntities[nextFollowIndex];
90
91 // Follow the next entity
92 mainCameraController.PanningHandler.SetFollowTarget(currFollowedEntity.transform);
93
94 // Subscribe to the entity's deselection event because we want to stop following as soon as the entity is deselected
95 currFollowedEntity.Selection.Deselected += HandleCurrentFollowedEntityDeselected;
96
97 // If we can iterate through selected entities
98 if (iterate)
99 {
100 nextFollowIndex++;
101 if (nextFollowIndex >= followedEntities.Count)
102 nextFollowIndex = 0;
103 }
104 }
