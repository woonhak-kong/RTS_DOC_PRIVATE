123 public ErrorMessage LaunchAttackLocal(LaunchAttackData<IReadOnlyList<IEntity>> data)
124 {
125 if (!data.source.IsValid())
126 {
127 logger.LogError("[AttackManager] Some or all entities that are attempting to attack are invalid!");
128 return ErrorMessage.invalid;
129 }
130 else if (data.source.Count < 2)
131 return LaunchAttackLocal(
132 new LaunchAttackData<IEntity>
133 {
134 source = data.source[0],
135 targetEntity = data.targetEntity,
136 targetPosition = data.targetPosition,
137 playerCommand = data.playerCommand,
138 allowTerrainAttack = data.allowTerrainAttack,
139 ignoreLOS = data.ignoreLOS,
140 attackTypeCode = data.attackTypeCode,
141 isMoveAttackRequest = data.isMoveAttackRequest,
142 mode = data.mode,
143 });
144 else if (!data.targetEntity.IsValid() && !CanLaunchTerrainAttack(data))
145 return ErrorMessage.attackTerrainDisabled;
146
147 // Take out the attack entities which do not use a movement component, for those, a direct target set is done where the attack position is the current entity position.
148 nonMovableAttackers.Clear();
149
150 // We first start by handling the movable attackers
151
152 // Sort the attack entities based on their codes, we assume that units that share the same code (which is the defining property of an entity in the RTS Engine) are identical.
153 // And filter out any units that do not have an attack component.
154 var sortedAttackers = RTSHelper.SortEntitiesByCode(
155 data.source,
156 x => x.CanAttack)
157 .Values
158 .OrderBy(mvtSourceSet => mvtSourceSet[0].MovementComponent.IsValid() ? mvtSourceSet[0].MovementComponent.MovementPriority : -1)
159 .ToList();
160
161 // At least one attacker to get the attack order audio from.
162 IEntity refAttacker = null;
163
164 foreach (IReadOnlyList<IEntity> attackerSet in sortedAttackers)
165 {
166 // If the current unit type is unable to have the entity as the target, move to the next unit type list
167 if (attackerSet[0].FirstActiveAttackComponent.IsTargetValid(data.ToSetTargetInputData()) != ErrorMessage.none)
168 continue;
169
170 // Generate movement path destinations for the current list of identical unit types:
171 mvtMgr.GeneratePathDestination(
172 attackerSet,
173 data.targetPosition,
174 attackerSet[0].FirstActiveAttackComponent.Formation.MovementFormation,
175 // Enabke onProgressEnableTest because if the target is a unit instance, we want the attacker unit to pick its destination based on the expected stopping distance it is gonna have when it reaches its target range
176 attackerSet[0].FirstActiveAttackComponent.Formation.GetStoppingDistance(data.targetEntity, min: true, onProgressEnableTest: true),
177 new MovementSource { playerCommand = data.playerCommand, isMoveAttackRequest = data.isMoveAttackRequest},
178 ref pathDestinations,
179 condition: RTSHelper.IsAttackLOSBlocked);
180
181 // No valid path destinations generated? do not continue as there is nowhere to move to
182 if (pathDestinations.Count == 0)
183 continue;
184
185 // Index counter for the generated path destinations.
186 int destinationID = 0;
187
188 foreach (IEntity attacker in attackerSet)
189 {
190 if (!attacker.CanMove())
191 {
192 nonMovableAttackers.Add(attacker);
193 continue;
194 }
195
196 // Create the next target struct without specifying the attack position
197 TargetData<IFactionEntity> nextTarget = new TargetData<IFactionEntity>
198 {
199 instance = data.targetEntity,
200 opPosition = data.targetPosition,
201 };
202
203 pathDestinations.Sort((pos1, pos2) => (pos1 - attacker.transform.position).sqrMagnitude.CompareTo((pos2 - attacker.transform.position).sqrMagnitude));
204 destinationID = pathDestinations.FindIndex(pos => attacker.FirstActiveAttackComponent.IsTargetInRange(pos, nextTarget));
205 if (destinationID == -1)
206 {
207 // Only disallow to set the attack target if this is not a playerCommand
208 // If it is a player command, pick the closest path destination to the source attacker...
209 // ...so that it will be able to move towards that position and not attack
210 if (!data.playerCommand)
211 continue;
212
213 destinationID = 0;
214 }
215
216 nextTarget.position = pathDestinations[destinationID];
217
218 // If current unit is able to engage with its target using the computed path then move to the next path, if not, test the path on the next unit.
219 // The last argument of the SetTarget method is set to the playerCommand because we still want to move the units to computed attack position...
220 // ...even if it is out of the attack range because the player issued the attack/movement command.
221 ErrorMessage setTargetErorr = attacker.FirstActiveAttackComponent.SetTargetLocal(
222 new SetTargetInputData
223 {
224 target = nextTarget,
225
226 playerCommand = data.playerCommand,
227
228 isMoveAttackRequest = data.isMoveAttackRequest,
229 ignoreLOS = data.ignoreLOS,
230 });
231 if (setTargetErorr == ErrorMessage.none || setTargetErorr == ErrorMessage.attackMoveToTargetOnly)
232 {
233 // Assign the reference unit from which the attack order will be played.
234 if (!refAttacker.IsValid())
235 refAttacker = attacker;
236
237 pathDestinations.RemoveAt(destinationID);
238
239 // No more paths to test, stop moving units to attack.
240 if (destinationID >= pathDestinations.Count)
241 break;
242 }
243 }
244 }
245
246 // Finally handle setting targets for the non movable attackers
247 foreach (IEntity attacker in nonMovableAttackers)
248 {
249 // Assign the reference attacker from which the attack order will be played, if none has been assigned yet.
250 if (!refAttacker.IsValid())
251 refAttacker = attacker;
252
253 attacker.FirstActiveAttackComponent?.SetTargetLocal(
254 new SetTargetInputData
255 {
256 target = new TargetData<IFactionEntity>
257 {
258 instance = data.targetEntity,
259 opPosition = data.targetEntity.IsValid() ? data.targetEntity.transform.position : data.targetPosition,
260
261 position = attacker.transform.position
262 },
263
264 playerCommand = data.playerCommand,
265
266 isMoveAttackRequest = data.isMoveAttackRequest,
267 ignoreLOS = data.ignoreLOS
268 });
269 }
270
271 if (data.playerCommand && refAttacker.IsValid() && refAttacker.IsLocalPlayerFaction())
272 {
273 if (!data.targetEntity.IsValid())
274 effectObjPool.Spawn(TerrainAttackTargetEffect, data.targetPosition);
275
276 audioMgr.PlaySFX(refAttacker.FirstActiveAttackComponent.OrderAudio, false);
277 }
278
279 return ErrorMessage.none;
280 }

303 public ErrorMessage LaunchAttackLocal(LaunchAttackData<IEntity> data)
304 {
305 if (!data.source.IsValid())
306 {
307 logger.LogError("[AttackManager] Can not attack with an invalid entity!");
308 return ErrorMessage.invalid;
309 }
310 else if (!data.source.CanAttack)
311 return ErrorMessage.attackSourceDisabled;
312 else if (!data.targetEntity.IsValid() && !CanLaunchTerrainAttack(data))
313 return ErrorMessage.attackTerrainDisabled;
314
315 IAttackComponent attackComponent;
316 switch(data.mode)
317 {
318 case LaunchAttackTypeMode.specific:
319 attackComponent = data.source.AttackComponentsDic[data.attackTypeCode];
320 break;
321 // Use the first active attack component by default
322 default:
323 attackComponent = data.source.FirstActiveAttackComponent;
324 break;
325 }
326
327
328 ErrorMessage errorMsg;
329 //check whether the new target is valid for this attack type.
330 if ((errorMsg = attackComponent.IsTargetValid(data.ToSetTargetInputData())) != ErrorMessage.none)
331 {
332 if (data.playerCommand && data.source.IsLocalPlayerFaction())
333 playerMsgHandler.OnErrorMessage(new PlayerErrorMessageWrapper
334 {
335 message = errorMsg,
336
337 source = data.source,
338 target = data.targetEntity
339 });
340 return errorMsg;
341 }
342
343 // If the attack order was issued by the local player and this is the local player's instance.
344 if (data.playerCommand && data.source.IsLocalPlayerFaction())
345 {
346 if (!data.targetEntity.IsValid())
347 effectObjPool.Spawn(TerrainAttackTargetEffect, data.targetPosition);
348
349 audioMgr.PlaySFX(attackComponent.OrderAudio, false);
350 }
351
352 // Calculate a target attack position and attempt to set a new attack target for the source unit.
353 return attackComponent.SetTargetLocal(
354 new SetTargetInputData
355 {
356 target = new TargetData<IFactionEntity>
357 {
358 instance = data.targetEntity,
359 opPosition = data.targetPosition,
360
361 position = data.source.CanMove() && TryGetAttackPosition(data.source, attackComponent, data.targetEntity, data.targetPosition, data.playerCommand, out Vector3 attackPosition)
362 ? attackPosition
363 : data.source.transform.position
364 },
365
366 playerCommand = data.playerCommand,
367
368 isMoveAttackRequest = data.isMoveAttackRequest,
369 ignoreLOS = data.ignoreLOS,
370 });
371 }

374 #region Generating Attack Position
375 public bool TryGetAttackPosition(IEntity attacker, IAttackComponent attackComponent, IFactionEntity target, Vector3 targetPosition, bool playerCommand, out Vector3 attackPosition)
376 {
377 attackPosition = Vector3.positiveInfinity;
378
379
380 if (!attacker.IsValid() || !attacker.CanAttack)
381 {
382 logger.LogError($"[AttackManager - {attacker.Code}] Can not calculate an attack position with an invalid entity instance or a non attack entity!");
383 return false;
384 }
385
386 // Generate movement attack path destination for the new target
387 mvtMgr.GeneratePathDestination(
388 attacker,
389 targetPosition,
390 attacker.MovementComponent.Formation,
391 attackComponent.Formation.GetStoppingDistance(target, min: true),
392 new MovementSource { playerCommand = playerCommand },
393 ref pathDestinations,
394 condition: RTSHelper.IsAttackLOSBlocked);
395
396 // If there's a valid attack movement destination produced, get the closest target position
397 if (pathDestinations.Count == 0)
398 return false;
399
400 attackPosition = pathDestinations.OrderBy(pos => (pos - attacker.transform.position).sqrMagnitude).First();
401 return true;
402 }
