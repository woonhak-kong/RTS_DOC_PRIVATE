62 public void RaiseEntitySelectedGlobal(IEntity entity, EntitySelectionEventArgs args)
63 {
64 var handler = EntitySelectedGlobal;
65 handler?.Invoke(entity, args);
66
67 CustomEventHandler<IEntity, EntitySelectionEventArgs> specificHandler = null;
68 if (entity.IsUnit())
69 specificHandler = selectedEvents[EntityType.unit];
70 else if (entity.IsBuilding())
71 specificHandler = selectedEvents[EntityType.building];
72 else if (entity.IsResource())
73 specificHandler = selectedEvents[EntityType.resource];
74
75 specificHandler?.Invoke(entity, args);
76 }
