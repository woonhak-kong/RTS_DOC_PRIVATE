33 protected override void OnPendingInit()
34 {
35 if (!logger.RequireTrue(upgradeTasks.All(task => task.PrefabObject.IsValid()),
36 $"[{GetType().Name} - {Entity.Code}] Some elements in the 'Upgrade Tasks' array have the 'Prefab Object' field unassigned!")
37
38 || !logger.RequireTrue(entityTargetUpgradeTasks.All(task => task.PrefabObject.IsValid()),
39 $"[{GetType().Name} - {Entity.Code}] Some elements in the 'Upgrade Target Upgrade Tasks' array have the 'Prefab Object' field unassigned!"))
40 return;
41
42 this.entityUpgradeMgr = gameMgr.GetService<IEntityUpgradeManager>();
43 this.entityCompUpgradeMgr = gameMgr.GetService<IEntityComponentUpgradeManager>();
44
45 // Initialize upgrade tasks
46 int taskID = 0;
47 allUpgradeTasks = new List<UpgradeTask>();
48 for (taskID = 0; taskID < upgradeTasks.Count; taskID++)
49 {
50 upgradeTasks[taskID].Init(this, taskID, gameMgr);
51 upgradeTasks[taskID].Enable();
52 }
53 allUpgradeTasks.AddRange(upgradeTasks);
54
55 foreach(var nextTask in entityTargetUpgradeTasks)
56 {
57 nextTask.Init(this, taskID, gameMgr);
58 taskID++;
59 }
60 allUpgradeTasks.AddRange(entityTargetUpgradeTasks);
61
62 foreach(var nextTask in entityComponentTargetUpgradeTasks)
63 {
64 nextTask.Init(this, taskID, gameMgr);
65 taskID++;
66 }
67 allUpgradeTasks.AddRange(entityComponentTargetUpgradeTasks);
68
69 if (!Entity.IsFree)
70 {
71 // Divide upgrade tasks into entity upgrades (group with key = true) and entity component upgrades (group with key = false)
72 var upgradeTaskGroups = upgradeTasks
73 .GroupBy(task => task.Prefab is EntityUpgrade);
74
75 var entityUpgradeTasks = upgradeTaskGroups
76 .Where(group => group.Key == true)
77 .SelectMany(group => group);
78
79 // Check for already launched entity upgrades:
80 if (entityUpgradeMgr.TryGet(Entity.FactionID, out UpgradeElement<IEntity>[] upgradedEntityElements))
81 {
82 foreach(var nextElement in upgradedEntityElements)
83 {
84 DisableTasksWithPrefabCode(nextElement.sourceCode, isEntityUpgrade: true);
85 EnableUpgradeTargetTasksWithPrefab(nextElement.target.Code, isEntityUpgrade: true);
86 }
87 }
88
89 // Get the EntityComponentUpgrade tasks
90 var entityComponentUpgradeTasks = upgradeTaskGroups
91 .Where(group => group.Key == false)
92 .SelectMany(group => group)
93 .ToList();
94
95 // Check for already launched entity component upgrades
96 // Go through each EntityComponentUpgrade in the above group
97 foreach (EntityComponentUpgrade componentUpgrade in entityComponentUpgradeTasks
98 .Where(task => (task.Prefab is EntityComponentUpgrade))
99 .Select(task => task.Prefab as EntityComponentUpgrade)
100 .ToArray())
101 {
102 // For each instance, check if the upgrades have been already launched
103 if (entityCompUpgradeMgr.TryGet(componentUpgrade.SourceEntity, Entity.FactionID, out List<UpgradeElement<IEntityComponent>> upgradedComponentElements))
104 {
105 foreach(var nextElement in upgradedComponentElements)
106 {
107 DisableTasksWithPrefabCode(nextElement.sourceCode, isEntityUpgrade: false);
108 EnableUpgradeTargetTasksWithPrefab(nextElement.target.Code, isEntityUpgrade: false);
109 }
110 }
111 }
112 }

124 private void DisableTasksWithPrefabCode(string prefabCode, bool isEntityUpgrade)
125 {
126 // Disable upgraded tasks
127 foreach (var task in upgradeTasks)
128 {
129 bool condition;
130 if (isEntityUpgrade)
131 {
132 EntityUpgrade entityUpgrade = (task.Prefab as EntityUpgrade);
133 condition = entityUpgrade.IsValid() && entityUpgrade.SourceCode == prefabCode;
134 }
135 else
136 {
137 EntityComponentUpgrade entityComponentUpgrade = (task.Prefab as EntityComponentUpgrade);
138 condition = entityComponentUpgrade.IsValid() && entityComponentUpgrade.GetUpgrade(task.UpgradeIndex).GetSourceCode(task.Prefab.SourceEntity) == prefabCode;
139 }
140
141 if(condition)
142 {
143 task.Disable();
144 // If there are pending tasks that use the upgraded entity then cancel them.
145 Entity.PendingTasksHandler.CancelBySourceID(this, task.ID);
146 }
147 }
148 }
149
150 private void EnableUpgradeTargetTasksWithPrefab(string targetCode, bool isEntityUpgrade)
151 {
152 foreach (var upgradeTargetTask in isEntityUpgrade ? entityTargetUpgradeTasks : entityComponentTargetUpgradeTasks)
153 {
154 string nextCode;
155 if (isEntityUpgrade)
156 {
157 nextCode = (upgradeTargetTask.Prefab as EntityUpgrade).SourceCode;
158 }
159 else
160 {
161 nextCode = (upgradeTargetTask.Prefab as EntityComponentUpgrade)
162 .GetUpgrade(upgradeTargetTask.UpgradeIndex)
163 .GetSourceCode(upgradeTargetTask.Prefab.SourceEntity);
164 }
165
166 if (nextCode == targetCode)
167 {
168 upgradeTasks.Add(upgradeTargetTask);
169 upgradeTargetTask.Enable();
170 }
171 }
