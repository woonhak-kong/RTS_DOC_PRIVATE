190 #region Caching Terrain Height Values
191 private IEnumerator CacheHeightValues()
192 {
193 if (heightCacheDistance < 0)
194 {
195 logger.LogError($"[{GetType().Name}] The height cache movement distance (field: 'Height Cache Distance') must be > 0.");
196 yield return null;
197 }
198
199 heightCacheDict = new Dictionary<string, Dictionary<Int2D, float>>();
200
201 foreach (TerrainAreaType areaType in areas)
202 {
203 heightCacheDict.Add(areaType.Key, new Dictionary<Int2D, float>());
204
205 for (int x = lowerLeftCorner.x; x < upperRightCorner.x; x += heightCacheDistance)
206 for (int y = lowerLeftCorner.y; y < upperRightCorner.y; y += heightCacheDistance)
207 {
208 // Each search cell instance is added to the dictionary after it is created for easier direct access using coordinates in the future.
209 Int2D nextPosition = new Int2D
210 {
211 x = x,
212 y = y
213 };
214
215 if(GetTerrainAreaPosition(
216 new Vector3(nextPosition.x, areaType.BaseHeight + heightCacheSampleOffset, nextPosition.y),
217 areaType.Key,
218 out Vector3 outPosition))
219 heightCacheDict[areaType.Key].Add(nextPosition, outPosition.y);
220 }
221 }
222
223 yield return null;
224 }

321 public TerrainAreaMask TerrainAreasToMask (IReadOnlyList<TerrainAreaType> areaTypes)
322 {
323 if (areaTypes.Count == 0)
324 return new TerrainAreaMask(~0); // All terrain area types.

326 TerrainAreaMask mask = new TerrainAreaMask(); // No terrain areas assigned initially.
327
328 // Add each area type index to the mask

330 mask.Add(AreaTypeToLayer[areaTypes[i]]);

378 public bool GetTerrainAreaPosition(Vector3 inPosition, IReadOnlyList<TerrainAreaType> areaTypes, out Vector3 outPosition)
379 {
380 outPosition = inPosition;
381
382 if (!areaTypes.IsValid() || areaTypes.Count == 0)
383 return true;
384
385 float closestHeightDistance = Mathf.Infinity;
386 float nextDistance;
387 bool hasFoundPosition = false;
388
389 for (int i = 0; i < areaTypes.Count; i++)
390 {
391 inPosition.y += areaTypes[i].TestHeightOffset;
392
393 if (GetTerrainAreaPosition(inPosition, areaTypes[i].Key, out Vector3 nextPosition))
394 {
395 hasFoundPosition = true;
396 if ((nextDistance = (nextPosition - inPosition).sqrMagnitude) < closestHeightDistance)
397 {
398 closestHeightDistance = nextDistance;
399 outPosition = nextPosition;
400 }
401 }
402 }
403
404 if(hasFoundPosition)
405 return true;
406
407 return false;
408 }

413 public bool GetTerrainAreaPosition(Vector3 inPosition, string areaKey, out Vector3 outPosition)
414 {
415 outPosition = inPosition;
416
417 if (!areasDic.TryGetValue(areaKey, out TerrainAreaType areaType))
418 {
419 logger.LogError($"[TerrainManager] The input area key: {areaKey} has not been registered for this map!");
420 return false;
421 }
422
423 inPosition.y += areaType.TestHeightOffset;
424
425 // Create a ray that goes down vertically and attempt to find the terrain area.
426 // Use JobSystem to shoot the rays into threads
427
428 NativeArray<RaycastHit> results = new NativeArray<RaycastHit>(1, Allocator.TempJob);
429 NativeArray<RaycastCommand> commands = new NativeArray<RaycastCommand>(1, Allocator.TempJob);
430
431 commands[0] = new RaycastCommand(inPosition, Vector3.down, Mathf.Infinity, areaType.Layers);
432
433 JobHandle handle = RaycastCommand.ScheduleBatch(commands, results, minCommandsPerJob: 1, dependsOn: default(JobHandle));
434 handle.Complete();
435
436 RaycastHit hit = results[0];
437
438 results.Dispose();
439 commands.Dispose();
440
441 if(hit.collider.IsValid())
442 {
443 outPosition = hit.point;
444 return true;
445 }
446
447 return false;
448 }

452 public bool ScreenPointToTerrainPoint(Vector3 screenPoint, IReadOnlyList<TerrainAreaType> areaTypes, out Vector3 terrainPoint)
453 {
454 int layers = 0;
455
456 IReadOnlyList<TerrainAreaType> nextAreaTypes = (areaTypes.IsValid() && areaTypes.Count > 0) ? areaTypes : areas;
457 for (int i = 0; i < nextAreaTypes.Count; i++)
458 layers |= nextAreaTypes[i].Layers;
459
460 if (Physics.Raycast(mainCameraController.MainCamera.ScreenPointToRay(screenPoint), out RaycastHit hit, Mathf.Infinity, layers))
461 {
462 terrainPoint = hit.point;
463 return true;
464 }
465
466 terrainPoint = Vector3.zero;
467 return false;
468 }
