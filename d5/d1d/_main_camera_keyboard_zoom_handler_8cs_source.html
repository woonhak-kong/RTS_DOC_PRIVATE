47
48 if (placementMgr.IsLocalPlayerPlacingBuilding && !allowBuildingPlaceZoom)
49 return;
50
51 // Camera zoom on keys
52 if (keyZoom.enabled)
53 {
54 if (controls.Get(keyZoom.inKey))
55 zoomValue -= Time.deltaTime;
56 else if (controls.Get(keyZoom.outKey))
57 zoomValue += Time.deltaTime;
58 }
59
60 // Camera zoom when the player is moving the mouse scroll wheel
61 if (mouseWheelZoom.enabled)
62 zoomValue += Input.GetAxis("Mouse ScrollWheel") * mouseWheelZoom.sensitivity
63 * (mouseWheelZoom.invert ? -1.0f : 1.0f) * Time.deltaTime;
64 }
