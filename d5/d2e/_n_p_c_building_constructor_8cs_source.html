86 private void ActivateBuilderRegulator()
87 {
88 builderMonitor = new NPCActiveRegulatorMonitor(gameMgr, factionMgr);
89
90 buildingToBuilderCodes = new Dictionary<string, List<string>>();
91
92 for (int i = 0; i < builders.Length; i++)
93 {
94 IUnit builder = builders[i].IsValid() ? builders[i].GetComponent<IUnit>() : null;
95 if (!logger.RequireValid(builder,
96 $"[{GetType().Name} - {factionMgr.FactionID}] 'Builders' field has some unassigned elements."))
97 return;
98
99 IBuilder builderComponent = builder.gameObject.GetComponentInChildren<IBuilder>();
100 if (!logger.RequireValid(builderComponent,
101 $"[{GetType().Name} - {factionMgr.FactionID}] 'Builders' field has some assigned units with no component that implements '{typeof(IBuilder).Name}' interface attached to them."))
102 return;
103
104 NPCUnitRegulator nextRegulator;
105 if ((nextRegulator = npcUnitCreator.ActivateUnitRegulator(builder)).IsValid())
106 {
107 builderMonitor.AddCode(nextRegulator.Prefab.Code);
108
109 // For each valid builder that gets added, get their available creation tasks.
110 List<BuildingCreationTask> availableCreationTasks = new List<BuildingCreationTask>(builderComponent.CreationTasks);
111 availableCreationTasks.AddRange(builderComponent.UpgradeTargetCreationTasks);
112 for (int j = 0; j < availableCreationTasks.Count; j++)
113 {
114 string buildingCode = availableCreationTasks[j].Prefab.Code;
115
116 if (!buildingToBuilderCodes.ContainsKey(buildingCode))
117 buildingToBuilderCodes.Add(buildingCode, new List<string>());
118
119 if (!buildingToBuilderCodes[buildingCode].Contains(builder.Code))
120 {
121 buildingToBuilderCodes[buildingCode].Add(builder.Code);
122 }
123 }
124 }
125 }
126
127 if (builderMonitor.Count == 0)
128 logger.LogWarning($"[{GetType().Name} - {factionMgr.FactionID}] No builder regulators have been assigned!");
129 }

186 protected override void OnActiveUpdate()
187 {
188 //checking buildings timer:
189 if (constructionTimer.ModifiedDecrease())
190 {
191 constructionTimer.Reload(constructionTimerRange);
192
193 constructionTimerTicks++;
194
195 bool canEnforceBuilders = constructionTimerTicks == enforceBuildersTimerTicks;
196 if (canEnforceBuilders)
197 constructionTimerTicks = 0;
198
199 // Only keep this component active if there are buildings that need construction
200 IsActive = buildingsToConstruct.Count > 0;
201
202 for (int i = 0; i < buildingsToConstruct.Count; i++)
203 {
204 int targetBuildersAmount = GetTargetBuildersAmount(buildingsToConstruct[i]);
205
206 if (targetBuildersAmount > buildingsToConstruct[i].WorkerMgr.Amount)
207 OnBuildingConstructionRequestInternal(
208 buildingsToConstruct[i],
209 targetBuildersAmount,
210 out _,
211 forceSwitch: canEnforceBuilders);
212
213 }
214 }

223 public void OnBuildingConstructionRequest(IBuilding building, int targetBuildersAmount, out int assignedBuilders, bool forceSwitch = false)
224 {
225 assignedBuilders = 0;
226
227 if (!building.IsValid())
228 return;
229
230 if (!constructOnDemand)
231 {
232 LogEvent($"{building.Code}: Unable to construct building on demand!");
233 return;
234 }

239 private void OnBuildingConstructionRequestInternal(IBuilding building, int targetBuildersAmount, out int assignedBuilders, bool forceSwitch = false)
240 {
241 assignedBuilders = 0;
242
243 if (!building.IsValid())
244 return;
245
246 if (building.Health.IsDead
247 || building.Health.CurrHealth >= building.Health.MaxHealth)
248 {
249 LogEvent($"{building.Code}: Unable to construct building! Building Dead? {building.Health.IsDead} - Max Health Reached? {building.Health.CurrHealth >= building.Health.MaxHealth}");
250 return;
251 }
252
253 int requiredBuilders = targetBuildersAmount - building.WorkerMgr.Amount;
254
255 var nextBuilderRegulator = npcUnitCreator
256 .GetActiveUnitRegulator(builderMonitor.RandomCode);
257
258 if (!nextBuilderRegulator.IsValid())
259 {
260 LogEvent($"{building.Code}: Unable to find valid unit regulator for builder unit!");
261 return;
262 }
263
264 for (int i = 0; i < nextBuilderRegulator.InstancesIdleFirst.Count; i++)
265 {
266 IUnit nextBuilder = nextBuilderRegulator.InstancesIdleFirst[i];
267
268 bool canStillEnforceSwitch = forceSwitch
269 && !nextBuilder.BuilderComponent.HasTarget;
270
271 if (nextBuilder.IsValid()
272 && (nextBuilder.IsIdle || canStillEnforceSwitch)
273 && nextBuilder.BuilderComponent.Target.instance != building)
274 {
275 nextBuilder.BuilderComponent.SetTarget(building.ToTargetData(), playerCommand: false);
276
277 requiredBuilders--;
278 assignedBuilders++;
279
280 if (requiredBuilders <= 0)
281 break;
282 }
283 }
284
285 LogEvent($"{building.Code}: Sent {assignedBuilders}/{assignedBuilders+requiredBuilders} builders to construct building!");
286 }
