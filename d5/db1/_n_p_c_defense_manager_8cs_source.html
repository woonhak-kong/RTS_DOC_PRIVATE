87 private void HandleFactionEntityHealthUpdated(IFactionEntity factionEntity, HealthUpdateArgs args)
88 {
89 // Only consider faction entities owned by the NPC faction who has been damaged by an enemy faction entity
90 if (!factionMgr.IsSameFaction(factionEntity)
91 || args.Value >= 0.0f
92 || !args.Source.IsValid()
93 || args.Source.IsFriendlyFaction(factionSlot))
94 return;
95
96 OnUnitSupportRequest(factionEntity.transform.position, args.Source as IFactionEntity);
97
98 foreach (IBuilding nextBuildingCenter in factionMgr.BuildingCenters)
99 if (nextBuildingCenter.BorderComponent.IsInBorder(factionEntity.transform.position))
100 {
101 LaunchDefense(nextBuildingCenter, forceUpdateDefenseCenter: false);
102 break;
103 }
104 }

120 // "forceUpdateDefenseCenter", when false, only units who do not have an active attack target will have their defense center forced
121 public void LaunchDefense(IBuilding nextDefenseCenter, bool forceUpdateDefenseCenter)
122 {
123 if (!canDefendTerritory
124 || !nextDefenseCenter.IsValid()
125 || !nextDefenseCenter.BorderComponent.IsValid())
126 return;
127
128 LogEvent($"{nextDefenseCenter.Code}: Enabled territory defense on building center");
129 IsDefending = true;
130
131 // Keep reloading the cancel defense timer until no calls to launch a defense happen
132 cancelTerritoryDefenseTimer.Reload(cancelTerritoryDefenseReloadRange);
133
134 if (cancelAttackOnTerritoryDefense && npcAttackMgr.IsAttacking)
135 npcAttackMgr.CancelAttack();
136
137 npcEvent.RaiseTerritoryDefenseOrder(
138 this,
139 new NPCTerritoryDefenseEngageEventArgs(
140 LastDefenseCenter,
141 nextDefenseCenter,
142 defenseRange: nextDefenseCenter.BorderComponent.IsValid() ? nextDefenseCenter.BorderComponent.Size : 0.0f));
143
144 LastDefenseCenter = nextDefenseCenter;
145 }
146

159 #region Handling Unit Support
160 public bool OnUnitSupportRequest(Vector3 supportPosition, IFactionEntity target)
161 {
162 if (!unitSupportEnabled
163 || !target.IsValid()
164 || target.Health.IsDead)
165 return false;
166
167 gridSearch.Search(
168 supportPosition,
169 unitSupportRange.RandomValue,
170 amount: -1, // negative value gets all potential units
171 IsValidUnitSupport,
172 playerCommand: false,
173 out IReadOnlyList<IUnit> supportUnits);
174
175 if (supportUnits.Count > 0)
176 {
177 LogEvent($"Enabled unit support on position {supportPosition} with {supportUnits.Count} units!");
178
179 attackMgr.LaunchAttack(new LaunchAttackData<IReadOnlyList<IEntity>>
180 {
181 source = supportUnits,
182 targetEntity = target,
183 targetPosition = target.transform.position,
184 playerCommand = false
185 });
186 }
187
188 return true;
189 }

191 private ErrorMessage IsValidUnitSupport(SetTargetInputData data)
192 {
193 if (!data.target.instance.IsValid()
194 || !data.target.instance.IsUnit()
195 || !factionMgr.IsSameFaction(data.target.instance)
196 || !data.target.instance.CanAttack)
197 return ErrorMessage.invalid;
198 // Make sure that the unit to test has a target that can not attack back so that it can switch to support
199 else if (data.target.instance.FirstActiveAttackComponent.HasTarget && data.target.instance.FirstActiveAttackComponent.Target.instance.CanAttack)
200 return ErrorMessage.attackTargetNoChange;
201
202 return ErrorMessage.none;
203 }
