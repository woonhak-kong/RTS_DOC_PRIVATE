108 public static void RefreshAssetFiles (bool requireTest, RTSEngineScriptableObject testInstance)
109 {
110 //Type nextType = testInstance ? testInstance.GetType() : typeof(RTSEngineScriptableObject);
111 Type nextType = typeof(RTSEngineScriptableObject);
112 if(!requireTest
113 || AssetFiles == null
114 || !AssetFiles.TryGetValue(nextType, out IEnumerable<RTSEngineScriptableObject> cache)
115 || !cache.Contains(testInstance))
116 {
117 var newAssetFiles = new Dictionary<Type, IEnumerable<RTSEngineScriptableObject>>();
118
119 if(TryGetAllAssetFiles(out List<RTSEngineScriptableObject> assets, filter: $"t:{nextType.ToString()}"))
120 {
121 var groupedAssets = assets
122 .GroupBy(asset => asset.GetType());
123
124 foreach(var group in groupedAssets)
125 {
126 if (newAssetFiles.ContainsKey(group.Key))
127 newAssetFiles[group.Key] = group;
128 else
129 newAssetFiles.Add(group.Key, group);
130 }
131 }
132
133 AssetFiles = newAssetFiles;
134 }
135 }

137 private static bool TryGetAllAssetFiles <T>(out List<T> assets, string filter = "DefaultAsset l:noLabel t:noType") where T : ScriptableObject
138 {
139 assets = new List<T> { };
140 string[] guids = AssetDatabase.FindAssets(filter);
141
142 if (guids.Length > 0)
143 {
144 foreach (string guid in guids)
145 {
146 string assetPath = AssetDatabase.GUIDToAssetPath(guid);
147 assets.Add(AssetDatabase.LoadAssetAtPath(assetPath, typeof(T)) as T);
148 }
149
150 return true;
151 }
152 else
153 return false;
154 }

156 public static bool GetAssetFilesDictionary <T> (out Dictionary<string, T> resultDic) where T : RTSEngineScriptableObject
157 {
158 resultDic = new Dictionary<string, T>();
159
160 resultDic.Add("Unassigned", null);
161
162 if (AssetFiles == null)
163 ReloadRTSPrefabsAndAssetFiles();
164
165 AssetFiles.TryGetValue(typeof(T), out IEnumerable<RTSEngineScriptableObject> cached);
166
167 if (cached == null)
168 return true;
169
170 foreach(T t in cached)
171 {
172 if (t == null)
173 continue;
174
175 if (resultDic.ContainsKey(t.Key))
176 {
177 Debug.LogError($"[RTSEditorHelper] '{t.Key}' is a duplicate key for the '{typeof(T).ToString()}' type in '{t.name}' and '{resultDic[t.Key].name}' asset files.", t);
178 return false;
179 }
180
181 resultDic.Add(t.Key, t);
182 }
183
184 return true;
185 }

187 public static string[] GetMatchingStrings (string searchInput, string[] searchTargets, string[] exceptions, System.Func<string, bool> condition = null)
188 {
189 List<string> matches = new List<string>();
190
191 string sPattern = $"{searchInput}";
192
193 foreach (string s in searchTargets)
194 {
195 if (!string.IsNullOrEmpty(s)
196 && !exceptions.Any(exception => exception == s)
197 && System.Text.RegularExpressions.Regex.IsMatch(s, sPattern, System.Text.RegularExpressions.RegexOptions.IgnoreCase)
198 && (condition == null || condition.Invoke(s)))
199 matches.Add(s);
200 }
201
202 return matches
203 .OrderBy(match => match)
204 .ToArray();
205 }

229 public static void SetEntities ()
230 {
231 bool allValid = true;
232
233 IEnumerable<IEntity> entityPrefabs = UnityEngine.Resources.LoadAll("Prefabs", typeof(GameObject))
234 .Cast<GameObject>()
235 .Where(obj => obj.GetComponent<IEntity>() != null)
236 .Select(obj => obj.GetComponent<IEntity>());
237
238 Entities = new Dictionary<string, IEntity>();
239 EntitiesPerCategory = new Dictionary<string, IEnumerable<IEntity>>();
240
241 foreach(IEntity entity in entityPrefabs)
242 {
243 if (GetEntities().ContainsKey(entity.Code))
244 {
245 Debug.LogError($"[RTSEditorHelper] Failed to cache entity prefab {entity.gameObject.name}: Entity code '{entity.Code}' has been already used on another entity prefab (Prefab name: '{GetEntities()[entity.Code].gameObject.name}')", entity.gameObject);
246 allValid = false;
247 }
248 else
249 GetEntities().Add(entity.Code, entity);
250
251 foreach (string category in entity.Category)
252 {
253 if (GetEntitiesPerCategory().ContainsKey(category))
254 GetEntitiesPerCategory()[category] = GetEntitiesPerCategory()[category].Append(entity);
255 else
256 GetEntitiesPerCategory().Add(category, Enumerable.Repeat(entity, 1));
257 }
258 }
259
260 if (allValid)
261 Debug.Log("[RTSEditorHelper] Cached entity prefabs placed in a path that ends with '*/Resources/Prefabs'. All cached entities passed the validity tests.");
262 else
263 Debug.LogError("[RTSEditorHelper] Cached entity prefabs placed in a path that ends with '*/Resources/Prefabs' are not all valid. Please fix the above errors!");
264 }
