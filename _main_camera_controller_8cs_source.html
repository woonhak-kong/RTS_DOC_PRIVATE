63 this.terrainMgr = gameMgr.GetService<ITerrainManager>();
64
65 if (!mainCamera.IsValid())
66 {
67 logger.LogError($"[{GetType().Name}] The field 'Main Camera' must be assigned!");
68 return;
69 }
70
71 controlHandlers = gameObject.transform.root.GetComponentsInChildren<IMainCameraControlHandler>();
72 foreach (var handler in controlHandlers)
73 {
74 if (handler is IMainCameraPanningHandler)
75 PanningHandler = handler as IMainCameraPanningHandler;
76 else if (handler is IMainCameraRotationHandler)
77 RotationHandler = handler as IMainCameraRotationHandler;
78 else if (handler is IMainCameraZoomHandler)
79 ZoomHandler = handler as IMainCameraZoomHandler;
80 }
81
82 if(!PanningHandler.IsValid())
83 {
84 logger.LogError($"[{GetType().Name}] A component that handles camera panning that implements the interface '{typeof(IMainCameraPanningHandler).Name}' must be attached to the same game object!");
85 return;
86 }
87 else if(!ZoomHandler.IsValid())
88 {
89 logger.LogError($"[{GetType().Name}] A component that handles camera zooming that implements the interface '{typeof(IMainCameraZoomHandler).Name}' must be attached to the same game object!");
90 return;
91 }
92 else if(!RotationHandler.IsValid())
93 {
94 logger.LogError($"[{GetType().Name}] A component that handles camera rotation that implements the interface '{typeof(IMainCameraRotationHandler).Name}' must be attached to the same game object!");
95 return;
96 }
97
98 foreach (var handler in controlHandlers)
99 handler.Init(gameMgr);

130 public Vector3 ScreenToWorldPoint(Vector3 position, bool applyOffset = true)
131 {
132 position.z = mainCamera.transform.position.y;
133 Vector3 worldPosition;
134 // First try with a raycast to hit the base terrain bottom collider and if that is not successful then use the ScreenToWorldPoint
135 // The raycast would guarantee getting a position that is on level with the base map terrain level
136 // While ScreenToWorldPoint will often not return that and therefore and is used as a backup solution for this method
137 if (Physics.Raycast(ScreenPointToRay(position), out RaycastHit hit, Mathf.Infinity, terrainMgr.BaseTerrainLayerMask))
138 worldPosition = hit.point;
139 else
140 worldPosition = mainCamera.ScreenToWorldPoint(position);
141
142 return applyOffset
143 ? new Vector3(worldPosition.x - PanningHandler.CurrOffsetX, worldPosition.y, worldPosition.z - PanningHandler.CurrOffsetZ)
144 : worldPosition;
145 }
